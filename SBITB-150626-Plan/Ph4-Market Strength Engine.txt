    @model_validator(mode="after")
    def validate_weight_sum(self) -> "MarketStrengthSettings":
        """Feature weights MUST sum to 1.0 (Kaufman Ch.7 normalization)."""
        total = (
            self.WEIGHT_AD_RATIO
            + self.WEIGHT_VWAP_DISTANCE
            + self.WEIGHT_OI_CHANGE
            + self.WEIGHT_PCR
            + self.WEIGHT_VOLUME_PROFILE
            + self.WEIGHT_MARKET_BREADTH
            + self.WEIGHT_SECTOR_MOMENTUM
            + self.WEIGHT_INDIA_VIX
            + self.WEIGHT_FII_DII_FLOW
        )
        if not (0.99 <= total <= 1.01):
            raise ValueError(f"Feature weights must sum to 1.0, got {total:.4f}")
        if self.WEAK_THRESHOLD >= self.STRONG_THRESHOLD:
            raise ValueError(
                f"WEAK_THRESHOLD ({self.WEAK_THRESHOLD}) must be < STRONG_THRESHOLD ({self.STRONG_THRESHOLD})"
            )
        return self

from pydantic import BaseModel, Field
from enum import Enum
from datetime import datetime, date


class MarketRegime(str, Enum):
    """Market strength regime classification.

    WEAK (0-30): Bearish regime. Selling strategies dominate. Short-biased.
    NEUTRAL (31-60): Range-bound. Options selling strategies. Premium collection.
    STRONG (61-100): Bullish regime. Buying strategies dominate. Long-biased.

    Reference: Plan.txt line 369-371; Dalton Ch.6 (market structure phases)
    """
    WEAK = "WEAK"
    NEUTRAL = "NEUTRAL"
    STRONG = "STRONG"


class VolumeProfilePosition(str, Enum):
    """Volume profile position relative to value area.

    Direct consumption from Phase 3 VolumeProfileResult.relative_position.
    Phase 4 does NOT recompute volume profile — it reads Phase 3 output.

    Reference: Dalton Ch.6 (value area as fair price)
    """
    CURRENT_ABOVE_VA = "CURRENT_ABOVE_VA"
    CURRENT_IN_VA = "CURRENT_IN_VA"
    CURRENT_BELOW_VA = "CURRENT_BELOW_VA"
    CURRENT_AT_POC = "CURRENT_AT_POC"
    UNKNOWN = "UNKNOWN"


class VIXTrend(str, Enum):
    """India VIX trend direction.

    Reference: Weis Ch.3 (fear/complacency cycles)
    """
    RISING = "RISING"
    FALLING = "FALLING"
    STABLE = "STABLE"
    UNKNOWN = "UNKNOWN"


class FeatureAvailability(BaseModel):
    """Tracks which features are available vs unavailable.

    When a feature source is unavailable (API down, data gap), the feature gets
    a neutral score (50.0) and its weight is redistributed proportionally to
    available features. This ensures the composite score is always computable.

    Reference: Kleppmann Ch.1 (fault tolerance); Stability principle
    """
    ad_ratio: bool = Field(True, description="Advance/Decline ratio available")
    vwap_distance: bool = Field(True, description="VWAP distance available")
    oi_change: bool = Field(True, description="Open Interest change available")
    pcr: bool = Field(True, description="Put/Call ratio available")
    volume_profile_position: bool = Field(True, description="Volume profile position available")
    market_breadth: bool = Field(True, description="Market breadth available")
    sector_momentum: bool = Field(True, description="Sector momentum available")
    india_vix: bool = Field(True, description="India VIX available")
    fii_dii_flow: bool = Field(True, description="FII/DII flow available")


class FeatureScore(BaseModel):
    """Individual feature normalized score [0, 100] and raw value."""
    score: float = Field(ge=0.0, le=100.0, description="Normalized score [0, 100]")
    raw_value: float | None = Field(None, description="Original raw value before normalization")
    direction: str = Field("NEUTRAL", description="BULLISH, BEARISH, or NEUTRAL interpretation")
    available: bool = Field(True, description="Whether the feature data was available")


class FeaturesBreakdown(BaseModel):
    """Detailed breakdown of all 9 feature scores.

    This is the features_breakdown field in MarketStrength output.
    Allows downstream consumers (Phase 5, Phase 7, Phase 11) to understand
    WHY the composite score is what it is.

    Reference: Dalton Ch.6 (transparency in market structure assessment)
    """
    ad_ratio: FeatureScore = Field(default_factory=lambda: FeatureScore(score=50.0, raw_value=None, direction="NEUTRAL", available=False))
    vwap_distance: FeatureScore = Field(default_factory=lambda: FeatureScore(score=50.0, raw_value=None, direction="NEUTRAL", available=False))
    oi_change: FeatureScore = Field(default_factory=lambda: FeatureScore(score=50.0, raw_value=None, direction="NEUTRAL", available=False))
    pcr: FeatureScore = Field(default_factory=lambda: FeatureScore(score=50.0, raw_value=None, direction="NEUTRAL", available=False))
    volume_profile_position: FeatureScore = Field(default_factory=lambda: FeatureScore(score=50.0, raw_value=None, direction="NEUTRAL", available=False))
    market_breadth: FeatureScore = Field(default_factory=lambda: FeatureScore(score=50.0, raw_value=None, direction="NEUTRAL", available=False))
    sector_momentum: FeatureScore = Field(default_factory=lambda: FeatureScore(score=50.0, raw_value=None, direction="NEUTRAL", available=False))
    india_vix: FeatureScore = Field(default_factory=lambda: FeatureScore(score=50.0, raw_value=None, direction="NEUTRAL", available=False))
    fii_dii_flow: FeatureScore = Field(default_factory=lambda: FeatureScore(score=50.0, raw_value=None, direction="NEUTRAL", available=False))
    effective_weights: dict[str, float] = Field(default_factory=dict, description="Weights actually used after redistribution for unavailable features")


class MarketStrength(BaseModel):
    """Composite market strength score 0-100 with regime classification.

    This is the PRIMARY output of Phase 4:
        compute_market_strength(market_data) -> MarketStrength

    Downstream consumers:
    - Phase 5: generate_signal() uses regime for strategy selection
    - Phase 7: compute_trailing_stop() uses score for trail tightness
    - Phase 11: Orchestrator weights Market Strength at 15%

    Reference: Dalton Ch.6 (market structure scoring); Kaufman Ch.7 (composite signals)
    """
    score: float = Field(ge=0.0, le=100.0, description="Composite market strength score [0, 100]")
    regime: MarketRegime = Field(MarketRegime.NEUTRAL, description="Regime classification")
    features_breakdown: FeaturesBreakdown = Field(default_factory=FeaturesBreakdown)
    as_of_date: date = Field(description="Date for which this score was computed (explicit, never today())")
    computed_at: datetime = Field(default_factory=datetime.now, description="Computation timestamp")
    processing_time_ms: float = Field(0.0, description="Computation time in ms")

    price_direction: int | None = Field(None, ge=-1, le=1, description="+1=price up, -1=price down, 0=flat")
    pcr: float | None = Field(None, ge=0.0, description="Put/Call ratio (Total PE OI / Total CE OI)")

def normalize_ad_ratio(ad_ratio: float | None, settings: MarketStrengthSettings) -> FeatureScore:
    """Normalize Advance/Decline ratio to [0, 100] score.

    A/D ratio = Advances / Declines
    - ad_ratio > 1.0: More advances than declines (bullish)
    - ad_ratio < 1.0: More declines than advances (bearish)
    - ad_ratio = 1.0: Balanced (neutral)

    LOG-SCALE normalization (Correction #15):
    A/D ratio is multiplicative — linear mapping compresses the bearish side.
    log(ad_ratio) is symmetric around 0, making [0.33, 3.0] map cleanly to [0, 100].

    Scoring:
    - ad_ratio <= EXTREME_BEARISH (0.33) → score = 0
    - ad_ratio = 1.0 → score = 50
    - ad_ratio >= EXTREME_BULLISH (3.0) → score = 100

    Reference: Kaufman Ch.7 (breadth indicators); Dalton Ch.6 (market structure)
    """
    if ad_ratio is None:
        return FeatureScore(score=50.0, raw_value=None, direction="NEUTRAL", available=False)
    if ad_ratio <= 0:
        return FeatureScore(score=0.0, raw_value=ad_ratio, direction="BEARISH", available=True)

    if ad_ratio <= settings.AD_RATIO_EXTREME_BEARISH:
        score = 0.0
    elif ad_ratio >= settings.AD_RATIO_EXTREME_BULLISH:
        score = 100.0
    else:
        log_val = math.log(ad_ratio)
        log_min = math.log(settings.AD_RATIO_EXTREME_BEARISH)
        log_max = math.log(settings.AD_RATIO_EXTREME_BULLISH)
        score = ((log_val - log_min) / (log_max - log_min)) * 100.0

    score = max(0.0, min(100.0, score))
    direction = "BULLISH" if score > 60 else ("BEARISH" if score < 40 else "NEUTRAL")
    return FeatureScore(score=round(score, 2), raw_value=ad_ratio, direction=direction, available=True)

4.4 Open Interest Change Normalizer
def normalize_oi_change(oi_change_pct: float | None, price_direction: int | None, settings: MarketStrengthSettings) -> FeatureScore:
    """Normalize OI change % to [0, 100] score with price direction context.

    OI change interpretation (Dalton Ch.4):
    - Rising OI + rising price = bullish (new long positions being built)
    - Rising OI + falling price = bearish (new short positions being built)
    - Falling OI + rising price = short covering (weak bullish)
    - Falling OI + falling price = long unwinding (weak bearish)

    price_direction: +1 = price up, -1 = price down, 0/None = unknown

    Scoring logic:
    - Large |OI change| with price direction = strong conviction → score toward extreme
    - Large |OI change| without price direction = strong participation → score toward 50+ (slight bullish bias)
    - Small |OI change| = low conviction → score near 50

    Reference: Dalton Ch.4 (OI as participation metric); Weis Ch.2
    """
    if oi_change_pct is None:
        return FeatureScore(score=50.0, raw_value=None, direction="NEUTRAL", available=False)

    spike = settings.OI_CHANGE_SPIKE_THRESHOLD
    abs_change = abs(oi_change_pct)

    if price_direction is not None and price_direction != 0:
        # Dalton Ch.4: OI + price direction = conviction
        # Large OI increase + price up = very bullish (score → 100)
        # Large OI increase + price down = very bearish (score → 0)
        # Large OI decrease + price up = short covering (score → 70)
        # Large OI decrease + price down = long unwinding (score → 30)
        participation_factor = min(abs_change / spike, 2.0) / 2.0  # [0, 1]

        if oi_change_pct > 0:  # Rising OI
            if price_direction > 0:  # Price up → bullish
                score = 50.0 + participation_factor * 50.0
            else:  # Price down → bearish
                score = 50.0 - participation_factor * 50.0
        else:  # Falling OI (unwinding)
            if price_direction > 0:  # Short covering → weak bullish
                score = 50.0 + participation_factor * 20.0
            else:  # Long unwinding → weak bearish
                score = 50.0 - participation_factor * 20.0
    else:
        # No price direction context: OI magnitude = participation strength
        if abs_change >= spike:
            participation_score = min(abs_change / (spike * 2), 1.0) * 50.0
            score = 50.0 + participation_score
        else:
            score = 50.0 + (oi_change_pct / spike) * 25.0

    score = max(0.0, min(100.0, score))
    direction = "BULLISH" if score > 60 else ("BEARISH" if score < 40 else "NEUTRAL")
    return FeatureScore(score=round(score, 2), raw_value=oi_change_pct, direction=direction, available=True)

class MarketStrengthEngine:
    """Composite Market Strength Engine.

    Computes a single 0-100 score from 9 market features using weighted
    normalization (Kaufman Ch.7) with regime classification
    (Dalton/Weis auction market theory).

    Weight redistribution (Correction #14):
    When features are unavailable, their weights are redistributed
    proportionally to available features. This ensures the composite
    score is always computable even when data sources fail.

    Performance target: < 5ms for all 9 features + scoring.

    Output: MarketStrength(score, regime, features_breakdown)
    """

    FEATURE_NAMES: list[str] = [
        "ad_ratio", "vwap_distance", "oi_change", "pcr",
        "volume_profile_position", "market_breadth", "sector_momentum",
        "india_vix", "fii_dii_flow",
    ]

    def __init__(self, settings: MarketStrengthSettings | None = None) -> None:
        self._settings = settings or MarketStrengthSettings()

    def compute_market_strength(self, market_data: MarketDataInput) -> MarketStrength:
        """Compute composite market strength score 0-100.

        This is the PRIMARY entry point for Phase 4.

        Args:
            market_data: MarketDataInput with all raw feature values.
                         None fields indicate unavailable data.

        Returns:
            MarketStrength with score, regime, and features_breakdown.

        Reference: Dalton Ch.6; Weis Ch.3; Kaufman Ch.7
        """
        start_time = time.perf_counter()
        s = self._settings

        # Step 1: Compute VWAP distance from components (if all parts available)
        vwap_distance_atr: float | None = None
        if market_data.vwap is not None and market_data.atr_14 is not None and market_data.close is not None and market_data.atr_14 > 0:
            vwap_distance_atr = (market_data.close - market_data.vwap) / market_data.atr_14

        # Step 2: Normalize all 9 features
        ad_ratio_score = normalize_ad_ratio(market_data.ad_ratio, s)
        vwap_score = normalize_vwap_distance(vwap_distance_atr, s)
        oi_score = normalize_oi_change(market_data.oi_change_pct, market_data.price_direction, s)
        pcr_score = normalize_pcr(market_data.pcr, s)
        vp_score = normalize_volume_profile_position(market_data.volume_profile_position, s)
        breadth_score = normalize_market_breadth(market_data.market_breadth, s)
        sector_score = normalize_sector_momentum(market_data.sector_momentum, s)
        vix_score = normalize_india_vix(market_data.india_vix, market_data.vix_trend, s)
        fii_dii_score = normalize_fii_dii_flow(market_data.fii_dii_net_flow_cr, s)

        # Step 3: Build features breakdown
        breakdown = FeaturesBreakdown(
            ad_ratio=ad_ratio_score,
            vwap_distance=vwap_score,
            oi_change=oi_score,
            pcr=pcr_score,
            volume_profile_position=vp_score,
            market_breadth=breadth_score,
            sector_momentum=sector_score,
            india_vix=vix_score,
            fii_dii_flow=fii_dii_score,
        )

        # Step 4: Compute availability and redistribute weights
        scores = [
            ad_ratio_score, vwap_score, oi_score, pcr_score,
            vp_score, breadth_score, sector_score, vix_score,
            fii_dii_score,
        ]
        weights = [
            s.WEIGHT_AD_RATIO, s.WEIGHT_VWAP_DISTANCE, s.WEIGHT_OI_CHANGE, s.WEIGHT_PCR,
            s.WEIGHT_VOLUME_PROFILE, s.WEIGHT_MARKET_BREADTH, s.WEIGHT_SECTOR_MOMENTUM,
            s.WEIGHT_INDIA_VIX, s.WEIGHT_FII_DII_FLOW,
        ]

        available_mask = [sc.available for sc in scores]
        available_weights = [w for w, avail in zip(weights, available_mask) if avail]
        available_scores = [sc.score for sc, avail in zip(scores, available_mask) if avail]

        total_available_weight = sum(available_weights)

        if total_available_weight == 0:
            # ALL features unavailable — return neutral
            logger.warning("all_features_unavailable", as_of_date=market_data.as_of_date.isoformat())
            effective_weights = {name: 0.0 for name in self.FEATURE_NAMES}
            composite_score = 50.0
        else:
            # Redistribute unavailable weights proportionally to available features
            weight_scale = 1.0 / total_available_weight if total_available_weight > 0 else 0.0
            effective_weights = {}
            scaled_weights = []
            for name, w, avail in zip(self.FEATURE_NAMES, weights, available_mask):
                if avail:
                    ew = w * weight_scale
                    effective_weights[name] = round(ew, 6)
                    scaled_weights.append(ew)
                else:
                    effective_weights[name] = 0.0

            # Weighted average of available feature scores
            composite_score = sum(sw * sc for sw, sc in zip(scaled_weights, available_scores))

        # Clamp to [0, 100]
        composite_score = max(0.0, min(100.0, composite_score))

        # NaN/inf guard
        if math.isnan(composite_score) or math.isinf(composite_score):
            logger.error("composite_score_nan_inf", as_of_date=market_data.as_of_date.isoformat())
            composite_score = 50.0

        breakdown.effective_weights = effective_weights

        # Step 5: Classify regime
        regime = self._classify_regime(composite_score)

        elapsed_ms = (time.perf_counter() - start_time) * 1000

        result = MarketStrength(
            score=round(composite_score, 2),
            regime=regime,
            features_breakdown=breakdown,
            as_of_date=market_data.as_of_date,
            processing_time_ms=round(elapsed_ms, 2),
        )

        logger.info(
            "market_strength_computed",
            as_of_date=market_data.as_of_date.isoformat(),
            score=result.score,
            regime=result.regime.value,
            available_features=sum(available_mask),
            total_features=len(scores),
            processing_time_ms=result.processing_time_ms,
        )

        return result

    def _classify_regime(self, score: float) -> MarketRegime:
        """Classify score into market regime.

        0-30 = WEAK (bearish)
        31-60 = NEUTRAL (range-bound)
        61-100 = STRONG (bullish)

        Boundaries are configurable via settings thresholds.
        """
        if score <= self._settings.WEAK_THRESHOLD:
            return MarketRegime.WEAK
        elif score > self._settings.STRONG_THRESHOLD:
            return MarketRegime.STRONG
        else:
            return MarketRegime.NEUTRAL

8.1 MarketDataFetcher Class
import structlog
import numpy as np
from datetime import date, timedelta
from typing import Any
from src.data.storage import TimescaleDBStore, RedisCache
from config.settings import MarketStrengthSettings

logger = structlog.get_logger(__name__)


class MarketDataFetcher:
    """Fetches raw data for all 9 market strength features.

    Data sources:
    1. A/D ratio: NSE CM bhavcopy via jugaad-data → TimescaleDB cm_spot_eod
    2. VWAP distance: Phase 3 TechnicalIndicators.trend.vwap + volatility.atr_14
    3. OI change: TimescaleDB fo_options_eod (Phase 2 pipeline)
    4. PCR: Phase 2 option chain data → computed from PE/CE OI
    5. Volume profile position: Phase 3 VolumeProfileResult.relative_position
    6. Market breadth: NSE CM bhavcopy → computed (% stocks above 20-EMA)
    7. Sector momentum: NSE sector index history via jugaad-data → TimescaleDB
    8. India VIX: NSE / Zerodha WebSocket / Redis cache (EXTERNAL INPUT)
    9. FII/DII flow: Zerodha Kite historical / jugaad-data → TimescaleDB

    All methods are async (Kleppmann Ch.1: operability).
    All methods handle failure gracefully (Stability principle).
    All methods return None on failure (never raise).
    """

    def __init__(
        self,
        db: TimescaleDBStore,
        cache: RedisCache,
        settings: MarketStrengthSettings | None = None,
    ) -> None:
        self._db = db
        self._cache = cache
        self._settings = settings or MarketStrengthSettings()

    async def fetch_ad_ratio(self, as_of_date: date) -> float | None:
        """Fetch Advance/Decline ratio for as_of_date.

        Source: NSE CM bhavcopy (stored in cm_spot_eod by Phase 2 pipeline).
        A/D = count(close > prev_close) / count(close < prev_close)
        Requires: cm_spot_eod data for as_of_date AND as_of_date - 1.
        """
        try:
            cache_key = f"mktstr:ad_ratio:{as_of_date.isoformat()}"
            cached = await self._cache.get(cache_key)
            if cached is not None:
                return float(cached)

            prev_date = as_of_date - timedelta(days=1)
            current_df = await self._db.query_cm_spot("NIFTY 50", as_of_date, as_of_date)
            prev_df = await self._db.query_cm_spot("NIFTY 50", prev_date, prev_date)

            if current_df is None or prev_df is None or len(current_df) == 0 or len(prev_df) == 0:
                logger.warning("ad_ratio_data_unavailable", date=as_of_date.isoformat())
                return None

            advances = len(current_df[current_df["close"] > current_df["prev_close"]])
            declines = len(current_df[current_df["close"] < current_df["prev_close"]])

            if declines == 0:
                ratio = 10.0 if advances > 0 else None
            else:
                ratio = advances / declines

            if ratio is not None:
                await self._cache.set(cache_key, str(ratio), ex=self._settings.FEATURE_CACHE_TTL_SECONDS)
                logger.info("ad_ratio_computed", date=as_of_date.isoformat(), ratio=round(ratio, 4))

            return ratio
        except Exception as e:
            logger.exception("ad_ratio_fetch_failed", date=as_of_date.isoformat(), error=str(e))
            return None

    async def fetch_oi_change(self, as_of_date: date, symbol: str = "NIFTY") -> float | None:
        """Fetch OI change % for as_of_date vs previous day.

        Source: TimescaleDB fo_options_eod table (Phase 2 pipeline).
        OI change % = (current_total_oi - prev_total_oi) / prev_total_oi × 100
        """
        try:
            cache_key = f"mktstr:oi_change:{symbol}:{as_of_date.isoformat()}"
            cached = await self._cache.get(cache_key)
            if cached is not None:
                return float(cached)

            prev_date = as_of_date - timedelta(days=1)
            current_oi = await self._db.query_oi_total(symbol, as_of_date)
            prev_oi = await self._db.query_oi_total(symbol, prev_date)

            if current_oi is None or prev_oi is None or prev_oi == 0:
                logger.warning("oi_change_data_unavailable", symbol=symbol, date=as_of_date.isoformat())
                return None

            change_pct = ((current_oi - prev_oi) / prev_oi) * 100.0
            await self._cache.set(cache_key, str(change_pct), ex=self._settings.FEATURE_CACHE_TTL_SECONDS)
            logger.info("oi_change_computed", symbol=symbol, date=as_of_date.isoformat(), change_pct=round(change_pct, 2))
            return change_pct
        except Exception as e:
            logger.exception("oi_change_fetch_failed", symbol=symbol, date=as_of_date.isoformat(), error=str(e))
            return None

    async def fetch_pcr(self, as_of_date: date, symbol: str = "NIFTY") -> float | None:
        """Fetch Put/Call ratio (OI-based) for as_of_date.

        Source: TimescaleDB fo_options_eod table.
        PCR = Total PE Open Interest / Total CE Open Interest
        """
        try:
            cache_key = f"mktstr:pcr:{symbol}:{as_of_date.isoformat()}"
            cached = await self._cache.get(cache_key)
            if cached is not None:
                return float(cached)

            pe_oi = await self._db.query_option_oi_by_type(symbol, as_of_date, "PE")
            ce_oi = await self._db.query_option_oi_by_type(symbol, as_of_date, "CE")

            if ce_oi is None or ce_oi == 0:
                logger.warning("pcr_data_unavailable", symbol=symbol, date=as_of_date.isoformat())
                return None

            pcr = pe_oi / ce_oi if pe_oi is not None else None

            if pcr is not None:
                await self._cache.set(cache_key, str(pcr), ex=self._settings.FEATURE_CACHE_TTL_SECONDS)
                logger.info("pcr_computed", symbol=symbol, date=as_of_date.isoformat(), pcr=round(pcr, 4))

            return pcr
        except Exception as e:
            logger.exception("pcr_fetch_failed", symbol=symbol, date=as_of_date.isoformat(), error=str(e))
            return None

    async def fetch_market_breadth(self, as_of_date: date) -> float | None:
        """Fetch market breadth: % of NIFTY 50 stocks above their 20-day EMA.

        Source: NSE CM bhavcopy + computed EMA from historical data.
        Returns: float in [0, 1] range (0.0 = 0% above, 1.0 = 100% above)
        """
        try:
            cache_key = f"mktstr:breadth:{as_of_date.isoformat()}"
            cached = await self._cache.get(cache_key)
            if cached is not None:
                return float(cached)

            breadth = await self._db.query_market_breadth(as_of_date, ema_period=20)

            if breadth is not None:
                await self._cache.set(cache_key, str(breadth), ex=self._settings.FEATURE_CACHE_TTL_SECONDS)
                logger.info("breadth_computed", date=as_of_date.isoformat(), breadth=round(breadth, 4))

            return breadth
        except Exception as e:
            logger.exception("breadth_fetch_failed", date=as_of_date.isoformat(), error=str(e))
            return None

    async def fetch_sector_momentum(self, as_of_date: date) -> float | None:
        """Fetch sector momentum: average relative strength of sector indices vs NIFTY.

        Source: NSE sector index history via jugaad-data → TimescaleDB.
        sector_rs = mean(sector_return - NIFTY_return) over SECTOR_MOMENTUM_LOOKBACK days
        """
        try:
            cache_key = f"mktstr:sector_momentum:{as_of_date.isoformat()}"
            cached = await self._cache.get(cache_key)
            if cached is not None:
                return float(cached)

            lookback = self._settings.SECTOR_MOMENTUM_LOOKBACK
            start_date = as_of_date - timedelta(days=lookback + 10)  # +10 for holidays

            nifty_return = await self._db.query_index_return("NIFTY 50", start_date, as_of_date)
            if nifty_return is None:
                logger.warning("sector_momentum_nifty_unavailable", date=as_of_date.isoformat())
                return None

            sector_returns = []
            for sector_name in self._settings.SECTOR_INDICES:
                sector_ret = await self._db.query_index_return(sector_name, start_date, as_of_date)
                if sector_ret is not None:
                    sector_returns.append(sector_ret - nifty_return)

            if not sector_returns:
                logger.warning("sector_momentum_no_data", date=as_of_date.isoformat())
                return None

            avg_rs = float(np.mean(sector_returns)) * 100.0

            await self._cache.set(cache_key, str(avg_rs), ex=self._settings.FEATURE_CACHE_TTL_SECONDS)
            logger.info("sector_momentum_computed", date=as_of_date.isoformat(), avg_rs=round(avg_rs, 4), num_sectors=len(sector_returns))
            return avg_rs
        except Exception as e:
            logger.exception("sector_momentum_fetch_failed", date=as_of_date.isoformat(), error=str(e))
            return None

    async def fetch_india_vix(self, as_of_date: date) -> tuple[float | None, str | None]:
        """Fetch India VIX value and trend direction.

        Source: NSE / Zerodha WebSocket / Redis cache.
        CRITICAL: India VIX is an EXTERNAL INPUT — NEVER computed from OHLCV.

        Returns: (vix_value, vix_trend) where trend is "RISING", "FALLING", or "STABLE"
        """
        try:
            cache_key = "mktstr:india_vix:latest"
            cached_vix = await self._cache.get(cache_key)
            if cached_vix is not None:
                vix_value = float(cached_vix)
            else:
                vix_value = await self._db.query_india_vix(as_of_date)

            if vix_value is None:
                logger.warning("india_vix_unavailable", date=as_of_date.isoformat())
                return None, None

            # Determine VIX trend: compare current vs 5-day ago
            prev_vix = await self._db.query_india_vix(as_of_date - timedelta(days=5))
            vix_trend = "STABLE"
            if prev_vix is not None and prev_vix > 0:
                change_pct = (vix_value - prev_vix) / prev_vix
                if change_pct > 0.05:
                    vix_trend = "RISING"
                elif change_pct < -0.05:
                    vix_trend = "FALLING"

            logger.info("india_vix_fetched", date=as_of_date.isoformat(), vix=vix_value, trend=vix_trend)
            return vix_value, vix_trend
        except Exception as e:
            logger.exception("india_vix_fetch_failed", date=as_of_date.isoformat(), error=str(e))
            return None, None

    async def fetch_fii_dii_flow(self, as_of_date: date) -> float | None:
        """Fetch net FII+DII flow for as_of_date (in Crores).

        Source: Zerodha Kite historical API or jugaad-data → TimescaleDB.
        net_flow = FII_net_buy + DII_net_buy (in Cr)
        """
        try:
            cache_key = f"mktstr:fii_dii_flow:{as_of_date.isoformat()}"
            cached = await self._cache.get(cache_key)
            if cached is not None:
                return float(cached)

            net_flow = await self._db.query_fii_dii_flow(as_of_date)

            if net_flow is not None:
                await self._cache.set(cache_key, str(net_flow), ex=self._settings.FEATURE_CACHE_TTL_SECONDS)
                logger.info("fii_dii_flow_fetched", date=as_of_date.isoformat(), net_flow_cr=round(net_flow, 2))

            return net_flow
        except Exception as e:
            logger.exception("fii_dii_flow_fetch_failed", date=as_of_date.isoformat(), error=str(e))
            return None

    async def fetch_all(self, as_of_date: date, technical_report=None) -> MarketDataInput:
        """Fetch all market data and assemble MarketDataInput.

        Convenience method for the orchestrator (Phase 11).
        Populates MarketDataInput from all data sources + Phase 3 TechnicalReport.
        Unavailable features are left as None — the engine handles this gracefully.
        """
        ad_ratio = await self.fetch_ad_ratio(as_of_date)
        oi_change = await self.fetch_oi_change(as_of_date)
        pcr = await self.fetch_pcr(as_of_date)
        breadth = await self.fetch_market_breadth(as_of_date)
        sector_momentum = await self.fetch_sector_momentum(as_of_date)
        vix_value, vix_trend = await self.fetch_india_vix(as_of_date)
        fii_dii_flow = await self.fetch_fii_dii_flow(as_of_date)

        # From Phase 3 TechnicalReport (if provided)
        vwap = None
        atr_14 = None
        close = None
        volume_profile_position = None
        price_direction = None

        if technical_report is not None:
            vwap = technical_report.indicators.trend.vwap
            atr_14 = technical_report.indicators.volatility.atr_14
            # Extract close from the last bar (if accessible)
            # TechnicalReport doesn't store raw OHLCV — the caller must pass close separately
            volume_profile_position = technical_report.volume_signals.profile.relative_position
            # Price direction from trend indicators
            if technical_report.indicators.trend.supertrend_direction is not None:
                price_direction = technical_report.indicators.trend.supertrend_direction

        return MarketDataInput(
            as_of_date=as_of_date,
            ad_ratio=ad_ratio,
            vwap=vwap,
            atr_14=atr_14,
            close=close,
            oi_change_pct=oi_change,
            price_direction=price_direction,
            pcr=pcr,
            volume_profile_position=volume_profile_position,
            market_breadth=breadth,
            sector_momentum=sector_momentum,
            india_vix=vix_value,
            vix_trend=vix_trend,
            fii_dii_net_flow_cr=fii_dii_flow,
        )

def test_market_strength_monotonic_with_vix(mktstr_engine):
    """As VIX increases (with FALLING trend held constant), score should generally decrease in the normal range."""
    scores = []
    for vix in [12.0, 15.0, 18.0, 22.0, 25.0]:
        data = MarketDataInput(
            as_of_date=date(2025, 1, 15),
            india_vix=vix,
            vix_trend="STABLE",
        )
        result = mktstr_engine.compute_market_strength(data)
        scores.append(result.features_breakdown.india_vix.score)
    # Scores should be monotonically decreasing in the normal VIX range
    for i in range(len(scores) - 1):
        assert scores[i] >= scores[i + 1], f"VIX score not decreasing: {scores}"
