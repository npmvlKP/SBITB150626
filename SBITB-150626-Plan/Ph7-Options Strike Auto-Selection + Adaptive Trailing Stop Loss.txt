from typing import Literal
from pydantic import model_validator


class StrikeSelectionSettings(BaseSettings):
    model_config = SettingsConfigDict(env_prefix="STRIKE_", env_file=".env", extra="ignore")

    # --- NIFTY lot size (Correction #1, carry-forward from Phase 5) ---
    NIFTY_LOT_SIZE: int = Field(default=25, ge=1, description="NIFTY options lot size (25 since Nov 2021)")

    # --- Strike interval ---
    NIFTY_STRIKE_INTERVAL: int = Field(default=50, ge=5, description="NIFTY strike interval in rupees (multiples of 50)")

    # --- Selling strike selection (2SD / ATR-based, Correction #10) ---
    SELLING_STRIKE_SD_MULTIPLIER: float = Field(default=2.0, ge=1.0, le=4.0, description="SD multiplier for selling strike selection (2SD default)")
    SELLING_STRIKE_MIN_OI: int = Field(default=10000, ge=100, description="Minimum OI on selected strike (Correction #12)")
    SELLING_STRIKE_MIN_VOLUME: int = Field(default=1000, ge=100, description="Minimum volume on selected strike (Correction #12)")
    SELLING_STRIKE_MAX_SEARCH_OFFSETS: int = Field(default=4, ge=1, le=10, description="Max ±offset intervals to search if primary strike fails validation")

    # --- Buying strike selection (delta-targeted, Correction #11) ---
    BUYING_DELTA_MIN: float = Field(default=0.50, ge=0.30, le=0.70, description="Minimum delta for buying strike")
    BUYING_DELTA_MAX: float = Field(default=0.60, ge=0.40, le=0.80, description="Maximum delta for buying strike")
    BUYING_DELTA_TARGET: float = Field(default=0.55, ge=0.35, le=0.75, description="Target delta for strike ranking (midpoint preference)")
    BUYING_DELTA_WIDEN_MIN: float = Field(default=0.45, ge=0.30, le=0.60, description="Widened delta range min if no strike in primary range")
    BUYING_DELTA_WIDEN_MAX: float = Field(default=0.65, ge=0.50, le=0.80, description="Widened delta range max if no strike in primary range")
    BUYING_STRIKE_MIN_OI: int = Field(default=10000, ge=100, description="Minimum OI on selected strike (Correction #12)")
    BUYING_STRIKE_MIN_VOLUME: int = Field(default=1000, ge=100, description="Minimum volume on selected strike (Correction #12)")

    # --- IV skew adjustment (Natenberg Ch.5) ---
    IV_SKEW_AWARENESS: bool = Field(default=True, description="Account for IV skew in strike selection (PE typically has higher IV than CE at same SD)")
    IV_SKEW_PE_ADJUSTMENT_PCT: float = Field(default=0.05, ge=0.0, le=0.20, description="PE strike distance adjustment: push PE slightly further OTM to account for higher IV (Natenberg Ch.5)")

    # --- Gamma risk scaling near expiry (Natenberg Ch.5-6, Zerodha Varsity Module 14) ---
    GAMMA_RISK_DAYS_THRESHOLD: int = Field(default=5, ge=1, le=10, description="Days to expiry below which gamma risk scaling activates")
    GAMMA_RISK_LOT_REDUCTION_PCT: float = Field(default=0.25, ge=0.0, le=0.50, description="Reduce lot size by this % when within gamma risk threshold")
    GAMMA_RISK_SELLING_PUSH_OTM_STRIKES: int = Field(default=1, ge=0, le=3, description="Push selling strikes this many intervals further OTM near expiry")

    # --- Strike ranking weights (Natenberg Ch.4, Ch.8) ---
    RANK_WEIGHT_DELTA_PROXIMITY: float = Field(default=0.30, ge=0.0, le=1.0, description="Weight: delta proximity to target in strike ranking")
    RANK_WEIGHT_LIQUIDITY: float = Field(default=0.25, ge=0.0, le=1.0, description="Weight: OI + volume composite in strike ranking")
    RANK_WEIGHT_IV_FAVORABILITY: float = Field(default=0.25, ge=0.0, le=1.0, description="Weight: IV favorability (low IV for buying, high IV for selling) in strike ranking")
    RANK_WEIGHT_BID_ASK_SPREAD: float = Field(default=0.20, ge=0.0, le=1.0, description="Weight: bid-ask spread tightness in strike ranking")

    @model_validator(mode="after")
    def validate_strike_selection_settings(self) -> "StrikeSelectionSettings":
        if self.BUYING_DELTA_MIN >= self.BUYING_DELTA_MAX:
            raise ValueError(f"BUYING_DELTA_MIN ({self.BUYING_DELTA_MIN}) must be < BUYING_DELTA_MAX ({self.BUYING_DELTA_MAX})")
        if self.BUYING_DELTA_WIDEN_MIN >= self.BUYING_DELTA_WIDEN_MAX:
            raise ValueError(f"BUYING_DELTA_WIDEN_MIN ({self.BUYING_DELTA_WIDEN_MIN}) must be < BUYING_DELTA_WIDEN_MAX ({self.BUYING_DELTA_WIDEN_MAX})")
        if self.BUYING_DELTA_WIDEN_MIN > self.BUYING_DELTA_MIN:
            raise ValueError(f"BUYING_DELTA_WIDEN_MIN ({self.BUYING_DELTA_WIDEN_MIN}) must be <= BUYING_DELTA_MIN ({self.BUYING_DELTA_MIN})")
        if self.BUYING_DELTA_WIDEN_MAX < self.BUYING_DELTA_MAX:
            raise ValueError(f"BUYING_DELTA_WIDEN_MAX ({self.BUYING_DELTA_WIDEN_MAX}) must be >= BUYING_DELTA_MAX ({self.BUYING_DELTA_MAX})")
        total_weight = self.RANK_WEIGHT_DELTA_PROXIMITY + self.RANK_WEIGHT_LIQUIDITY + self.RANK_WEIGHT_IV_FAVORABILITY + self.RANK_WEIGHT_BID_ASK_SPREAD
        if abs(total_weight - 1.0) > 0.01:
            raise ValueError(f"Strike ranking weights must sum to 1.0, got {total_weight:.4f}")
        return self

class StrikeSelectorEngine:
    """Production-grade strike auto-selection engine.

    Extracts and hardens Phase 5 strike_selector.py into a standalone engine with:
    - Delta-targeted selection for buying (Natenberg Ch.4)
    - Sigma/ATR-based selection for selling (Kaufman Ch.7)
    - IV skew awareness (Natenberg Ch.5)
    - Gamma risk scaling near expiry (Natenberg Ch.5-6)
    - Multi-factor strike ranking (Natenberg Ch.4, Ch.8)
    - Comprehensive rejection tracking

    Reference: Natenberg Ch.4-8; Zerodha Varsity Modules 5-14; Kaufman Ch.7, Ch.10
    """

    def __init__(self, settings: StrikeSelectionSettings | None = None):
        self.settings = settings or StrikeSelectionSettings()

    def select_strike(
        self,
        spot: float,
        direction: str,
        option_chain: OptionChainSnapshot,
        atr_14: float,
        iv_rank: float,
        as_of_date: date,
        market_strength_score: float = 50.0,
    ) -> StrikeSelectionResult:
        """Unified strike selection entry point.

        Dispatches to buying or selling logic based on direction:
        - BULLISH/BEARISH → delta-targeted buying strike
        - NEUTRAL_SELL → 2SD ATR-based selling strikes (strangle)

        Args:
            spot: Current spot price
            direction: BULLISH, BEARISH, or NEUTRAL_SELL
            option_chain: Option chain snapshot from Phase 2
            atr_14: ATR(14) from Phase 3 technical indicators
            iv_rank: IV rank percentile [0, 100] from Phase 2 greeks_snapshot
            as_of_date: Explicit date parameter (Correction #9)
            market_strength_score: Market strength score 0-100 from Phase 4

        Returns:
            StrikeSelectionResult with selected strikes or rejection reason

        Reference: Natenberg Ch.4 (delta); Kaufman Ch.7 (2SD≈2×ATR); Correction #10, #11, #12
        """
        if direction in ("BULLISH", "BEARISH"):
            strike_choice = self._select_buying_strike(
                spot=spot, direction=direction, option_chain=option_chain,
                atr_14=atr_14, iv_rank=iv_rank, as_of_date=as_of_date,
                market_strength_score=market_strength_score,
            )
            if strike_choice is None:
                return StrikeSelectionResult(
                    success=False,
                    rejection_reason=StrikeRejectionReason.NO_STRIKES_AVAILABLE,
                    as_of_date=as_of_date,
                )
            return StrikeSelectionResult(
                success=True, buying_strike=strike_choice, as_of_date=as_of_date,
            )
        elif direction == "NEUTRAL_SELL":
            selling_choice = self._select_selling_strikes(
                spot=spot, option_chain=option_chain, atr_14=atr_14,
                iv_rank=iv_rank, as_of_date=as_of_date,
            )
            if selling_choice is None:
                return StrikeSelectionResult(
                    success=False,
                    rejection_reason=StrikeRejectionReason.NO_STRIKES_AVAILABLE,
                    as_of_date=as_of_date,
                )
            return StrikeSelectionResult(
                success=True, selling_strikes=selling_choice, as_of_date=as_of_date,
            )
        else:
            logger.warning("Unknown direction for strike selection", direction=direction)
            return StrikeSelectionResult(
                success=False,
                rejection_reason=StrikeRejectionReason.NO_STRIKES_AVAILABLE,
                as_of_date=as_of_date,
            )

4.2 Buying Strike Selection (Delta-Targeted, Natenberg Ch.4)
    def _select_buying_strike(
        self,
        spot: float,
        direction: str,
        option_chain: OptionChainSnapshot,
        atr_14: float,
        iv_rank: float,
        as_of_date: date,
        market_strength_score: float,
    ) -> StrikeChoice | None:
        """Select strike for directional buying using delta targeting.

        Process (Natenberg Ch.4; Correction #11):
        1. Determine option type: BULLISH→CE, BEARISH→PE
        2. Primary search: find strikes with delta in [BUYING_DELTA_MIN, BUYING_DELTA_MAX]
        3. Rank candidates by multi-factor score (delta proximity, liquidity, IV, spread)
        4. If no candidates in primary range, widen to [BUYING_DELTA_WIDEN_MIN, BUYING_DELTA_WIDEN_MAX]
        5. Validate OI and volume on best candidate (Correction #12)
        6. If within gamma risk days, prefer strikes slightly further OTM (Natenberg Ch.5-6)

        Returns: StrikeChoice or None
        """
        option_type = "CE" if direction == "BULLISH" else "PE"
        delta_target = self.settings.BUYING_DELTA_TARGET

        # Gamma risk adjustment: near expiry, prefer slightly OTM strikes
        if option_chain.days_to_expiry <= self.settings.GAMMA_RISK_DAYS_THRESHOLD:
            delta_target = delta_target - 0.05  # Push toward slightly OTM
            logger.info("Gamma risk: adjusting delta target toward OTM",
                       original_target=self.settings.BUYING_DELTA_TARGET,
                       adjusted_target=delta_target,
                       days_to_expiry=option_chain.days_to_expiry)

        # Primary search
        candidates = self._rank_strikes(
            option_type=option_type,
            delta_min=self.settings.BUYING_DELTA_MIN,
            delta_max=self.settings.BUYING_DELTA_MAX,
            delta_target=delta_target,
            option_chain=option_chain,
            iv_rank=iv_rank,
            is_buying=True,
        )

        # Widen if no candidates
        if not candidates:
            logger.info("No strikes in primary delta range, widening search",
                       as_of_date=str(as_of_date))
            candidates = self._rank_strikes(
                option_type=option_type,
                delta_min=self.settings.BUYING_DELTA_WIDEN_MIN,
                delta_max=self.settings.BUYING_DELTA_WIDEN_MAX,
                delta_target=delta_target,
                option_chain=option_chain,
                iv_rank=iv_rank,
                is_buying=True,
            )

        if not candidates:
            logger.warning("No valid buying strike found after widening",
                          direction=direction, option_type=option_type,
                          as_of_date=str(as_of_date))
            return None

        best = candidates[0]  # Highest ranked
        logger.info("Buying strike selected",
                   strike=best.strike, option_type=option_type,
                   delta=best.delta, score=best.rank_score.weighted_total,
                   as_of_date=str(as_of_date))
        return best

4.3 Selling Strike Selection (2SD ATR-Based, Kaufman Ch.7)
    def _select_selling_strikes(
        self,
        spot: float,
        option_chain: OptionChainSnapshot,
        atr_14: float,
        iv_rank: float,
        as_of_date: date,
    ) -> SellingStrikeChoice | None:
        """Select OTM CE and PE strikes for strangle selling.

        Process (Kaufman Ch.7; Correction #10, #12):
        1. Compute SD distance = SELLING_STRIKE_SD_MULTIPLIER × ATR(14)
        2. CE target = spot + sd_distance; PE target = spot - sd_distance
        3. Round to nearest strike interval (NIFTY=50)
        4. IV skew adjustment: push PE slightly further OTM (Natenberg Ch.5, Correction #4 in this table)
        5. Gamma risk adjustment: near expiry, push strikes further OTM
        6. Validate OI and volume; search ±offsets if primary fails
        7. Rank valid candidates and select best

        Returns: SellingStrikeChoice or None
        """
        sd_distance = self.settings.SELLING_STRIKE_SD_MULTIPLIER * atr_14

        # IV skew adjustment for PE (Natenberg Ch.5: PE typically has higher IV)
        pe_extra_distance = 0.0
        if self.settings.IV_SKEW_AWARENESS:
            pe_extra_distance = sd_distance * self.settings.IV_SKEW_PE_ADJUSTMENT_PCT

        # Gamma risk adjustment: push strikes further OTM near expiry
        gamma_extra_offsets = 0
        if option_chain.days_to_expiry <= self.settings.GAMMA_RISK_DAYS_THRESHOLD:
            gamma_extra_offsets = self.settings.GAMMA_RISK_SELLING_PUSH_OTM_STRIKES
            logger.info("Gamma risk: pushing selling strikes further OTM",
                       extra_offsets=gamma_extra_offsets,
                       days_to_expiry=option_chain.days_to_expiry)

        interval = self.settings.NIFTY_STRIKE_INTERVAL

        ce_target = spot + sd_distance + (gamma_extra_offsets * interval)
        pe_target = spot - sd_distance - pe_extra_distance - (gamma_extra_offsets * interval)

        ce_strike_price = self._round_to_strike_interval(ce_target)
        pe_strike_price = self._round_to_strike_interval(pe_target)

        ce_choice = self._find_validated_strike(
            target_price=ce_strike_price, option_type="CE",
            option_chain=option_chain, iv_rank=iv_rank, is_buying=False,
        )
        pe_choice = self._find_validated_strike(
            target_price=pe_strike_price, option_type="PE",
            option_chain=option_chain, iv_rank=iv_rank, is_buying=False,
        )

        # Search nearby if primary strike fails validation
        if ce_choice is None:
            ce_choice = self._search_nearby_strike(
                target_price=ce_strike_price, option_type="CE",
                option_chain=option_chain, iv_rank=iv_rank,
                max_offsets=self.settings.SELLING_STRIKE_MAX_SEARCH_OFFSETS,
            )
        if pe_choice is None:
            pe_choice = self._search_nearby_strike(
                target_price=pe_strike_price, option_type="PE",
                option_chain=option_chain, iv_rank=iv_rank,
                max_offsets=self.settings.SELLING_STRIKE_MAX_SEARCH_OFFSETS,
            )

        if ce_choice is None or pe_choice is None:
            logger.warning("No valid selling strikes found",
                          ce_target=ce_target, pe_target=pe_target,
                          as_of_date=str(as_of_date))
            return None

        combined_premium = ce_choice.ltp + pe_choice.ltp

        result = SellingStrikeChoice(
            ce_strike=ce_choice,
            pe_strike=pe_choice,
            combined_premium=combined_premium,
            as_of_date=as_of_date,
            metadata={
                "sd_distance": sd_distance,
                "iv_skew_pe_adjusted": pe_extra_distance > 0,
                "gamma_risk_pushed_otm": gamma_extra_offsets > 0,
                "gamma_extra_offsets": gamma_extra_offsets,
                "atr_14": atr_14,
                "spot": spot,
            },
        )

        logger.info("Selling strikes selected",
                    ce_strike=ce_choice.strike, pe_strike=pe_choice.strike,
                    combined_premium=combined_premium,
                    as_of_date=str(as_of_date))
        return result

4.4 Strike Ranking (Multi-Factor, Natenberg Ch.4, Ch.8)
    def _rank_strikes(
        self,
        option_type: str,
        delta_min: float,
        delta_max: float,
        delta_target: float,
        option_chain: OptionChainSnapshot,
        iv_rank: float,
        is_buying: bool,
    ) -> list[StrikeChoice]:
        """Rank candidate strikes by multi-factor score.

        Scoring components (weights from StrikeSelectionSettings):
        1. Delta proximity: how close is delta to target (Natenberg Ch.4)
        2. Liquidity: OI + volume composite (higher = better execution)
        3. IV favorability: for buying, low IV is favorable; for selling, high IV is favorable (Natenberg Ch.5)
        4. Bid-ask spread: tighter = better execution quality

        Returns: List of StrikeChoice sorted by weighted_total descending (best first)
        """
        candidates = []
        for s in option_chain.strikes:
            if s.option_type != option_type or s.delta is None:
                continue
            abs_delta = abs(s.delta)
            if not (delta_min <= abs_delta <= delta_max):
                continue

            # Validate OI and volume
            min_oi = self.settings.BUYING_STRIKE_MIN_OI if is_buying else self.settings.SELLING_STRIKE_MIN_OI
            min_vol = self.settings.BUYING_STRIKE_MIN_VOLUME if is_buying else self.settings.SELLING_STRIKE_MIN_VOLUME
            if s.oi < min_oi or s.volume < min_vol:
                continue

            # Score: delta proximity (0-1)
            delta_dist = abs(abs_delta - delta_target)
            max_delta_range = delta_max - delta_min
            delta_proximity = max(0.0, 1.0 - (delta_dist / max_delta_range)) if max_delta_range > 0 else 1.0

            # Score: liquidity (0-1, relative to max in chain)
            max_oi = max((st.oi for st in option_chain.strikes if st.option_type == option_type), default=1)
            liquidity = min(1.0, s.oi / max_oi) if max_oi > 0 else 0.0

            # Score: IV favorability (0-1)
            if s.iv is not None and s.iv > 0:
                if is_buying:
                    iv_favorability = max(0.0, 1.0 - (s.iv / 100.0))  # Low IV = favorable for buying
                else:
                    iv_favorability = min(1.0, s.iv / 100.0)  # High IV = favorable for selling
            else:
                iv_favorability = 0.5  # Neutral if IV unavailable

            # Score: bid-ask spread (approximation from ltp, 0-1)
            bid_ask_spread = 0.5  # Default neutral; improved if bid/ask data available

            # Compute weighted total
            weighted_total = (
                delta_proximity * self.settings.RANK_WEIGHT_DELTA_PROXIMITY
                + liquidity * self.settings.RANK_WEIGHT_LIQUIDITY
                + iv_favorability * self.settings.RANK_WEIGHT_IV_FAVORABILITY
                + bid_ask_spread * self.settings.RANK_WEIGHT_BID_ASK_SPREAD
            )

            # Classify moneyness
            moneyness = self._classify_moneyness(s.strike, option_chain.atm_strike, option_type)

            rank_score = StrikeRankScore(
                delta_proximity=round(delta_proximity, 4),
                liquidity=round(liquidity, 4),
                iv_favorability=round(iv_favorability, 4),
                bid_ask_spread=round(bid_ask_spread, 4),
                weighted_total=round(weighted_total, 4),
            )

            strike_choice = StrikeChoice(
                strike=s.strike,
                option_type=option_type,
                moneyness=moneyness,
                delta=s.delta,
                gamma=s.gamma,
                iv=s.iv,
                ltp=s.ltp,
                oi=s.oi,
                volume=s.volume,
                rank_score=rank_score,
                as_of_date=option_chain.as_of_date,
                metadata={"iv_rank": iv_rank},
            )
            candidates.append(strike_choice)

        candidates.sort(key=lambda x: x.rank_score.weighted_total, reverse=True)
        return candidates

4.5 Helper Functions
    @staticmethod
    def _round_to_strike_interval(price: float, symbol: str = "NIFTY") -> float:
        """Round price to nearest valid strike interval.

        NIFTY: multiples of 50.
        BANKNIFTY: multiples of 100 (future extension).
        """
        interval = 50 if symbol == "NIFTY" else 100
        return round(price / interval) * interval

    @staticmethod
    def _classify_moneyness(strike: float, atm_strike: float, option_type: str) -> Moneyness:
        """Classify strike as ITM/ATM/OTM.

        Reference: Zerodha Varsity Module 6 (Option Moneyness)
        """
        if option_type == "CE":
            if strike < atm_strike:
                return Moneyness.ITM
            elif strike == atm_strike:
                return Moneyness.ATM
            else:
                return Moneyness.OTM
        else:  # PE
            if strike > atm_strike:
                return Moneyness.ITM
            elif strike == atm_strike:
                return Moneyness.ATM
            else:
                return Moneyness.OTM

    def _find_validated_strike(
        self,
        target_price: float,
        option_type: str,
        option_chain: OptionChainSnapshot,
        iv_rank: float,
        is_buying: bool,
    ) -> StrikeChoice | None:
        """Find and validate a strike at the target price."""
        ranked = self._rank_strikes(
            option_type=option_type,
            delta_min=0.0, delta_max=1.0,  # Wide range for selling (sigma-based pre-filter)
            delta_target=0.5,  # Not used for selling ranking
            option_chain=option_chain,
            iv_rank=iv_rank,
            is_buying=is_buying,
        )
        # Find closest strike to target price among ranked candidates
        for candidate in ranked:
            if candidate.strike == target_price:
                return candidate
        # If exact match not found, find closest
        closest = None
        min_dist = float("inf")
        for candidate in ranked:
            if candidate.option_type == option_type:
                dist = abs(candidate.strike - target_price)
                if dist < min_dist:
                    min_dist = dist
                    closest = candidate
        return closest

    def _search_nearby_strike(
        self,
        target_price: float,
        option_type: str,
        option_chain: OptionChainSnapshot,
        iv_rank: float,
        max_offsets: int = 4,
    ) -> StrikeChoice | None:
        """Search ±offset intervals from target for a valid strike (Correction #12)."""
        interval = self.settings.NIFTY_STRIKE_INTERVAL
        for offset in range(1, max_offsets + 1):
            for candidate_price in [target_price + offset * interval, target_price - offset * interval]:
                result = self._find_validated_strike(
                    target_price=candidate_price,
                    option_type=option_type,
                    option_chain=option_chain,
                    iv_rank=iv_rank,
                    is_buying=False,
                )

logger = structlog.get_logger(__name__)


class PositionSizerEngine:
    """Production-grade position sizing engine.

    Extracts and hardens Phase 5's compute_lots() into a standalone engine with:
    - Three sizing methods: fixed_fractional, kelly, optimal_f (Kaufman Ch.9-10)

    - Margin constraint checking (Correction #22)

    - Multiple hard caps: max lots, max daily notional, max open positions

    Reference: Kaufman Ch.9-10 (position sizing); Natenberg Ch.5-6 (gamma risk); Phase 6 cost model

6.2 calculate_lot_size() — PRIMARY ENTRY POINT
    def calculate_lot_size(
        self,
        account_capital: float,
        stop_distance_per_lot: float,
        as_of_date: date,
        premium_per_lot: float = 0.0,
        days_to_expiry: int = 30,
        num_legs: int = 1,
        is_selling: bool = False,
        available_margin: float | None = None,
        trade_history: list[dict] | None = None,
        win_rate: float | None = None,
        avg_win_loss_ratio: float | None = None,
        daily_notional_used: float = 0.0,
        open_positions: int = 0,
    ) -> LotSizeResult:
        """Compute number of lots to trade — SECOND PRIMARY OUTPUT of Phase 7.

        Process:
        1. Compute raw risk_per_trade using configured sizing method
        2. Deduct estimated transaction costs from risk_per_trade (Correction #23)
        3. Compute raw lots = net_risk / stop_distance_per_lot
        4. Apply gamma risk reduction if near expiry (Natenberg Ch.5-6)
        5. Apply hard caps: MAX_LOTS_PER_TRADE, MAX_DAILY_NOTIONAL
        6. Check margin constraint and reduce if necessary (Correction #22)
        7. Ensure lots >= 1 (minimum one lot)

        Args:
            account_capital: Total account capital in INR
            stop_distance_per_lot: Maximum risk per lot in INR (stop distance × lot_size)
            as_of_date: Explicit date parameter (Correction #9)
            premium_per_lot: Premium × lot_size — used for cost estimation
            days_to_expiry: Calendar days to expiry — for gamma risk scaling
            num_legs: Number of legs in the trade (1 for buying, 2 for strangle)
            is_selling: Whether this is a selling strategy (affects cost model)
            available_margin: Available margin from broker API (None = use SPAN estimate)
            trade_history: Historical trades for optimal-f method
            win_rate: Win rate for Kelly method (if None, computed from trade_history)
            avg_win_loss_ratio: Avg win/loss for Kelly method
            daily_notional_used: Notional exposure already used today
            open_positions: Current number of open positions

        Returns:
            LotSizeResult with computed lots, cost breakdown, margin check, and all flags

        Reference: Kaufman Ch.9-10 (position sizing); Correction #16, #22, #23
        """
        if stop_distance_per_lot <= 0:
            logger.warning("stop_distance_per_lot must be > 0, returning 1 lot")
            return LotSizeResult(lots=1, method="fallback", as_of_date=as_of_date,
                               stop_distance_per_lot=stop_distance_per_lot)

        # Step 1: Compute raw risk using configured method
        method = self.settings.POSITION_SIZING_METHOD
        if method == "fixed_fractional":
            risk_per_trade = account_capital * self.settings.FIXED_FRACTIONAL_PCT
        elif method == "kelly":
            risk_per_trade = self._kelly_risk(account_capital, win_rate, avg_win_loss_ratio)
        elif method == "optimal_f":
            risk_per_trade = self._optimal_f_risk(account_capital, trade_history)
        else:
            logger.warning("Unknown position sizing method, falling back to fixed_fractional", method=method)
            risk_per_trade = account_capital * self.settings.FIXED_FRACTIONAL_PCT
            method = "fixed_fractional"

        # Step 2: Deduct estimated transaction costs (Correction #23)
        cost_breakdown = CostBreakdown()
        risk_after_costs = risk_per_trade
        if self.settings.COST_DEDUCTION_ENABLED and premium_per_lot > 0:
            cost_breakdown = self._estimate_transaction_costs(
                premium_per_lot=premium_per_lot, lots=1,  # Estimate per-lot cost first
                is_selling=is_selling, num_legs=num_legs,
            )
            # Total cost scales with lots, so estimate for 1 lot and subtract from risk
            risk_after_costs = risk_per_trade - cost_breakdown.total
            if risk_after_costs <= 0:
                logger.warning("Risk per trade consumed entirely by transaction costs",
                             risk_per_trade=risk_per_trade, costs=cost_breakdown.total)
                risk_after_costs = risk_per_trade  # Fallback: ignore costs rather than block trade

        # Step 3: Compute raw lots
        lots = int(risk_after_costs / stop_distance_per_lot)
        lots = max(1, lots)

        # Step 4: Gamma risk reduction (Natenberg Ch.5-6)
        gamma_risk_reduced = False
        gamma_risk_reduction_pct = 0.0
        if days_to_expiry <= self.settings.GAMMA_RISK_DAYS_THRESHOLD:
            reduction = self.settings.GAMMA_RISK_LOT_REDUCTION_PCT
            lots = max(1, int(lots * (1.0 - reduction)))
            gamma_risk_reduced = True
            gamma_risk_reduction_pct = reduction
            logger.info("Gamma risk: lots reduced near expiry",
                       days_to_expiry=days_to_expiry, reduction_pct=reduction,
                       adjusted_lots=lots)

        # Step 5: Hard caps
        capped_by_max_lots = False
        if lots > self.settings.MAX_LOTS_PER_TRADE:
            lots = self.settings.MAX_LOTS_PER_TRADE
            capped_by_max_lots = True
            logger.info("Lots capped by MAX_LOTS_PER_TRADE", max_lots=self.settings.MAX_LOTS_PER_TRADE)

        capped_by_daily_notional = False
        if premium_per_lot > 0:
            notional_per_lot = premium_per_lot * self.settings.NIFTY_LOT_SIZE
            proposed_notional = lots * notional_per_lot
            remaining_daily_notional = self.settings.MAX_DAILY_NOTIONAL - daily_notional_used
            if proposed_notional > remaining_daily_notional:
                lots = max(1, int(remaining_daily_notional / notional_per_lot))
                capped_by_daily_notional = True
                logger.info("Lots reduced by MAX_DAILY_NOTIONAL", remaining=remaining_daily_notional)

        # Step 6: Recompute actual costs with final lot count
        if self.settings.COST_DEDUCTION_ENABLED and premium_per_lot > 0:
            cost_breakdown = self._estimate_transaction_costs(
                premium_per_lot=premium_per_lot, lots=lots,
                is_selling=is_selling, num_legs=num_legs,
            )

        # Step 7: Margin constraint (Correction #22)
        margin_check = MarginCheck()
        capped_by_margin = False
        if self.settings.MARGIN_CHECK_ENABLED and premium_per_lot > 0:
            notional = lots * premium_per_lot * self.settings.NIFTY_LOT_SIZE
            required_margin = notional * self.settings.SPAN_MARGIN_PCT
            if available_margin is None:
                available_margin = account_capital * 0.80  # Conservative: assume 80% of capital is margin
            effective_margin = available_margin * (1.0 - self.settings.MARGIN_SAFETY_BUFFER_PCT)
            margin_sufficient = required_margin <= effective_margin
            margin_utilization = required_margin / effective_margin if effective_margin > 0 else 1.0

            if not margin_sufficient:
                # Reduce lots to fit within margin
                lots = max(1, int(effective_margin / (premium_per_lot * self.settings.NIFTY_LOT_SIZE * self.settings.SPAN_MARGIN_PCT)))
                capped_by_margin = True
                logger.info("Lots reduced by margin constraint",
                           available=available_margin, required=required_margin,
                           adjusted_lots=lots)
                # Recompute margin check
                required_margin = lots * premium_per_lot * self.settings.NIFTY_LOT_SIZE * self.settings.SPAN_MARGIN_PCT
                margin_sufficient = required_margin <= effective_margin
                margin_utilization = required_margin / effective_margin if effective_margin > 0 else 1.0

            margin_check = MarginCheck(
                available_margin=available_margin,
                required_margin=required_margin,
                margin_sufficient=margin_sufficient,
                margin_utilization_pct=round(margin_utilization, 4),
            )

        return LotSizeResult(
            lots=lots, method=method,
            risk_per_trade=round(risk_per_trade, 2),
            risk_after_costs=round(risk_after_costs, 2),
            stop_distance_per_lot=stop_distance_per_lot,
            cost_breakdown=cost_breakdown,
            margin_check=margin_check,
            gamma_risk_reduced=gamma_risk_reduced,
            gamma_risk_reduction_pct=gamma_risk_reduction_pct,
            capped_by_max_lots=capped_by_max_lots,
            capped_by_margin=capped_by_margin,
            capped_by_daily_notional=capped_by_daily_notional,
            as_of_date=as_of_date,
        )

6.3 Transaction Cost Estimation (Phase 6 Cost Model, Correction #23)
    def _estimate_transaction_costs(
        self,
        premium_per_lot: float,
        lots: int,
        is_selling: bool,
        num_legs: int = 1,
    ) -> CostBreakdown:
        """Estimate total transaction costs for a trade.

        Uses Phase 6 cost model rates (Corrections #2-#10 in Phase 6 table).
        Reference: Zerodha brokerage calculator 2025; Phase 6 BacktestSettings
        """
        lot_size = self.settings.NIFTY_LOT_SIZE
        premium_value = premium_per_lot * lot_size * lots  # Total premium value
        num_orders = num_legs * 2  # Entry + exit for each leg

        # Brokerage: Rs 20 per executed order
        brokerage = self.settings.BROKERAGE_PER_ORDER * num_orders

        # STT: 0.0625% on premium value sell side (options)
        stt = premium_value * self.settings.STT_SELL_PCT if is_selling else 0.0

        # Stamp duty: 0.003% buy side options
        stamp_duty = premium_value * self.settings.STAMP_DUTY_BUY_PCT

        # Exchange transaction: 0.035% of premium value both sides
        exchange_transaction = premium_value * self.settings.EXCHANGE_TRANSACTION_PCT * 2  # Both sides

        # SEBI turnover: Rs 10 per crore
        sebi_turnover = (premium_value / 1e7) * self.settings.SEBI_TURNOVER_FEE_PER_CRORE * 2  # Both sides

        # SEBI CTCL: Rs 1.50 per lakh sell side
        sebi_ctcl = (premium_value / 1e5) * self.settings.SEBI_CTCL_PER_LAKH if is_selling else 0.0

        # GST: 18% on (brokerage + SEBI fees + exchange charges)
        gst_base = brokerage + sebi_turnover + exchange_transaction + sebi_ctcl
        gst = gst_base * self.settings.GST_PCT

        # Slippage
        slippage = premium_value * self.settings.SLIPPAGE_PCT * 2  # Both entry and exit

        total = brokerage + stt + stamp_duty + exchange_transaction + sebi_turnover + sebi_ctcl + gst + slippage

        return CostBreakdown(
            brokerage=round(brokerage, 2),
            stt=round(stt, 2),
            stamp_duty=round(stamp_duty, 2),
            exchange_transaction=round(exchange_transaction, 2),
            sebi_turnover=round(sebi_turnover, 4),
            sebi_ctcl=round(sebi_ctcl, 4),
            gst=round(gst, 2),
            slippage=round(slippage, 2),
            total=round(total, 2),
        )

6.4 Kelly Fraction Risk
    def _kelly_risk(
        self,
        account_capital: float,
        win_rate: float | None = None,
        avg_win_loss_ratio: float | None = None,
    ) -> float:
        """Compute risk per trade using Kelly fraction with half-Kelly for options.

        Kelly formula: f* = (p × b - q) / b
        where p = win_rate, q = 1 - p, b = avg_win / avg_loss

        Half-Kelly: f* / 2 (options have fat tails, Kaufman Ch.9)

        Reference: Kaufman Ch.9 (Kelly criterion); Correction #16
        """
        p = win_rate if win_rate is not None else 0.55
        b = avg_win_loss_ratio if avg_win_loss_ratio is not None else 1.5
        q = 1.0 - p

        if b <= 0 or p <= 0:
            return account_capital * self.settings.FIXED_FRACTIONAL_PCT

        kelly = (p * b - q) / b
        if kelly <= 0:
            logger.warning("Kelly fraction <= 0, falling back to fixed_fractional", kelly=kelly)
            return account_capital * self.settings.FIXED_FRACTIONAL_PCT

        if self.settings.KELLY_USE_HALF:
            kelly = kelly / 2.0

        return account_capital * kelly

6.5 Optimal-f Risk
    def _optimal_f_risk(
        self,
        account_capital: float,
        trade_history: list[dict] | None = None,
    ) -> float:
        """Compute risk per trade using optimal-f (Ralph Vince method).

        If trade_history is insufficient, falls back to fixed_fractional.

        Reference: Kaufman Ch.9 (optimal-f); Ralph Vince "Portfolio Management Formulas"
        """
        if trade_history is None or len(trade_history) < self.settings.OPTIMAL_F_LOOKBACK_TRADES:
            logger.warning("Insufficient trade history for optimal-f, falling back to fixed_fractional",
                          history_len=len(trade_history) if trade_history else 0)
            return account_capital * self.settings.FIXED_FRACTIONAL_PCT

        losses = [t["pnl"] for t in trade_history if t["pnl"] < 0]
        if not losses:
            largest_loss = -1.0
        else:
            largest_loss = min(losses)

        if largest_loss >= 0:
            return account_capital * self.settings.FIXED_FRACTIONAL_PCT

        best_f = 0.01
        best_geo_mean = -float("inf")

        for f_pct in [i / 100.0 for i in range(1, 100)]:
            geo_product = 1.0
            for trade in trade_history:
                hpr = 1.0 + f_pct * (-trade["pnl"] / largest_loss)
                if hpr <= 0:
                    geo_product = 0
                    break
                geo_product *= hpr
            n = len(trade_history)
            geo_mean = geo_product ** (1.0 / n) if geo_product > 0 and n > 0 else 0
            if geo_mean > best_geo_mean:
                best_geo_mean = geo_mean
                best_f = f_pct

        return account_capital * best_f

8.2 compute_trailing_stop() — PRIMARY ENTRY POINT
    def compute_trailing_stop(
        self,

        2. Apply continuous VIX scaling: trail × (1 + VIX/100 × VIX_SCALE_FACTOR)

            current_price: Current market price
            atr_14: ATR(14) from Phase 3 technical indicators
            as_of_date: Explicit date parameter (Correction #9)
            india_vix: India VIX level (external from NSE)
            market_strength_score: Market strength score 0-100 from Phase 4
            market_regime: Market regime (WEAK/NEUTRAL/STRONG) from Phase 4
            volume_ratio: Current volume / average volume
            iv_rank: IV rank percentile [0, 100] from Phase 2
            is_long: True for long position (stop moves up), False for short (stop moves down)
            previous_stop: Previous trailing stop value (for monotonicity, Correction #18)
            highest_since_entry: Highest price since entry (for Chandelier Exit, long positions)
            lowest_since_entry: Lowest price since entry (for Chandelier Exit, short positions)

        Returns:
            TrailingStopResult with stop price, all adjustments, and order details

        Reference: Kaufman Ch.10 (adaptive stops); Natenberg Ch.5 (IV); Correction #15-#21, #25
        """
        if atr_14 <= 0:
            logger.warning("ATR(14) must be > 0, using entry_price as stop")
            return TrailingStopResult(
                stop_price=entry_price, trail_distance=0.0,
                method=TrailingStopMethod(self.settings.TRAILING_METHOD),
                adjustment_breakdown=TrailAdjustmentBreakdown(base_trail=0.0, final_trail_distance=0.0),
                sl_trigger_price=entry_price, as_of_date=as_of_date,
            )

        # Step 1: Base trail distance
        base_trail = atr_14 * self.settings.BASE_ATR_MULTIPLIER

        # Step 2: VIX scaling (continuous, Correction #17)
        vix_adjustment = (india_vix / 100.0) * self.settings.VIX_SCALE_FACTOR
        trail = base_trail * (1.0 + vix_adjustment)

        # Step 3: Market strength adjustment
        strength_adjustment = (market_strength_score / 100.0) * self.settings.STRENGTH_SCALE_FACTOR
        trail = trail * (1.0 + strength_adjustment)

        # Step 4: Volume spike tightening
        volume_spike_adjustment = 0.0
        if volume_ratio > self.settings.VOLUME_SPIKE_THRESHOLD:
            volume_spike_adjustment = self.settings.VOLUME_SPIKE_TIGHTEN
            trail = trail * (1.0 - volume_spike_adjustment)

        # Step 5: IV rank adjustment (Correction #20, Natenberg Ch.5)
        iv_rank_adjustment = (iv_rank / 100.0) * self.settings.IV_RANK_SCALE_FACTOR
        trail = trail * (1.0 + iv_rank_adjustment)

        # Extra widening for high IV rank
        if iv_rank > self.settings.IV_RANK_HIGH_THRESHOLD:
            extra_widen = self.settings.IV_RANK_HIGH_EXTRA_WIDEN
            trail = trail * (1.0 + extra_widen)
            iv_rank_adjustment += extra_widen

        # Step 6: Regime adjustment (Correction #21)
        regime_factor = self.settings.REGIME_NEUTRAL_FACTOR  # Default
        if market_regime == MarketRegime.STRONG:
            regime_factor = self.settings.REGIME_STRONG_FACTOR
        elif market_regime == MarketRegime.WEAK:
            regime_factor = self.settings.REGIME_WEAK_FACTOR

        regime_adjustment = regime_factor * self.settings.REGIME_SCALE_FACTOR
        trail = trail * (1.0 + regime_adjustment)

        # Step 7: Chandelier Exit trail (Kaufman Ch.10, Correction #19)
        chandelier_trail = 0.0
        if self.settings.TRAILING_METHOD in ("chandelier", "blended"):
            if is_long and highest_since_entry is not None:
                chandelier_trail = highest_since_entry - (atr_14 * self.settings.CHANDELIER_ATR_MULTIPLIER)
            elif not is_long and lowest_since_entry is not None:
                chandelier_trail = (atr_14 * self.settings.CHANDELIER_ATR_MULTIPLIER) + lowest_since_entry
            else:
                # Fallback: use current price as reference
                if is_long:
                    chandelier_trail = current_price - (atr_14 * self.settings.CHANDELIER_ATR_MULTIPLIER)
                else:
                    chandelier_trail = (atr_14 * self.settings.CHANDELIER_ATR_MULTIPLIER) + current_price

        # Step 8: Blended trail (Correction #19)
        blended_trail = 0.0
        final_trail_distance = trail  # Default: ATR-adjusted
        if self.settings.TRAILING_METHOD == "chandelier":
            final_trail_distance = abs(current_price - chandelier_trail) if chandelier_trail > 0 else trail
        elif self.settings.TRAILING_METHOD == "blended":
            atr_trail_distance = trail
            chandelier_trail_distance = abs(current_price - chandelier_trail) if chandelier_trail > 0 else trail
            blended_trail = (
                atr_trail_distance * self.settings.BLENDED_ATR_WEIGHT
                + chandelier_trail_distance * self.settings.BLENDED_CHANDELIER_WEIGHT
            )
            final_trail_distance = blended_trail

        # Step 9: Minimum trail floor
        min_floor = atr_14 * self.settings.MIN_TRAIL_FLOOR_ATR_MULT
        final_trail_distance = max(final_trail_distance, min_floor)

        # Step 10: Compute stop price from reference
        if is_long:
            stop_price = current_price - final_trail_distance
            stop_price = max(stop_price, 0.0)  # Floor at zero
        else:
            stop_price = current_price + final_trail_distance

        # Step 11: Monotonicity enforcement (Correction #18)
        monotonicity_enforced = False
        if self.settings.MONOTONICITY_ENFORCED and previous_stop is not None:
            if is_long and stop_price < previous_stop:
                stop_price = previous_stop
                monotonicity_enforced = True
            elif not is_long and stop_price > previous_stop:
                stop_price = previous_stop
                monotonicity_enforced = True

        # Step 12: SL-M order details (Correction #25)
        sl_trigger_price = stop_price
        sl_limit_price = None
        if self.settings.SL_ORDER_TYPE == "SL-L":
            if is_long:
                sl_limit_price = stop_price * (1.0 - self.settings.SL_LIMIT_BUFFER_PCT)
            else:
                sl_limit_price = stop_price * (1.0 + self.settings.SL_LIMIT_BUFFER_PCT)

        adjustment_breakdown = TrailAdjustmentBreakdown(
            base_trail=round(base_trail, 4),
            vix_adjustment=round(vix_adjustment, 6),
            strength_adjustment=round(strength_adjustment, 6),
            volume_spike_adjustment=round(volume_spike_adjustment, 6),
            iv_rank_adjustment=round(iv_rank_adjustment, 6),
            regime_adjustment=round(regime_adjustment, 6),
            chandelier_trail=round(chandelier_trail, 4),
            blended_trail=round(blended_trail, 4),
            final_trail_distance=round(final_trail_distance, 4),
        )

        return TrailingStopResult(
            stop_price=round(stop_price, 2),
            trail_distance=round(final_trail_distance, 4),
            method=TrailingStopMethod(self.settings.TRAILING_METHOD),
            adjustment_breakdown=adjustment_breakdown,
            monotonicity_enforced=monotonicity_enforced,
            previous_stop=previous_stop,
            sl_order_type=self.settings.SL_ORDER_TYPE,
            sl_trigger_price=round(sl_trigger_price, 2),
            sl_limit_price=round(sl_limit_price, 2) if sl_limit_price is not None else None,
            as_of_date=as_of_date,
            metadata={
                "atr_14": atr_14,
                "india_vix": india_vix,
                "market_strength_score": market_strength_score,
                "market_regime": market_regime.value,
                "volume_ratio": volume_ratio,
                "iv_rank": iv_rank,
                "is_long": is_long,
                "entry_price": entry_price,
            },
        )
