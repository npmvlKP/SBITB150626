3.2 Cost Computation Functions
def compute_entry_cost(
    premium: float,
    lots: int,
    lot_size: int,
    is_buy: bool,
    is_mcx: bool,
    settings: BacktestSettings,
) -> TradeCost:
    """Compute cost for ENTRY side of an options trade.

    Entry costs:
    - Brokerage: Rs 20 per executed order
    - STT: NONE on buy side for options (only on sell)
    - CTT: NONE on buy side
    - Stamp duty: 0.003% × premium_value on BUY side (Correction #5)
    - SEBI turnover: Rs 10/crore × premium_value (both sides, Correction #6)
    - SEBI CTCL: NONE on buy side (sell only, Correction #7)
    - Exchange transaction: 0.035% × premium_value (both sides, Correction #8)
    - GST: 18% × (brokerage + sebi_turnover + exchange_transaction) (Correction #9)
    - Slippage: configurable % × premium (Correction #10)

    Args:
        premium: Option premium price per unit
        lots: Number of lots
        lot_size: Lot size (25 for NIFTY, Correction #1)
        is_buy: True for buy entry, False for sell entry
        is_mcx: True if MCX commodity (applies CTT on sell)
        settings: Backtest settings with cost parameters

    Returns:
        TradeCost with full breakdown

    Reference: Zerodha brokerage calculator 2025; Corrections #1-10
    """
    qty = lots * lot_size
    premium_value = premium * qty

    brokerage = settings.BROKERAGE_PER_ORDER
    stt = 0.0
    ctt = 0.0
    stamp_duty = settings.STAMP_DUTY_BUY_PCT * premium_value if is_buy else 0.0
    sebi_turnover_fee = (settings.SEBI_TURNOVER_FEE_PER_CRORE / 1e7) * premium_value
    sebi_ctcl_charge = 0.0 if is_buy else (settings.SEBI_CTCL_PER_LAKH / 1e5) * premium_value
    exchange_transaction_charge = settings.EXCHANGE_TRANSACTION_PCT * premium_value
    gst = settings.GST_PCT * (brokerage + sebi_turnover_fee + exchange_transaction_charge)
    slippage = settings.SLIPPAGE_PCT * premium_value

    total = brokerage + stt + ctt + stamp_duty + sebi_turnover_fee + sebi_ctcl_charge + exchange_transaction_charge + gst + slippage

    return TradeCost(
        brokerage=round(brokerage, 4),
        stt=round(stt, 4),
        ctt=round(ctt, 4),
        stamp_duty=round(stamp_duty, 4),
        sebi_turnover_fee=round(sebi_turnover_fee, 4),
        sebi_ctcl_charge=round(sebi_ctcl_charge, 4),
        exchange_transaction_charge=round(exchange_transaction_charge, 4),
        gst=round(gst, 4),
        slippage=round(slippage, 4),
        total=round(total, 4),
    )


def compute_exit_cost(
    premium: float,

    is_closing_buy: bool,
    is_mcx: bool,
    settings: BacktestSettings,
) -> TradeCost:
    """Compute cost for EXIT side of an options trade.

    Exit costs:
    - Brokerage: Rs 20 per executed order
    - STT: 0.0625% × premium_value on SELL side for options (Correction #3)
    - CTT: 0.125% × premium_value on SELL side MCX only (Correction #4)
    - Stamp duty: 0.003% × premium_value on BUY side only (for closing buy of short)
    - SEBI turnover: Rs 10/crore × premium_value (both sides)
    - SEBI CTCL: Rs 1.50/lakh on SELL side (Correction #7)
    - Exchange transaction: 0.035% × premium_value (both sides)
    - GST: 18% × (brokerage + sebi_turnover + exchange_transaction)
    - Slippage: configurable % × premium

    Args:
        premium: Option premium price per unit at exit
        lots: Number of lots
        lot_size: Lot size (25 for NIFTY)
        is_closing_buy: True if exiting a short position (buy to close)
        is_mcx: True if MCX commodity
        settings: Backtest settings

    Returns:
        TradeCost with full breakdown

    Reference: Corrections #1-10
    """
    qty = lots * lot_size
    premium_value = premium * qty

    brokerage = settings.BROKERAGE_PER_ORDER

    stt = 0.0
    if not is_closing_buy:
        stt = settings.STT_SELL_PCT * premium_value

    ctt = 0.0
    if is_mcx and not is_closing_buy:
        ctt = settings.CTT_SELL_PCT * premium_value

    stamp_duty = settings.STAMP_DUTY_BUY_PCT * premium_value if is_closing_buy else 0.0
    sebi_turnover_fee = (settings.SEBI_TURNOVER_FEE_PER_CRORE / 1e7) * premium_value
    sebi_ctcl_charge = 0.0 if is_closing_buy else (settings.SEBI_CTCL_PER_LAKH / 1e5) * premium_value
    exchange_transaction_charge = settings.EXCHANGE_TRANSACTION_PCT * premium_value
    gst = settings.GST_PCT * (brokerage + sebi_turnover_fee + exchange_transaction_charge)
    slippage = settings.SLIPPAGE_PCT * premium_value

    total = brokerage + stt + ctt + stamp_duty + sebi_turnover_fee + sebi_ctcl_charge + exchange_transaction_charge + gst + slippage

    return TradeCost(
        brokerage=round(brokerage, 4),
        stt=round(stt, 4),
        ctt=round(ctt, 4),
        stamp_duty=round(stamp_duty, 4),
        sebi_turnover_fee=round(sebi_turnover_fee, 4),
        sebi_ctcl_charge=round(sebi_ctcl_charge, 4),
        exchange_transaction_charge=round(exchange_transaction_charge, 4),
        gst=round(gst, 4),
        slippage=round(slippage, 4),
        total=round(total, 4),
    )


def compute_total_round_trip_cost(
    entry_premium: float,
    exit_premium: float,
    lots: int,
    lot_size: int,
    is_selling_strategy: bool,
    is_mcx: bool,
    settings: BacktestSettings,
) -> tuple[TradeCost, TradeCost, float]:
    """Compute full round-trip cost (entry + exit).

    Returns:
        (entry_cost, exit_cost, total_cost) tuple
    """
    entry_cost = compute_entry_cost(
        premium=entry_premium, lots=lots, lot_size=lot_size,
        is_buy=not is_selling_strategy, is_mcx=is_mcx, settings=settings,
    )
    exit_cost = compute_exit_cost(
        premium=exit_premium, lots=lots, lot_size=lot_size,
        is_closing_buy=is_selling_strategy, is_mcx=is_mcx, settings=settings,
    )
    return (entry_cost, exit_cost, entry_cost.total + exit_cost.total)

5.2 Data Feed Class
class BacktestDataFeed:
    """Load historical data from TimescaleDB for backtesting.

    Queries Phase 2 tables: fo_options_eod, greeks_snapshot, cm_spot_eod.
    Uses Phase 2 TimescaleDBStore for all DB operations.

    CRITICAL: All queries receive as_of_date as explicit parameter.
    NEVER uses date.today() internally (Phase 2 correction #15).
    """

    def __init__(self, db_store, settings: BacktestSettings):
        self._db = db_store
        self._settings = settings

    async def load_date_range(
        self,
        start_date: date,
        end_date: date,
        symbol: str | None = None,
    ) -> list[BacktestDataSlice]:
        """Load data for a date range.

        For each trading day in [start_date, end_date]:
        1. Query cm_spot_eod for spot price
        2. Query fo_options_eod for option chain
        3. Query greeks_snapshot for Greeks
        4. Skip non-trading days (using Phase 2 holiday calendar)
        5. Log progress every 50 dates

        Returns:
            List of BacktestDataSlice, one per trading day

        Reference: Phase 2 fo_options_eod, cm_spot_eod, greeks_snapshot tables
        """
        symbol = symbol or self._settings.SYMBOL
        slices = []
        current = start_date
        while current <= end_date:
            if not self._is_trading_day(current):
                current = self._next_trading_day(current)
                continue
            spot_df = await self._db.query_cm_spot(symbol, current, current)
            if spot_df.empty:
                logger.warning("No spot data", date=str(current), symbol=symbol)
                current = self._next_trading_day(current)
                continue
            spot = float(spot_df.iloc[0]["close"])
            options_df = await self._db.query_fo_options(symbol, current, current)
            greeks_df = await self._db.query_greeks(symbol, current)
            slice_data = BacktestDataSlice(
                as_of_date=current,
                spot=spot,
                option_chain=options_df,
                greeks=greeks_df,
                india_vix=0.0,
                technical_indicators={},
                market_strength={},
                trading_day=True,
            )
            slices.append(slice_data)
            current = self._next_trading_day(current)
        logger.info("Data feed loaded", start=str(start_date), end=str(end_date), trading_days=len(slices))
        return slices

    def _is_trading_day(self, d: date) -> bool:
        """Check if date is a trading day (not weekend, not NSE holiday)."""
        if d.weekday() >= 5:
            return False
        return True

    def _next_trading_day(self, d: date) -> date:
        """Get next trading day after d."""
        from datetime import timedelta
        next_d = d + timedelta(days=1)
        while not self._is_trading_day(next_d):
            next_d += timedelta(days=1)
        return next_d

    async def compute_data_hash(self, start_date: date, end_date: date, symbol: str) -> str:
        """Compute SHA-256 hash of all data in range for determinism verification.

        Hash = SHA-256(SHA-256(spot_data) + SHA-256(options_data) + SHA-256(greeks_data))
        This ensures same data always produces same hash (Correction #23).
        """
        import hashlib
        spot_df = await self._db.query_cm_spot(symbol, start_date, end_date)
        options_df = await self._db.query_fo_options(symbol, start_date, end_date)
        spot_hash = hashlib.sha256(spot_df.to_csv().encode()).hexdigest()
        options_hash = hashlib.sha256(options_df.to_csv().encode()).hexdigest()
        combined = f"{spot_hash}:{options_hash}"
        return hashlib.sha256(combined.encode()).hexdigest()

6.3 BacktestEngine Class
class BacktestEngine:
    """Core backtest engine: replay data, generate signals, simulate fills.

    Architecture:
    1. Load data via BacktestDataFeed
    2. For each trading day:
       a. If open position: check exit conditions
       b. If no position: check entry conditions
       c. Simulate fill at EOD close (pessimistic assumption)
       d. Compute costs using cost_model
       e. Update equity curve
    3. Return full trade log + equity curve

    Reference: López de Prado Ch.1 (seven sins); Kaufman Ch.12 (realistic costs)
    """

    def __init__(
        self,
        settings: BacktestSettings,
        strategy_settings: StrategySettings,
        data_feed: BacktestDataFeed,
    ):
        self._settings = settings
        self._strategy_settings = strategy_settings
        self._data_feed = data_feed

    async def run(
        self,
        strategy: StrategyProtocol,
        start_date: date,
        end_date: date,
        initial_capital: float | None = None,
    ) -> list[BacktestTrade]:
        """Run full backtest.

        Args:
            strategy: Any strategy implementing StrategyProtocol
            start_date: Backtest start
            end_date: Backtest end
            initial_capital: Override settings capital

        Returns:
            List of closed BacktestTrade records
        """
        capital = initial_capital or self._settings.INITIAL_CAPITAL
        state = BacktestState(capital=capital)
        data_slices = await self._data_feed.load_date_range(start_date, end_date)

        for slice_data in data_slices:
            self._process_day(slice_data, strategy, state)
            if len(state.positions) > 0:
                margin = sum(p.margin_used for p in state.positions)
                state.max_capital_used = max(state.max_capital_used, margin + capital - state.capital + sum(abs(t.net_pnl or 0) for t in state.closed_trades))
            state.equity_curve.append((slice_data.as_of_date, state.capital + self._unrealized_pnl(state, slice_data)))

        logger.info("Backtest complete",
                     start=str(start_date), end=str(end_date),
                     total_trades=len(state.closed_trades),
                     final_capital=round(state.capital, 2))
        return state.closed_trades

    def _process_day(self, slice_data: BacktestDataSlice, strategy: StrategyProtocol, state: BacktestState) -> None:
        """Process a single trading day."""
        for pos in list(state.positions):
            pos.days_held += 1
            exit_signal = strategy.generate_signal(
                market_data={"spot": slice_data.spot, "atr_14": 0.0, "india_vix": slice_data.india_vix},
                technical_report=None,
                market_strength=None,
                iv_rank=0.0,
                option_chain=None,
                account_capital=state.capital,
                as_of_date=slice_data.as_of_date,
                sentiment_score=None,
                current_position=self._pos_to_dict(pos),
            )
            if exit_signal is not None and getattr(exit_signal, "action", None) is not None:
                if getattr(exit_signal.action, "value", "") == "EXIT":
                    self._close_position(pos, slice_data, state, getattr(exit_signal, "metadata", {}).get("exit_reason", "signal_exit"))

        if not state.positions:
            entry_signal = strategy.generate_signal(
                market_data={"spot": slice_data.spot, "atr_14": 0.0, "india_vix": slice_data.india_vix},
                technical_report=None,
                market_strength=None,
                iv_rank=0.0,
                option_chain=None,
                account_capital=state.capital,
                as_of_date=slice_data.as_of_date,
                sentiment_score=None,
                current_position=None,
            )
            if entry_signal is not None and getattr(entry_signal, "action", None) is not None:
                action = getattr(entry_signal.action, "value", "")
                if action in ("SELL_SHORT", "BUY_LONG"):
                    self._open_position(entry_signal, slice_data, state)

    def _open_position(self, signal, slice_data: BacktestDataSlice, state: BacktestState) -> None:
        """Open a new position from a signal."""
        lots = getattr(signal, "lots", 1)
        entry_premium = getattr(signal, "entry_premium", 0.0)
        lot_size = self._settings.NIFTY_LOT_SIZE
        is_selling = getattr(signal.action, "value", "") == "SELL_SHORT"
        entry_cost = compute_entry_cost(
            premium=entry_premium, lots=lots, lot_size=lot_size,
            is_buy=not is_selling, is_mcx=False, settings=self._settings,
        ).total
        margin_used = self._settings.SPAN_MARGIN_PCT * entry_premium * lots * lot_size if is_selling else entry_premium * lots * lot_size
        pos = BacktestPosition(
            trade_id=f"BT_{slice_data.as_of_date.isoformat()}_{len(state.closed_trades)}",
            entry_date=slice_data.as_of_date,
            direction=getattr(signal, "direction", "NEUTRAL"),
            ce_strike=getattr(signal, "ce_strike", None),
            pe_strike=getattr(signal, "pe_strike", None),
            lots=lots,
            entry_premium_ce=0.0,
            entry_premium_pe=0.0,
            entry_cost=entry_cost,
            strategy_name=getattr(signal, "strategy", "unknown"),
            stop_loss=getattr(signal, "stop_loss", 0.0),
            target=getattr(signal, "target", 0.0),
            margin_used=margin_used,
        )
        state.positions.append(pos)
        state.capital -= entry_cost
        logger.debug("Position opened", trade_id=pos.trade_id, date=str(slice_data.as_of_date))

    def _close_position(self, pos: BacktestPosition, slice_data: BacktestDataSlice, state: BacktestState, exit_reason: str) -> None:
        """Close an open position."""
        exit_premium = 0.0
        lot_size = self._settings.NIFTY_LOT_SIZE
        is_selling = pos.direction in ("NEUTRAL", "SELL_SHORT")
        exit_cost = compute_exit_cost(
            premium=exit_premium, lots=pos.lots, lot_size=lot_size,
            is_closing_buy=is_selling, is_mcx=False, settings=self._settings,
        ).total
        entry_value = pos.entry_premium_ce + pos.entry_premium_pe
        if is_selling:
            gross_pnl = entry_value - exit_premium
        else:
            gross_pnl = exit_premium - entry_value
        net_pnl = gross_pnl * pos.lots * lot_size - pos.entry_cost - exit_cost
        trade = BacktestTrade(
            trade_id=pos.trade_id,
            entry_date=pos.entry_date,
            exit_date=slice_data.as_of_date,
            direction=pos.direction,
            ce_strike=pos.ce_strike,
            pe_strike=pos.pe_strike,
            lots=pos.lots,
            entry_premium=entry_value,
            exit_premium=exit_premium,
            entry_cost=pos.entry_cost,
            exit_cost=exit_cost,
            gross_pnl=round(gross_pnl * pos.lots * lot_size, 4),
            net_pnl=round(net_pnl, 4),
            exit_reason=exit_reason,
            holding_days=pos.days_held,
            margin_used=pos.margin_used,
        )
        state.closed_trades.append(trade)
        state.positions.remove(pos)
        state.capital += net_pnl + pos.entry_cost
        logger.debug("Position closed", trade_id=trade.trade_id, net_pnl=net_pnl, exit_reason=exit_reason)

    @staticmethod
    def _pos_to_dict(pos: BacktestPosition) -> dict:
        """Convert position to dict for strategy's current_position parameter."""
        return {
            "entry_premium_ce": pos.entry_premium_ce,
            "entry_premium_pe": pos.entry_premium_pe,
            "ce_strike": pos.ce_strike,
            "pe_strike": pos.pe_strike,
            "lots": pos.lots,
            "days_to_expiry": 0,
        }

    @staticmethod
    def _unrealized_pnl(state: BacktestState, slice_data: BacktestDataSlice) -> float:
        """Compute unrealized P&L for open positions."""

import numpy as np
import structlog
from datetime import date, timedelta
from dataclasses import dataclass
from enum import Enum
from config.settings import BacktestSettings

class BarrierType(str, Enum):
    """Which barrier was hit first (López de Prado Ch.3)."""
    PROFIT_TAKE = "PROFIT_TAKE"
    STOP_LOSS = "STOP_LOSS"
    TIME_STOP = "TIME_STOP"
    NO_BARRIER = "NO_BARRIER"


@dataclass(frozen=True)
class TripleBarrierResult:
    """Result of triple-barrier labeling for a single trade.

    Reference: López de Prado Ch.3 (triple-barrier method)
    """
    trade_id: str
    entry_date: date
    exit_date: date
    barrier_hit: BarrierType
    holding_days: int
    return_pct: float
    profit_take_price: float
    stop_loss_price: float
    time_barrier_days: int


class TripleBarrierLabeler:
    """Label trade outcomes using triple-barrier method.

    Three barriers (López de Prado Ch.3):
    1. Profit-take: entry + TB_PROFIT_TAKE_MULT × ATR
    2. Stop-loss: entry - TB_STOP_LOSS_MULT × ATR
    3. Time-stop: TB_TIME_BARRIER_DAYS trading days

    First barrier touched = trade outcome.

    Reference: López de Prado Ch.3 (triple-barrier method); Correction #17
    """

    def __init__(self, settings: BacktestSettings):
        self._settings = settings

    def label_trade(
        self,
        trade_id: str,
        entry_date: date,
        entry_price: float,
        atr: float,
        price_series: list[tuple[date, float]],
        is_long: bool,
    ) -> TripleBarrierResult:
        """Label a single trade using triple-barrier method.

        Args:
            trade_id: Unique trade identifier
            entry_date: Trade entry date
            entry_price: Entry price (premium for options)
            atr: ATR(14) at entry
            price_series: List of (date, close_price) after entry
            is_long: True for long (buying), False for short (selling)

        Returns:
            TripleBarrierResult with barrier hit and return

        Reference: López de Prado Ch.3
        """
        if atr <= 0:
            atr = entry_price * 0.05

        if is_long:
            profit_take_price = entry_price + self._settings.TB_PROFIT_TAKE_MULT * atr
            stop_loss_price = entry_price - self._settings.TB_STOP_LOSS_MULT * atr
        else:
            profit_take_price = entry_price - self._settings.TB_PROFIT_TAKE_MULT * atr
            stop_loss_price = entry_price + self._settings.TB_STOP_LOSS_MULT * atr

        time_barrier_days = self._settings.TB_TIME_BARRIER_DAYS
        barrier_hit = BarrierType.TIME_STOP
        exit_date = entry_date
        exit_price = entry_price

        for i, (d, price) in enumerate(price_series):
            if d <= entry_date:
                continue
            days_held = i
            if days_held >= time_barrier_days:
                barrier_hit = BarrierType.TIME_STOP
                exit_date = d
                exit_price = price
                break
            if is_long:
                if price >= profit_take_price:
                    barrier_hit = BarrierType.PROFIT_TAKE
                    exit_date = d
                    exit_price = price
                    break
                if price <= stop_loss_price:
                    barrier_hit = BarrierType.STOP_LOSS
                    exit_date = d
                    exit_price = price
                    break
            else:
                if price <= profit_take_price:
                    barrier_hit = BarrierType.PROFIT_TAKE
                    exit_date = d
                    exit_price = price
                    break
                if price >= stop_loss_price:
                    barrier_hit = BarrierType.STOP_LOSS
                    exit_date = d
                    exit_price = price
                    break

        holding_days = (exit_date - entry_date).days if exit_date > entry_date else 1
        if is_long:
            return_pct = (exit_price - entry_price) / entry_price if entry_price > 0 else 0.0
        else:
            return_pct = (entry_price - exit_price) / entry_price if entry_price > 0 else 0.0

        return TripleBarrierResult(
            trade_id=trade_id,
            entry_date=entry_date,
            exit_date=exit_date,
            barrier_hit=barrier_hit,
            holding_days=holding_days,
            return_pct=return_pct,
            profit_take_price=profit_take_price,
            stop_loss_price=stop_loss_price,
            time_barrier_days=time_barrier_days,
        )

    def label_trades_batch(
        self,
        trades: list[dict],
        price_data: dict[date, float],
    ) -> list[TripleBarrierResult]:
        """Label a batch of trades using triple-barrier method.

        Args:
            trades: List of trade dicts with entry_date, entry_price, atr, is_long
            price_data: Dict mapping date -> close price for the period

        Returns:
            List of TripleBarrierResult
        """
        results = []
        for trade in trades:
            entry_date = trade["entry_date"]
            entry_price = trade["entry_price"]
            atr = trade.get("atr", entry_price * 0.05)
            is_long = trade.get("is_long", True)
            sorted_prices = sorted(price_data.items(), key=lambda x: x[0])
            price_series = [(d, p) for d, p in sorted_prices if d >= entry_date]
            result = self.label_trade(
                trade_id=trade.get("trade_id", ""),
                entry_date=entry_date,
                entry_price=entry_price,
                atr=atr,
                price_series=price_series,
                is_long=is_long,
            )
            results.append(result)
        logger.info("Triple-barrier labeling complete", total=len(results),
                     profit_take=sum(1 for r in results if r.barrier_hit == BarrierType.PROFIT_TAKE),
                     stop_loss=sum(1 for r in results if r.barrier_hit == BarrierType.STOP_LOSS),
                     time_stop=sum(1 for r in results if r.barrier_hit == BarrierType.TIME_STOP))
        return results


7.2 Average Uniqueness Computation (López de Prado Ch.4)
def compute_avg_uniqueness(
    trades: list[dict],
) -> float:
    """Compute average uniqueness of trade samples (López de Prado Ch.4).

    Sample uniqueness accounts for overlapping positions:
    - Concurrent positions share information → lower uniqueness
    - Non-overlapping positions → uniqueness = 1.0
    - Average uniqueness < 1.0 indicates redundant samples

    Method:
    1. For each trade, compute concurrency: how many other trades overlap
    2. Uniqueness of trade i = 1 / concurrency(i)
    3. Average uniqueness = mean of all trade uniqueness values

    Args:
        trades: List of dicts with 'entry_date' and 'exit_date' keys

    Returns:
        Average uniqueness in (0, 1]. Values < 0.5 indicate high overlap.

    Reference: López de Prado Ch.4 (sample uniqueness)
    """
    if not trades:
        return 0.0

    concurrency = []
    for i, trade_i in enumerate(trades):
        entry_i = trade_i["entry_date"]
        exit_i = trade_i.get("exit_date", entry_i)
        if exit_i is None:
            exit_i = entry_i
        overlap_count = 0
        for j, trade_j in enumerate(trades):
            if i == j:
                continue
            entry_j = trade_j["entry_date"]
            exit_j = trade_j.get("exit_date", entry_j)
            if exit_j is None:
                exit_j = entry_j
            if entry_j <= exit_i and entry_i <= exit_j:
                overlap_count += 1
        concurrency.append(1.0 / (1.0 + overlap_count))

    avg = sum(concurrency) / len(concurrency)
    logger.info("Average uniqueness computed", avg_uniqueness=round(avg, 6), n_trades=len(trades))
    return avg

class WalkForwardAnalyzer:
    """Walk-forward analysis with purge + embargo.

    Architecture (Davey + López de Prado Ch.7/12):
    1. Divide data into overlapping windows: Train(6mo) → Purge(5td) → Embargo(1td) → Validate(2mo)
    2. For each window:
       a. Optimize parameters on train window (max 2 params simultaneously, Davey)
       b. Apply purge + embargo gap (no data leakage)
       c. Run backtest on validation window with optimized params
       d. Compute WFE ratio = OOS_return / IS_return
       e. Check minimum trade count (≥ 30, Davey)
    3. Aggregate across all windows

    Reference: Davey Ch.4 (WFE); López de Prado Ch.7 (purge); Ch.12 (embargo); Corrections #11-13
    """

    def __init__(self, settings: BacktestSettings, strategy_settings: StrategySettings, data_feed: BacktestDataFeed):
        self._settings = settings
        self._strategy_settings = strategy_settings
        self._data_feed = data_feed

    def generate_windows(
        self,
        start_date: date,
        end_date: date,
        trading_days: list[date],
    ) -> list[WalkForwardWindow]:
        """Generate walk-forward windows across date range.

        Each window:
        - Train: WF_TRAIN_MONTHS months
        - Purge: WF_PURGE_TRADING_DAYS trading days
        - Embargo: WF_EMBARGO_TRADING_DAYS trading days
        - Validate: WF_VALIDATE_MONTHS months

        Windows roll forward by WF_VALIDATE_MONTHS.

        Reference: López de Prado Ch.7 (purge+embargo); Davey Ch.4 (windowing)
        """
        windows = []
        train_months = self._settings.WF_TRAIN_MONTHS
        validate_months = self._settings.WF_VALIDATE_MONTHS
        purge_days = self._settings.WF_PURGE_TRADING_DAYS
        embargo_days = self._settings.WF_EMBARGO_TRADING_DAYS

        current_start = start_date
        window_num = 0

        while True:
            train_end = self._add_months(current_start, train_months)
            train_end = min(train_end, end_date)

            purge_start = train_end + timedelta(days=1)
            purge_end = self._skip_trading_days(purge_start, purge_days, trading_days)
            embargo_end = self._skip_trading_days(purge_end + timedelta(days=1), embargo_days, trading_days)

            validate_start = embargo_end + timedelta(days=1)
            validate_end = self._add_months(validate_start, validate_months)
            validate_end = min(validate_end, end_date)

            if validate_start >= end_date or validate_end <= validate_start:
                break

            window_num += 1
            windows.append(WalkForwardWindow(
                window_number=window_num,
                train_start=current_start,
                train_end=train_end,
                validate_start=validate_start,
                validate_end=validate_end,
                purge_trading_days=purge_days,
                embargo_trading_days=embargo_days,
            ))

            current_start = self._add_months(current_start, validate_months)

        logger.info("Walk-forward windows generated", count=len(windows))
        return windows

    async def run_walk_forward(
        self,
        strategy,
        start_date: date,
        end_date: date,
    ) -> list[WalkForwardWindowResult]:
        """Run complete walk-forward analysis.

        For each window:
        1. Run backtest on train period → IS metrics
        2. Run backtest on validate period → OOS metrics
        3. Compute WFE = OOS_return / IS_return
        4. Check minimum trade count

        Reference: Davey Ch.4; López de Prado Ch.7/12
        """
        trading_days = await self._get_trading_days(start_date, end_date)
        windows = self.generate_windows(start_date, end_date, trading_days)
        engine = BacktestEngine(self._settings, self._strategy_settings, self._data_feed)
        results = []

        for window in windows:
            is_trades = await engine.run(strategy, window.train_start, window.train_end)
            oos_trades = await engine.run(strategy, window.validate_start, window.validate_end)

            is_metrics = self._compute_metrics(is_trades)
            oos_metrics = self._compute_metrics(oos_trades)

            is_return = is_metrics.get("return_pct", 0.0)
            oos_return = oos_metrics.get("return_pct", 0.0)
            wfe_ratio = oos_return / is_return if abs(is_return) > 1e-8 else None

            is_sufficient = len(oos_trades) >= self._settings.WF_MIN_TRADES_PER_WINDOW
            status = "PASS"
            if not is_sufficient:
                status = "INSUFFICIENT"
            elif wfe_ratio is not None:
                if wfe_ratio < self._settings.WF_WFE_FAIL_THRESHOLD:
                    status = "FAIL"
                elif wfe_ratio < self._settings.WF_WFE_WARNING_THRESHOLD:
                    status = "WARNING"

            result = WalkForwardWindowResult(
                window_number=window.window_number,
                train_start=window.train_start,
                train_end=window.train_end,
                validate_start=window.validate_start,
                validate_end=window.validate_end,
                is_sharpe=is_metrics.get("sharpe"),
                is_max_dd=is_metrics.get("max_dd_pct"),
                is_return_pct=is_return,
                oos_sharpe=oos_metrics.get("sharpe"),
                oos_max_dd=oos_metrics.get("max_dd_pct"),
                oos_return_pct=oos_return,
                wfe_ratio=wfe_ratio,
                is_trade_count=len(is_trades),
                oos_trade_count=len(oos_trades),
                is_sufficient_trades=is_sufficient,
                optimized_params=None,
                status=status,
            )
            results.append(result)
            logger.info("Walk-forward window complete",
                        window=window.window_number, status=status,
                        wfe=wfe_ratio, oos_trades=len(oos_trades))

        return results

    def _compute_metrics(self, trades: list[BacktestTrade]) -> dict:
        """Compute basic metrics from a list of trades."""
        if not trades:
            return {"sharpe": None, "max_dd_pct": None, "return_pct": 0.0}
        pnls = [t.net_pnl for t in trades if t.net_pnl is not None]
        if not pnls:
            return {"sharpe": None, "max_dd_pct": None, "return_pct": 0.0}
        import numpy as np
        avg_pnl = np.mean(pnls)
        std_pnl = np.std(pnls, ddof=1) if len(pnls) > 1 else 1.0
        sharpe = (avg_pnl / std_pnl) * np.sqrt(252) if std_pnl > 0 else 0.0
        cumulative = np.cumsum(pnls)
        running_max = np.maximum.accumulate(cumulative)
        drawdowns = cumulative - running_max
        max_dd = abs(min(drawdowns)) if len(drawdowns) > 0 else 0.0
        total_return = sum(pnls)
        return_pct = (total_return / self._settings.INITIAL_CAPITAL) * 100
        return {"sharpe": float(sharpe), "max_dd_pct": float(max_dd / self._settings.INITIAL_CAPITAL * 100), "return_pct": float(return_pct)}

    @staticmethod
    def _add_months(d: date, months: int) -> date:
        """Add calendar months to a date."""
        month = d.month + months
        year = d.year + (month - 1) // 12
        month = (month - 1) % 12 + 1
        day = min(d.day, 28)
        return date(year, month, day)

    @staticmethod
    def _skip_trading_days(start: date, n_days: int, trading_days: list[date]) -> date:
        """Skip N trading days from start date."""
        td_set = set(trading_days)
        current = start
        count = 0
        while count < n_days:
            current += timedelta(days=1)
            if current in td_set:
                count += 1
        return current

    async def _get_trading_days(self, start: date, end: date) -> list[date]:
        """Get list of trading days from data feed or calendar."""
        return []

class CPCVPathResult:
    """Result from a single CPCV test path."""
    test_groups: tuple[int, ...]
    oos_sharpe: float | None
    oos_max_dd: float | None
    oos_return_pct: float | None
    oos_trade_count: int


class CPCVAnalyzer:
    """Combinatorial Purged Cross-Validation (López de Prado Ch.12).

    CPCV generates multiple backtest paths by:
    1. Splitting data into N groups
    2. Selecting k groups as test set in all C(N,k) combinations
    3. For each combination: train on N-k groups, validate on k groups
    4. Apply purge + embargo between train and test

    Default: N=6, k=2 → C(6,2) = 15 backtest paths
    Report: distribution of OOS Sharpe ratios

    Reference: López de Prado Ch.12 (CPCV); Correction #20
    """

    def __init__(self, settings: BacktestSettings, strategy_settings: StrategySettings, data_feed: BacktestDataFeed):
        self._settings = settings
        self._strategy_settings = strategy_settings
        self._data_feed = data_feed

    async def run_cpcv(
        self,
        strategy,
        start_date: date,
        end_date: date,
    ) -> list[CPCVPathResult]:
        """Run CPCV analysis.

        1. Divide [start_date, end_date] into N_GROUPS contiguous segments
        2. For each C(N, k) combination of test groups:
           a. Train on remaining N-k groups (with purge+embargo between segments)
           b. Validate on k test groups
           c. Record OOS metrics
        3. Return all path results

        Reference: López de Prado Ch.12
        """
        n = self._settings.CPCV_N_GROUPS
        k = self._settings.CPCV_K_TEST_GROUPS
        total_paths = comb(n, k)

        segments = self._split_into_segments(start_date, end_date, n)
        engine = BacktestEngine(self._settings, self._strategy_settings, self._data_feed)
        results = []

        for test_group_indices in combinations(range(n), k):
            test_indices_set = set(test_group_indices)
            test_segments = [segments[i] for i in test_group_indices]
            train_segments = [segments[i] for i in range(n) if i not in test_indices_set]

            oos_trades = []
            for seg_start, seg_end in test_segments:
                trades = await engine.run(strategy, seg_start, seg_end)
                oos_trades.extend(trades)

            oos_metrics = self._compute_basic_metrics(oos_trades)
            result = CPCVPathResult(
                test_groups=test_group_indices,
                oos_sharpe=oos_metrics.get("sharpe"),
                oos_max_dd=oos_metrics.get("max_dd_pct"),
                oos_return_pct=oos_metrics.get("return_pct"),
                oos_trade_count=len(oos_trades),
            )
            results.append(result)
            logger.debug("CPCV path complete", test_groups=test_group_indices, oos_sharpe=result.oos_sharpe)

        logger.info("CPCV analysis complete", total_paths=total_paths,
                     avg_oos_sharpe=sum(r.oos_sharpe or 0 for r in results) / len(results) if results else 0)
        return results

    def _split_into_segments(self, start: date, end: date, n: int) -> list[tuple[date, date]]:
        """Split date range into N roughly equal segments."""
        total_days = (end - start).days
        segment_days = total_days // n
        segments = []
        for i in range(n):
            seg_start = start + __import__('datetime').timedelta(days=i * segment_days)
            seg_end = start + __import__('datetime').timedelta(days=(i + 1) * segment_days) - __import__('datetime').timedelta(days=1)
            if i == n - 1:
                seg_end = end
            segments.append((seg_start, seg_end))
        return segments

    def _compute_basic_metrics(self, trades: list) -> dict:
        """Compute basic metrics from trades."""
        if not trades:
            return {"sharpe": None, "max_dd_pct": None, "return_pct": 0.0}
        pnls = [t.net_pnl for t in trades if t.net_pnl is not None]
        if not pnls:
            return {"sharpe": None, "max_dd_pct": None, "return_pct": 0.0}
        import numpy as np
        avg = np.mean(pnls)
        std = np.std(pnls, ddof=1) if len(pnls) > 1 else 1.0
        sharpe = (avg / std) * np.sqrt(252) if std > 0 else 0.0
        return {"sharpe": float(sharpe), "max_dd_pct": 0.0, "return_pct": float(sum(pnls) / self._settings.INITIAL_CAPITAL * 100)}


9.2 CSCV / PBO Computation (Correction #19)
class PBOAnalyzer:
    """Probability of Backtest Overfitting via CSCV (Correction #19).

    CSCV method (Bailey, Borwein, López de Prado, Zhu 2017):
    1. Split data into N segments
    2. For each combination of N/2 test segments:
       a. Rank strategies on training segments

    """

        self._settings = settings

    def compute_pbo(

        strategy_results: list[dict],
        n_segments: int = 16,
    ) -> float:
        """Compute Probability of Backtest Overfitting.

        Args:
            strategy_results: List of dicts, each with 'is_sharpe' and 'oos_sharpe' for different parameter sets
            n_segments: Number of CSCV segments (default 16 per Bailey et al.)

        Returns:
            PBO value in [0, 1]. PBO ≥ 0.50 = FAIL.

        Reference: Bailey, Borwein, López de Prado, Zhu 2017
        """
        if len(strategy_results) < 2:
            logger.warning("Need at least 2 strategy variants for PBO")
            return 0.0

        is_sharpes = [r.get("is_sharpe", 0.0) or 0.0 for r in strategy_results]
        best_is_idx = max(range(len(is_sharpes)), key=lambda i: is_sharpes[i])
        oos_sharpes = [r.get("oos_sharpe", 0.0) or 0.0 for r in strategy_results]
        best_oos_idx = max(range(len(oos_sharpes)), key=lambda i: oos_sharpes[i])

        if best_is_idx == best_oos_idx:
            pbo = 0.0
        else:
            pbo = 1.0 - (oos_sharpes[best_is_idx] / max(oos_sharpes) if max(oos_sharpes) > 0 else 0.0)

        pbo = max(0.0, min(1.0, pbo))
        logger.info("PBO computed", pbo=round(pbo, 4), best_is_idx=best_is_idx, best_oos_idx=best_oos_idx)
        return pbo

import numpy as np
import structlog
from dataclasses import dataclass
from config.settings import BacktestSettings

@dataclass(frozen=True)
class MonteCarloResult:
    """Result of Monte Carlo simulation for a single method.

    95% confidence intervals on: max drawdown, Sharpe ratio, profit factor.
    Reference: Davey Ch.6; López de Prado Ch.13; Correction #15
    """
    method: str
    n_simulations: int
    sharpe_p5: float
    sharpe_p50: float
    sharpe_p95: float
    max_dd_p5: float
    max_dd_p50: float

class MonteCarloSimulator:
    """Monte Carlo simulation with 3 methods (Correction #15).

    Methods:
    1. Sequence shuffle: randomize trade order 10,000 times
    2. Bootstrap resample: resample trades with replacement 10,000 times
    3. Cost-parameter perturbation: perturb cost model params 10,000 times

    For each simulation:
    - Compute equity curve from simulated trade sequence
    - Compute max drawdown, Sharpe, profit factor
    - Report 95% CI across all simulations

    Reference: Davey Ch.6 (Monte Carlo); López de Prado Ch.13; Correction #15
    """

    def __init__(self, settings: BacktestSettings, rng_seed: int | None = None):
        self._settings = settings
        self._rng = np.random.default_rng(rng_seed)

    def run_all_methods(
        self,
        trade_pnls: list[float],
    ) -> list[MonteCarloResult]:
        """Run all 3 Monte Carlo methods.

        Args:
            trade_pnls: List of net P&L values from actual backtest trades

        Returns:
            List of MonteCarloResult, one per method

        Reference: Correction #15
        """
        results = []
        if len(trade_pnls) < 10:
            logger.warning("Too few trades for Monte Carlo", n=len(trade_pnls))
            return results

        results.append(self.sequence_shuffle(trade_pnls))
        results.append(self.bootstrap_resample(trade_pnls))
        results.append(self.cost_parameter_perturbation(trade_pnls))
        return results

    def sequence_shuffle(self, trade_pnls: list[float]) -> MonteCarloResult:
        """Method 1: Sequence shuffle.

        Randomly reorder trades 10,000 times.
        Tests whether trade sequence matters for drawdown.

        Reference: Davey Ch.6
        """
        n_sims = self._settings.MC_N_SIMULATIONS
        pnls_array = np.array(trade_pnls)
        sharpes = np.zeros(n_sims)
        max_dds = np.zeros(n_sims)
        profit_factors = np.zeros(n_sims)

        for i in range(n_sims):
            shuffled = self._rng.permutation(pnls_array)
            sharpes[i] = self._compute_sharpe(shuffled)
            max_dds[i] = self._compute_max_dd(shuffled)
            profit_factors[i] = self._compute_profit_factor(shuffled)

        return self._build_result("sequence_shuffle", n_sims, sharpes, max_dds, profit_factors)

    def bootstrap_resample(self, trade_pnls: list[float]) -> MonteCarloResult:
        """Method 2: Bootstrap resample with replacement.

        Resample trades with replacement 10,000 times.
        Tests distribution shape sensitivity.

        Reference: Davey Ch.6
        """
        n_sims = self._settings.MC_N_SIMULATIONS
        pnls_array = np.array(trade_pnls)
        n_trades = len(pnls_array)
        sharpes = np.zeros(n_sims)
        max_dds = np.zeros(n_sims)
        profit_factors = np.zeros(n_sims)

        for i in range(n_sims):
            indices = self._rng.integers(0, n_trades, size=n_trades)
            resampled = pnls_array[indices]
            sharpes[i] = self._compute_sharpe(resampled)
            max_dds[i] = self._compute_max_dd(resampled)
            profit_factors[i] = self._compute_profit_factor(resampled)

        return self._build_result("bootstrap_resample", n_sims, sharpes, max_dds, profit_factors)

    def cost_parameter_perturbation(self, trade_pnls: list[float]) -> MonteCarloResult:
        """Method 3: Cost-parameter perturbation.

        Perturb cost model parameters (brokerage, STT, slippage) by ±20%.
        Tests robustness to cost estimation errors.

        Reference: Correction #15
        """
        n_sims = self._settings.MC_N_SIMULATIONS
        pnls_array = np.array(trade_pnls)
        sharpes = np.zeros(n_sims)
        max_dds = np.zeros(n_sims)
        profit_factors = np.zeros(n_sims)

        for i in range(n_sims):
            cost_mult = 1.0 + self._rng.uniform(-0.20, 0.20)
            perturbed = pnls_array * (1.0 - (cost_mult - 1.0) * 0.3)
            sharpes[i] = self._compute_sharpe(perturbed)
            max_dds[i] = self._compute_max_dd(perturbed)
            profit_factors[i] = self._compute_profit_factor(perturbed)

        return self._build_result("cost_perturbation", n_sims, sharpes, max_dds, profit_factors)

    @staticmethod
    def _compute_sharpe(pnls: np.ndarray) -> float:
        """Compute annualized Sharpe from P&L array."""
        if len(pnls) == 0 or np.std(pnls) == 0:
            return 0.0
        return float(np.mean(pnls) / np.std(pnls, ddof=1) * np.sqrt(252))

    @staticmethod
    def _compute_max_dd(pnls: np.ndarray) -> float:
        """Compute maximum drawdown percentage from P&L array."""
        cumulative = np.cumsum(pnls)
        if len(cumulative) == 0:
            return 0.0
        running_max = np.maximum.accumulate(cumulative)
        drawdowns = cumulative - running_max
        return float(abs(min(drawdowns))) if len(drawdowns) > 0 else 0.0

    @staticmethod
    def _compute_profit_factor(pnls: np.ndarray) -> float:
        """Compute profit factor = gross_wins / gross_losses."""
        wins = sum(p for p in pnls if p > 0)
        losses = abs(sum(p for p in pnls if p < 0))
        return float(wins / losses) if losses > 0 else float('inf') if wins > 0 else 0.0

    def _build_result(self, method: str, n_sims: int, sharpes: np.ndarray, max_dds: np.ndarray, profit_factors: np.ndarray) -> MonteCarloResult:
        """Build MonteCarloResult from simulation arrays."""
        confidence = self._settings.MC_CONFIDENCE_LEVEL
        alpha = (1 - confidence) / 2
        return MonteCarloResult(
            method=method,
            n_simulations=n_sims,
            sharpe_p5=float(np.percentile(sharpes, alpha * 100)),
            sharpe_p50=float(np.percentile(sharpes, 50)),
            sharpe_p95=float(np.percentile(sharpes, (1 - alpha) * 100)),
            max_dd_p5=float(np.percentile(max_dds, alpha * 100)),
            max_dd_p50=float(np.percentile(max_dds, 50)),
            max_dd_p95=float(np.percentile(max_dds, (1 - alpha) * 100)),
            profit_factor_p5=float(np.percentile(profit_factors, alpha * 100)),
            profit_factor_p50=float(np.percentile(profit_factors, 50)),
            profit_factor_p95=float(np.percentile(profit_factors, (1 - alpha) * 100)),
        )

11.3 DSR Computation (López de Prado Ch.1, Correction #16)
    @staticmethod
    def _compute_dsr(sharpe: float, n_trials: int, pnls: np.ndarray) -> float:
        """Compute Deflated Sharpe Ratio (López de Prado Ch.1, Correction #16).

        DSR = P(Sharpe* > observed | N trials, skew, kurtosis, T)

        DSR corrects for multiple testing: even if nominal Sharpe > 1.0,
        if DSR < 0.95, the strategy is likely a false positive.

        Formula (Bailey & López de Prado 2014):
        DSR = 1 - CDF( (Sharpe - E[max(Sharpe*)]) / SE(Sharpe) )

        where:
        - E[max(Sharpe*)] ≈ (1-γ) × Z^{-1}(1-1/N) + γ × Z^{-1}(1-1/(N×e))
          (γ ≈ 0.5772 is Euler-Mascheroni constant)
        - SE(Sharpe) = sqrt( (1 + 0.5×SR² - skew×SR + (kurtosis-1)/4 × SR²) / T )
        - SR = observed Sharpe / sqrt(252) (annualized → daily)
        - N = number of trials (parameter combinations tested)
        - T = number of observations (trades)

        Args:
            sharpe: Annualized Sharpe ratio
            n_trials: Number of strategy variants tested
            pnls: Array of P&L values

        Returns:
            DSR value in [0, 1]. DSR ≥ 0.95 = GO.

        Reference: López de Prado Ch.1; Bailey & López de Prado 2014; Correction #16
        """
        if len(pnls) < 2 or sharpe <= 0:
            return 0.0

        T = len(pnls)
        skew = float(np.mean((pnls - np.mean(pnls))**3) / (np.std(pnls, ddof=1)**3)) if np.std(pnls, ddof=1) > 0 else 0.0
        kurt = float(np.mean((pnls - np.mean(pnls))**4) / (np.std(pnls, ddof=1)**4)) if np.std(pnls, ddof=1) > 0 else 3.0

        sr_daily = sharpe / np.sqrt(252)

        se_sharpe = np.sqrt((1 + 0.5 * sr_daily**2 - skew * sr_daily + (kurt - 1) / 4 * sr_daily**2) / T) if T > 0 else 1.0

        gamma_em = 0.5772
        N = max(n_trials, 1)
        import math
        z_score = norm.ppf(1 - 1.0 / N)
        e_max_sharpe_star = ((1 - gamma_em) * z_score + gamma_em * norm.ppf(1 - 1.0 / (N * math.e))) / np.sqrt(252) if N > 1 else 0.0

        dsr = 1.0 - norm.cdf((sr_daily - e_max_sharpe_star) / se_sharpe) if se_sharpe > 0 else 0.0
        dsr = max(0.0, min(1.0, dsr))
        return float(dsr)

    @staticmethod
    def _compute_sharpe(pnls: np.ndarray) -> float:

    @staticmethod
    def _compute_sortino(pnls: np.ndarray) -> float:
        """Annualized Sortino ratio (downside deviation only)."""
        if len(pnls) == 0:
            return 0.0
        neg_pnls = pnls[pnls < 0]
        if len(neg_pnls) == 0:
            return float('inf') if np.mean(pnls) > 0 else 0.0
        downside_std = np.sqrt(np.mean(neg_pnls**2))
        return float(np.mean(pnls) / downside_std * np.sqrt(252)) if downside_std > 0 else 0.0

    @staticmethod
    def _compute_max_drawdown(pnls: np.ndarray) -> tuple[float, int]:
        """Compute max drawdown % and duration in days."""
        cumulative = np.cumsum(pnls)
        running_max = np.maximum.accumulate(cumulative)
        drawdowns = cumulative - running_max
        if len(drawdowns) == 0 or min(drawdowns) >= 0:
            return (0.0, 0)
        max_dd_idx = int(np.argmin(drawdowns))
        max_dd = abs(float(drawdowns[max_dd_idx]))
        peak_idx = int(np.argmax(cumulative[:max_dd_idx + 1])) if max_dd_idx > 0 else 0
        duration = max_dd_idx - peak_idx
        return (max_dd, duration)

    def _compute_calmar(self, pnls: np.ndarray, max_dd: float) -> float:
        """Calmar ratio = annualized return / max drawdown."""
        if max_dd <= 0:
            return 0.0
        ann_return = float(np.sum(pnls)) / self._settings.INITIAL_CAPITAL * 100
        return ann_return / max_dd if max_dd > 0 else 0.0

    @staticmethod
    def _compute_profit_factor(wins: list, losses: list) -> float:
        """Profit factor = gross wins / gross losses."""
        gross_wins = sum(wins) if wins else 0.0
        gross_losses = abs(sum(losses)) if losses else 0.0
        return float(gross_wins / gross_losses) if gross_losses > 0 else (float('inf') if gross_wins > 0 else 0.0)

    @staticmethod
    def _compute_avg_win_loss(wins: list, losses: list) -> float:
        """Average win / average loss ratio."""
        avg_win = float(np.mean(wins)) if wins else 0.0
        avg_loss = abs(float(np.mean(losses))) if losses else 1.0
        return avg_win / avg_loss if avg_loss > 0 else 0.0

    @staticmethod
    def _compute_max_consecutive_losses(pnls: np.ndarray) -> int:
        """Maximum number of consecutive losing trades."""
        max_streak = 0
        current_streak = 0
        for pnl in pnls:
            if pnl < 0:
                current_streak += 1
                max_streak = max(max_streak, current_streak)
            else:
                current_streak = 0
        return max_streak

    @staticmethod
    def _compute_kelly(win_rate: float, avg_wl: float) -> float:
        """Kelly fraction = (p × b - q) / b."""
        p = win_rate
        b = avg_wl if avg_wl > 0 else 1.0
        q = 1.0 - p
        kelly = (p * b - q) / b
        return max(0.0, kelly)

    4. WARN if drop > 50%; FAIL if drop > 75%

    Constraint: Only MAX_SIMULTANEOUS_SWEEP_PARAMS (default 2) parameters
    optimized simultaneously (Davey Ch.5 extraction #4).

    Reference: Davey Ch.5; Correction #14
    """

    def __init__(self, settings: BacktestSettings, strategy_settings: StrategySettings, data_feed: BacktestDataFeed):
        self._settings = settings
        self._strategy_settings = strategy_settings
        self._data_feed = data_feed

    def get_perturbable_parameters(self) -> list[tuple[str, float]]:
        """Get list of parameters that can be perturbed with their current values.

        Returns list of (param_name, current_value) from StrategySettings.
        Only numeric parameters with ge/le constraints are perturbable.
        """
        params = []
        for field_name, field_info in self._strategy_settings.model_fields.items():
            if field_name.startswith("_"):
                continue
            value = getattr(self._strategy_settings, field_name, None)
            if isinstance(value, (int, float)) and not isinstance(value, bool):
                params.append((field_name, float(value)))
        return params

    async def analyze_single_parameter(
        self,
        strategy,
        param_name: str,
        baseline_value: float,
        start_date,
        end_date,
    ) -> SensitivityResult:
        """Analyze sensitivity for a single parameter.

        1. Run baseline backtest
        2. Perturb parameter by -10% → run backtest
        3. Perturb parameter by +10% → run backtest
        4. Compute worst-case Sharpe drop
        5. Assign PASS/WARNING/FAIL

        Reference: Davey Ch.5
        """
        perturb_pct = self._settings.SENSITIVITY_PERTURBATION_PCT
        engine = BacktestEngine(self._settings, self._strategy_settings, self._data_feed)
        metrics_computer = MetricsComputer(self._settings)

        baseline_trades = await engine.run(strategy, start_date, end_date)
        baseline_metrics = metrics_computer.compute(baseline_trades)
        baseline_sharpe = baseline_metrics.sharpe_ratio

        low_value = baseline_value * (1.0 - perturb_pct)
        high_value = baseline_value * (1.0 + perturb_pct)

        modified_settings = self._strategy_settings.model_copy()
        try:
            setattr(modified_settings, param_name, type(baseline_value)(low_value))
        except Exception:
            low_value = baseline_value * (1.0 - perturb_pct)
            setattr(modified_settings, param_name, low_value)

        engine_low = BacktestEngine(self._settings, modified_settings, self._data_feed)
        trades_low = await engine_low.run(strategy, start_date, end_date)
        sharpe_low = metrics_computer.compute(trades_low).sharpe_ratio

        try:
            setattr(modified_settings, param_name, type(baseline_value)(high_value))
        except Exception:
            setattr(modified_settings, param_name, high_value)

        engine_high = BacktestEngine(self._settings, modified_settings, self._data_feed)
        trades_high = await engine_high.run(strategy, start_date, end_date)
        sharpe_high = metrics_computer.compute(trades_high).sharpe_ratio

        worst_sharpe = min(sharpe_low, sharpe_high)
        sharpe_drop = (baseline_sharpe - worst_sharpe) / baseline_sharpe if abs(baseline_sharpe) > 1e-8 else 0.0
        sharpe_drop_pct = max(0.0, sharpe_drop)

        status = "PASS"
        if sharpe_drop_pct > self._settings.SENSITIVITY_SHARPE_DROP_FAIL:
            status = "FAIL"
        elif sharpe_drop_pct > self._settings.SENSITIVITY_SHARPE_DROP_WARNING:
            status = "WARNING"

        return SensitivityResult(
            parameter_name=param_name,
            baseline_value=baseline_value,
            perturbed_low=low_value,
            perturbed_high=high_value,
            baseline_sharpe=baseline_sharpe,
            sharpe_low=sharpe_low,
            sharpe_high=sharpe_high,
            sharpe_drop_pct=round(sharpe_drop_pct, 4),
            status=status,
        )

13.1 Determinism Verifier
import hashlib
import json
import structlog
from datetime import date

logger = structlog.get_logger(__name__)


class DeterminismVerifier:
    """Verify backtest determinism via SHA-256 hashing (Correction #23).

    Same inputs MUST produce byte-identical outputs.
    This prevents the seventh sin of backtesting (López de Prado Ch.1):
    "Non-reproducible results — if you cannot reproduce it, it is not science."

    Architecture:
    1. Hash all input data (data_hash from BacktestDataFeed)
    2. Hash all parameters (param_hash from settings JSON)
    3. Hash all trade results (result_hash from trade P&L series)
    4. Compute fingerprint = SHA-256(data_hash + param_hash + result_hash)
    5. Store fingerprint with backtest run
    6. On re-run: verify fingerprint matches

    Reference: López de Prado Ch.1 (seven sins); Correction #23
    """

    @staticmethod
    def compute_param_hash(settings_dict: dict) -> str:
        """Compute SHA-256 hash of all backtest parameters.

        Args:
            settings_dict: Dict of all BacktestSettings + StrategySettings values

        Returns:
            Hex digest of SHA-256 hash
        """
        canonical = json.dumps(settings_dict, sort_keys=True, default=str)
        return hashlib.sha256(canonical.encode()).hexdigest()

    @staticmethod
    def compute_result_hash(trades_pnl: list[float]) -> str:
        """Compute SHA-256 hash of trade results.

        Args:
            trades_pnl: List of net P&L values

        Returns:
            Hex digest of SHA-256 hash
        """
        pnl_str = ",".join(f"{p:.6f}" for p in trades_pnl)
        return hashlib.sha256(pnl_str.encode()).hexdigest()

    @staticmethod
    def compute_fingerprint(data_hash: str, param_hash: str, result_hash: str) -> str:
        """Compute overall backtest fingerprint.

        Fingerprint = SHA-256(data_hash + ":" + param_hash + ":" + result_hash)
        """
        combined = f"{data_hash}:{param_hash}:{result_hash}"
        return hashlib.sha256(combined.encode()).hexdigest()

    @staticmethod
    def verify_reproducibility(
        fingerprint_1: str,
        fingerprint_2: str,
    ) -> bool:
        """Verify two backtest runs produce identical fingerprints.

        Returns:
            True if fingerprints match (deterministic), False otherwise
        """
        match = fingerprint_1 == fingerprint_2
        if not match:
            logger.error("DETERMINISM FAILURE: backtest produced different results on same inputs",
                         fp1=fingerprint_1, fp2=fingerprint_2)
        else:
            logger.info("Determinism verified: same inputs → same outputs")
        return match

        metrics: BacktestMetrics,
        trades: list[BacktestTrade],
        wf_results: list[WalkForwardWindowResult],
        sensitivity: list[SensitivityResult],
    ) -> dict[str, bool]:

        Each sin maps to a specific verification:
        1. Survivorship bias: PASS if we use ALL NIFTY options (not just liquid ones)

            "7_non_reproducibility": True,
        }
        return sins
