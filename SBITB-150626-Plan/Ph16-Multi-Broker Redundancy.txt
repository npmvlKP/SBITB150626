```python
# ADD these 5 abstract methods to BrokerInterface class:

@abstractmethod
async def health_check(self) -> dict:
    """Multi-dimensional health check for broker availability.

    Returns dict with keys:
    - is_healthy: bool (True only if ALL checks pass)
    - api_reachable: bool (HTTP endpoint responds)
    - session_valid: bool (access_token/JWT is valid and not expired)
    - margin_available: bool (get_margins() returns positive available margin)
    - order_api_responsive: bool (order placement endpoint responds within timeout)
    - ws_connected: bool (WebSocket is connected, if applicable; False if no WS)
    - supported_segments: list[str] (e.g., ["NSE", "MCX"] for Zerodha/Angel, ["NSE"] for Dhan)
    - response_latency_ms: float (time for health check round-trip in milliseconds)
    - last_checked_timestamp: float (monotonic timestamp of last health check)
    - error_details: str (empty string if healthy; error description if unhealthy)

    Per Kleppmann Ch.8: health check must be multi-dimensional —
    a simple ping is insufficient to determine operational readiness.
    Per Nygard Ch.6: handshake pattern — verify service is actually
    ready to serve requests, not just that it's online.
    """

@abstractmethod
async def get_depth(self, symbol: str, segment: str) -> dict:
    """Get market depth data for a symbol.

    Returns dict with keys:
    - depth_type: str ("level5" for 5-level, "level200" for 200-level)
    - bids: list[dict] (each dict: price, quantity, orders)
    - asks: list[dict] (each dict: price, quantity, orders)
    - timestamp: float

    Note: Only Dhan provides 200-level depth. Zerodha and Angel One
    provide 5-level depth (standard). Callers must handle both cases.
    """

@abstractmethod
async def get_session_status(self) -> dict:
    """Get current session/authentication status.

    Returns dict with keys:
    - is_authenticated: bool
    - token_type: str ("access_token" for Zerodha, "JWT" for Angel One, "access_token" for Dhan)
    - token_expiry_time: float | None (monotonic timestamp when token expires; None if unknown)
    - broker_name: str (e.g., "zerodha", "angel_one", "dhan")
    - user_id: str (broker-assigned user/client ID)
    """

@abstractmethod
def get_broker_name(self) -> str:
    """Return unique broker identifier string.

    Returns: "zerodha" | "angel_one" | "dhan"
    """

@abstractmethod
def get_supported_segments(self) -> list[str]:
    """Return list of segments this broker supports.

    Zerodha: ["NSE", "MCX"]
    Angel One: ["NSE", "MCX"]
    Dhan: ["NSE"] (NO MCX support per Correction #6)
    """
```

```python
class AngelBroker(BrokerInterface):
    """Angel One SmartAPI broker adapter.

    Per Correction #1: Free for account holders, historical data limited to 1 year.
    Per Correction #7: Supports MCX segment (order placement available, WS paid).
    Per Correction #12: Order tag field is 'remarks' (50 char max).
    Per Correction #11: TOTP-based daily re-auth at 6:01 AM IST.

    Implements all 12 BrokerInterface methods (7 original + 5 new from Step 1).
    Every API call is wrapped with CircuitBreaker and retry_with_backoff.

    Args:
        settings: AngelOneSettings with API keys and configuration.
        compliance_settings: ComplianceSettings for rate limits and session times.
        circuit_breaker: CircuitBreaker instance for this broker (independent).
        audit_logger: AuditLogger for audit trail.
        alert_dispatcher: AlertDispatcher for health/failover alerts.
    """

    def __init__(
        self,
        settings: AngelOneSettings,
        compliance_settings: ComplianceSettings,
        circuit_breaker: CircuitBreaker,
        audit_logger: AuditLogger,
        alert_dispatcher: AlertDispatcher,
    ) -> None:
        self._settings = settings
        self._compliance = compliance_settings
        self._circuit_breaker = circuit_breaker
        self._audit = audit_logger
        self._alerts = alert_dispatcher
        self._smart_connect: SmartConnect | None = None
        self._authenticated: bool = False
        self._last_auth_time: float = 0.0

    # --- BrokerInterface implementation ---

    async def authenticate(self) -> str:
        """Authenticate with Angel One SmartAPI.

        Steps:
        1. Create SmartConnect instance with API_KEY
        2. Generate session with CLIENT_CODE, PASSWORD, TOTP
        3. Store JWT token for subsequent API calls
        4. If TOTP generation fails (no TOTP_SECRET), raise and require manual login
        5. Log authentication event to audit trail
        6. Schedule re-auth at 6:01 AM IST (next day)

        Error handling:
        - Invalid credentials: raise AuthenticationError
        - TOTP failure: raise MFARequiredError (requires manual intervention)
        - Network error: raise, let retry_with_backoff handle
        """

    @retry_with_backoff(max_retries=3, base_delay_seconds=1.0, retryable_exceptions=(ConnectionError, TimeoutError))
    async def place_order(self, params: dict) -> dict:
        """Place order via Angel One SmartAPI.

        Maps our internal order params to SmartAPI order params:
        - symbol -> tradingsymbol
        - direction -> transactiontype (BUY/SELL)
        - order_type -> ordertype (MARKET/LIMIT/SL/SL-M)
        - product_type -> product (MIS/NRML/CNC)
        - segment -> exchange (NSE/MCX)
        - tag -> remarks (Correction #12: Angel uses 'remarks' not 'tag')

        Pre-submission checks:
        1. CircuitBreaker.allow_request() — if OPEN, fail fast
        2. Rate limit check (3 req/sec for orders)
        3. Segment validation (if MCX, warn about WS limitations)
        4. Tag field mapping (remarks, max 50 chars)

        Post-submission:
        1. CircuitBreaker.record_success() on 200 response
        2. CircuitBreaker.record_failure() on 4xx/5xx
        3. Log to audit trail
        4. Increment BROKER_API_ERRORS_TOTAL on failure
        """

    @retry_with_backoff(max_retries=3, retryable_exceptions=(ConnectionError, TimeoutError))
    async def cancel_order(self, order_id: str) -> dict:
        """Cancel order via SmartAPI."""

    @retry_with_backoff(max_retries=3, retryable_exceptions=(ConnectionError, TimeoutError))
    async def cancel_all_orders(self) -> list[dict]:
        """Cancel all open orders via SmartAPI.

        Per MiFID II Art.17: kill switch requires cancel_all capability.
        Fetches all open orders, cancels each sequentially.
        """

    @retry_with_backoff(max_retries=3, retryable_exceptions=(ConnectionError, TimeoutError))
    async def get_positions(self) -> list[dict]:
        """Get positions from Angel One.

        Maps Angel position response to standard format:
        - netqty -> quantity
        - tradingsymbol -> symbol
        - exchange -> segment
        - pnl -> unrealized_pnl
        """

    @retry_with_backoff(max_retries=3, retryable_exceptions=(ConnectionError, TimeoutError))
    async def get_margins(self) -> dict:

        Maps Angel margin response to standard format:
        - availablecash -> available_margin
        - utilizedmargins -> used_margin
        """

    @retry_with_backoff(max_retries=3, retryable_exceptions=(ConnectionError, TimeoutError))
    async def get_order_book(self) -> list[dict]:
        """Get current day order book from Angel One."""

    @retry_with_backoff(max_retries=3, retryable_exceptions=(ConnectionError, TimeoutError))
    async def get_instruments(self, segment: str) -> list[dict]:
        """Get instruments for segment from Angel One."""

    async def health_check(self) -> dict:
        """Multi-dimensional health check for Angel One.

        Checks (in order):
        1. API reachable: SmartConnect.profile() call
        2. Session valid: check JWT not expired (6 AM IST refresh)
        3. Margin available: get_margins() returns positive cash
```

```python
    """Cross-broker position reconciliation for failover safety.

    Per Correction #10: The MOST CRITICAL safety step during failover.
    Per MiFID II Art.25: record keeping — positions must be verifiable.

    1. Position count matches between brokers
    2. Each position's symbol, quantity, and direction matches
    3. P&L values match within tolerance (Rs 500 per Correction #10)
    4. Margin values match within tolerance (5% difference allowed)
```
