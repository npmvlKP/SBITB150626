    Supports both standard and mini contracts (configurable via LOT_SIZE_MODE).

async def check_mcx_session_break(
    now: datetime,
    settings: ComplianceSettings,
) -> bool:
    """Return True if currently in MCX session break (14:30–17:00).
    
    During session break, NO orders are allowed for MCX segment.
    Per SEBI/HO/CDMRD/DMP/CIR/P/2016/97.
    
    Args:
        now: Current datetime (IST).
        settings: Compliance settings with MCX session times.
    
    Returns:
        True if in session break, False otherwise.
    """
    current_time = now.time()
    return (
        settings.MCX_TRADING_END_MORNING_IST
        <= current_time
        < settings.MCX_TRADING_START_EVENING_IST
    )


async def validate_mcx_product_type(
    instrument: MCXInstrument,
    product_type: MCXProductType,
    settings: MCXSettings,
) -> bool:
    """Validate product type is allowed for the MCX instrument.
    
    CRUDEOIL: only MIS (intraday) allowed.
    GOLD/SILVER: MIS or NRML allowed.
    BO (bracket order): ALWAYS rejected (Zerodha disabled BO in 2021).
    CO (cover order): ALWAYS rejected (not supported for MCX).
    
    Args:
        instrument: MCX instrument enum.
        product_type: Requested product type.
        settings: MCX settings with allowed product types per instrument.
    
    Returns:
        True if product type is valid for this instrument, False otherwise.
    """
    if product_type in (MCXProductType.BO, MCXProductType.CO):
        return False  # Zerodha bracket/cover orders disabled since 2021
    allowed_map = {
        MCXInstrument.GOLD: settings.GOLD_PRODUCT_TYPES,
        MCXInstrument.GOLDM: settings.GOLD_PRODUCT_TYPES,
        MCXInstrument.SILVER: settings.SILVER_PRODUCT_TYPES,
        MCXInstrument.SILVERM: settings.SILVER_PRODUCT_TYPES,
        MCXInstrument.CRUDEOIL: settings.CRUDEOIL_PRODUCT_TYPES,
        MCXInstrument.CRUDEOILM: settings.CRUDEOIL_PRODUCT_TYPES,
    }
    allowed = allowed_map.get(instrument, [])
    return product_type.value in allowed

def get_lot_size(
    instrument: MCXInstrument,
    settings: MCXSettings,
) -> int:
    """Return lot size for instrument based on LOT_SIZE_MODE setting.
    
    Per MCX circulars:
    - Standard: GOLD=100, SILVER=30, CRUDEOIL=100
    - Mini:     GOLDM=10, SILVERM=5, CRUDEOILM=10
    
    Args:
        instrument: MCX instrument enum.
        settings: MCX settings with lot size configuration.
    
    Returns:
        Lot size in units.
    """
    lot_map_standard = {
        MCXInstrument.GOLD: settings.GOLD_LOT_SIZE,
        MCXInstrument.GOLDM: settings.GOLD_MINI_LOT_SIZE,
        MCXInstrument.SILVER: settings.SILVER_LOT_SIZE,
        MCXInstrument.SILVERM: settings.SILVER_MINI_LOT_SIZE,
        MCXInstrument.CRUDEOIL: settings.CRUDEOIL_LOT_SIZE,
        MCXInstrument.CRUDEOILM: settings.CRUDEOIL_MINI_LOT_SIZE,
    }
    return lot_map_standard[instrument]


def get_effective_instrument(
    instrument: MCXInstrument,
    settings: MCXSettings,
) -> MCXInstrument:
    """Map instrument to correct contract based on lot size mode.
    
    If LOT_SIZE_MODE="mini", GOLD maps to GOLDM, SILVER to SILVERM, etc.
    If LOT_SIZE_MODE="standard", returns instrument as-is.
    
    Args:
        instrument: Requested instrument.
        settings: MCX settings.
    
    Returns:
        Effective instrument (standard or mini variant).
    """
    if settings.LOT_SIZE_MODE == "mini":
        mini_map = {
            MCXInstrument.GOLD: MCXInstrument.GOLDM,
            MCXInstrument.SILVER: MCXInstrument.SILVERM,
            MCXInstrument.CRUDEOIL: MCXInstrument.CRUDEOILM,
        }
        return mini_map.get(instrument, instrument)
    return instrument


def compute_notional_value(
    price: Decimal,
    instrument: MCXInstrument,
    quantity_lots: int,
    settings: MCXSettings,
) -> Decimal:
    """Compute notional value = price * lot_size * quantity_lots.
    
    Args:
        price: Price per unit.
        instrument: MCX instrument.
        quantity_lots: Number of lots.
        settings: MCX settings.
    
    Returns:
        Notional value in INR.
    """
    lot_size = get_lot_size(instrument, settings)
    return price * Decimal(lot_size) * Decimal(quantity_lots)

@dataclass(frozen=True)
class MCXTradeCost:
    """Complete cost breakdown for an MCX trade."""
    brokerage: Decimal
    ctt: Decimal                     # CTT on sell side only
    exchange_txn_charge: Decimal
    gst: Decimal
    sebi_turnover_fee: Decimal
    stamp_duty: Decimal
    total_cost: Decimal
    break_even_ticks: Decimal        # Minimum price ticks to cover costs


def compute_trade_cost(
    side: str,
    price: Decimal,
    quantity_lots: int,
    instrument: MCXInstrument,
    mcx_settings: MCXSettings,
    cost_settings: MCXCostSettings,
) -> MCXTradeCost:
    """Compute full transaction cost for an MCX trade.
    
    Args:
        side: "BUY" or "SELL".
        price: Price per unit.
        quantity_lots: Number of lots.
        instrument: MCX instrument.
        mcx_settings: MCX settings (lot sizes).
        cost_settings: MCX cost settings (rates).
    
    Returns:
        MCXTradeCost with complete cost breakdown.
    
    Cost computation:
    1. notional = price * lot_size * quantity_lots
    2. brokerage = Rs 20 per executed order
    3. CTT = 0 if BUY; notional * 0.00125 if SELL
    4. exchange_txn = notional * exchange_txn_rate
    5. gst = (brokerage + exchange_txn) * 0.18
    6. sebi_fee = notional * 0.000001 if SELL; 0 if BUY
    7. stamp_duty = notional * 0.00002 if BUY; 0 if SELL
    8. total = sum of all above
    9. break_even_ticks = total / (lot_size * quantity_lots * tick_size)
    """
    notional = price * Decimal(get_lot_size(instrument, mcx_settings)) * Decimal(quantity_lots)
    
    brokerage = cost_settings.BROKERAGE_PER_ORDER
    ctt = Decimal("0") if side == "BUY" else notional * cost_settings.CTT_RATE_SELL
    exchange_txn = notional * cost_settings.EXCHANGE_TXN_RATE
    gst = (brokerage + exchange_txn) * cost_settings.GST_RATE
    sebi_fee = Decimal("0") if side == "BUY" else notional * cost_settings.SEBI_TURNOVER_FEE_RATE
    stamp_duty = notional * cost_settings.STAMP_DUTY_RATE_BUY if side == "BUY" else Decimal("0")
    total = brokerage + ctt + exchange_txn + gst + sebi_fee + stamp_duty
    
    tick_size = _get_tick_size(instrument, mcx_settings)
    lot_size = get_lot_size(instrument, mcx_settings)
    break_even_ticks = total / (Decimal(lot_size) * Decimal(quantity_lots) * tick_size)
    
    return MCXTradeCost(
        brokerage=brokerage,
        ctt=ctt,
        exchange_txn_charge=exchange_txn,
        gst=gst,
        sebi_turnover_fee=sebi_fee,
        stamp_duty=stamp_duty,
        total_cost=total,
        break_even_ticks=break_even_ticks,
    )


def compute_round_trip_cost(
    price: Decimal,
    quantity_lots: int,
    instrument: MCXInstrument,
    mcx_settings: MCXSettings,
    cost_settings: MCXCostSettings,
) -> MCXTradeCost:
    """Compute round-trip (buy+sell) cost for MCX trade.
    
    Args:
        price: Entry price (assumes same exit price for cost calc).
        quantity_lots: Number of lots.
        instrument: MCX instrument.
        mcx_settings: MCX settings.
        cost_settings: MCX cost settings.
    
    Returns:
        Combined MCXTradeCost for buy + sell.
    """
    buy_cost = compute_trade_cost("BUY", price, quantity_lots, instrument, mcx_settings, cost_settings)
    sell_cost = compute_trade_cost("SELL", price, quantity_lots, instrument, mcx_settings, cost_settings)
    return MCXTradeCost(
        brokerage=buy_cost.brokerage + sell_cost.brokerage,
        ctt=buy_cost.ctt + sell_cost.ctt,
        exchange_txn_charge=buy_cost.exchange_txn_charge + sell_cost.exchange_txn_charge,
        gst=buy_cost.gst + sell_cost.gst,
        sebi_turnover_fee=buy_cost.sebi_turnover_fee + sell_cost.sebi_turnover_fee,
        stamp_duty=buy_cost.stamp_duty + sell_cost.stamp_duty,
        total_cost=buy_cost.total_cost + sell_cost.total_cost,
        break_even_ticks=buy_cost.break_even_ticks + sell_cost.break_even_ticks,
    )
