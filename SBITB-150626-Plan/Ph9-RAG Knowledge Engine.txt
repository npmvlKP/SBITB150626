  """RAG Knowledge Engine — Configuration (Pydantic Settings).

  All configuration loaded from environment variables with sane defaults.
  ZERO hardcoded secrets. Validates on instantiation.
  """

  from __future__ import annotations

  from pathlib import Path
  from pydantic import Field, field_validator
  from pydantic_settings import BaseSettings, SettingsConfigDict


  class RAGConfig(BaseSettings):
      model_config = SettingsConfigDict(
          env_prefix="RAG_",
          env_file=".env",
          env_file_encoding="utf-8",
          extra="forbid",
          frozen=True,
      )

      # ── LLM Settings ──
      ollama_base_url: str = Field(
          default="http://localhost:11434",
          description="Ollama API base URL",
      )
      ollama_model: str = Field(
          default="qwen2.5:7b-instruct",
          description="Ollama model identifier for generation",
      )
      ollama_timeout_seconds: int = Field(
          default=120,
          ge=10,
          le=600,
          description="Timeout for Ollama LLM generation requests",
      )
      ollama_temperature: float = Field(
          default=0.1,
          ge=0.0,
          le=2.0,
          description="LLM temperature — low for factual RAG responses",
      )
      ollama_num_ctx: int = Field(
          default=4096,
          ge=512,
          le=131072,
          description="Context window size for Qwen 2.5 (supports up to 128K)",
      )

      # ── Embedding Settings ──
      embedding_model_name: str = Field(
          default="sentence-transformers/all-MiniLM-L6-v2",
          description="HuggingFace sentence-transformer model for embeddings",
      )
      embedding_device: str = Field(
          default="cpu",
          description="Device for embedding inference: cpu or cuda",
      )
      embedding_batch_size: int = Field(
          default=32,
          ge=1,
          le=256,
          description="Batch size for embedding computation",
      )

      # ── ChromaDB Settings ──
      chroma_persist_directory: Path = Field(
          default=Path("data/chroma_db"),
          description="ChromaDB persistent storage directory",
      )
      chroma_collection_name: str = Field(
          default="indian_markets_knowledge",
          description="ChromaDB collection name for market knowledge",
      )
      chroma_distance_function: str = Field(
          default="cosine",
          description="Distance function: cosine, l2, or ip",
      )

      # ── Retrieval Settings ──
      retrieval_top_k: int = Field(
          default=5,
          ge=1,
          le=50,
          description="Number of documents to retrieve from vector store",
      )
      retrieval_score_threshold: float = Field(
          default=0.3,
          ge=0.0,
          le=1.0,
          description="Minimum similarity score for retrieval candidates",
      )
      retrieval_search_type: str = Field(
          default="mmr",

  """RAG Knowledge Engine — Embedding Model Wrapper.

  Wraps sentence-transformers/all-MiniLM-L6-v2 with:
  - Lazy singleton initialization (load once, reuse)
  - Device selection (CPU/CUDA)
  - Batch embedding with error handling
  - Dimension validation (384 for all-MiniLM-L6-v2)
  """

  from __future__ import annotations

  import logging
  from functools import lru_cache
  from typing import Any

  from rag_knowledge.config import RAGConfig, get_config
  from rag_knowledge.exceptions import EmbeddingModelError

  logger = logging.getLogger(__name__)

  EXPECTED_DIMENSION = 384


  class EmbeddingEngine:
      def __init__(self, config: RAGConfig | None = None) -> None:
          self._config = config or get_config()
          self._model: Any = None

      def _load_model(self) -> Any:
          if self._model is not None:
              return self._model
          try:
              from sentence_transformers import SentenceTransformer

              logger.info(
                  "Loading embedding model: %s on %s",
                  self._config.embedding_model_name,
                  self._config.embedding_device,
              )
              self._model = SentenceTransformer(
                  model_name=self._config.embedding_model_name,
                  device=self._config.embedding_device,
              )
              dim = self._model.get_sentence_embedding_dimension()
              if dim != EXPECTED_DIMENSION:
                  msg = (
                      f"Embedding dimension mismatch: expected {EXPECTED_DIMENSION}, "
                      f"got {dim} from model {self._config.embedding_model_name}"
                  )
                  raise EmbeddingModelError(msg)
              logger.info("Embedding model loaded. Dimension=%d", dim)
              return self._model
          except EmbeddingModelError:
              raise
          except Exception as e:
              msg = f"Failed to load embedding model '{self._config.embedding_model_name}': {e}"
              raise EmbeddingModelError(msg) from e

      @property
      def model(self) -> Any:
          return self._load_model()

      def embed_query(self, text: str) -> list[float]:
          if not text or not text.strip():
              msg = "Cannot embed empty query text"
              raise EmbeddingModelError(msg)
          try:
              embedding = self.model.encode(text, show_progress_bar=False)
              return embedding.tolist()
          except Exception as e:
              msg = f"Failed to embed query: {e}"
              raise EmbeddingModelError(msg) from e

      def embed_documents(self, texts: list[str]) -> list[list[float]]:
          if not texts:
              return []
          try:
              embeddings = self.model.encode(
                  texts,
                  batch_size=self._config.embedding_batch_size,
                  show_progress_bar=False,
              )
              return embeddings.tolist()
          except Exception as e:
              msg = f"Failed to embed {len(texts)} documents: {e}"
              raise EmbeddingModelError(msg) from e

      @property
      def dimension(self) -> int:
          return EXPECTED_DIMENSION


  @lru_cache(maxsize=1)
  def get_embedding_engine(config: RAGConfig | None = None) -> EmbeddingEngine:
      return EmbeddingEngine(config=config)

  """RAG Knowledge Engine — ChromaDB Vector Store Manager.

  Persistent local ChromaDB with:
  - Cosine similarity default
  - Collection lifecycle (create, get, reset)
  - Batch insertion with metadata
  - MMR and similarity search

  import logging
  from typing import Any

  from rag_knowledge.config import RAGConfig, get_config
  from rag_knowledge.exceptions import (
      ChromaDBCollectionNotFoundError,
      ChromaDBError,
  )

  logger = logging.getLogger(__name__)


  class VectorStore:
      def __init__(self, config: RAGConfig | None = None) -> None:
          self._config = config or get_config()
          self._client: Any = None
          self._collection: Any = None

      def _get_client(self) -> Any:
          if self._client is not None:
              return self._client
          try:
              import chromadb

              persist_dir = str(self._config.chroma_persist_directory)
              self._client = chromadb.PersistentClient(path=persist_dir)
              logger.info("ChromaDB client initialized at: %s", persist_dir)
              return self._client
          except Exception as e:
              msg = f"Failed to initialize ChromaDB client: {e}"
              raise ChromaDBError(msg) from e

      def get_collection(self) -> Any:
          if self._collection is not None:
              return self._collection
          client = self._get_client()
          try:
              self._collection = client.get_or_create_collection(
                  name=self._config.chroma_collection_name,
                  metadata={"hnsw:space": self._config.chroma_distance_function},
              )
              logger.info(
                  "ChromaDB collection '%s' loaded. Count=%d",
                  self._config.chroma_collection_name,
                  self._collection.count(),
              )
              return self._collection
          except Exception as e:
              msg = f"Failed to get/create ChromaDB collection '{self._config.chroma_collection_name}': {e}"
              raise ChromaDBError(msg) from e

      def add_documents(
          self,
          ids: list[str],
          documents: list[str],
          embeddings: list[list[float]],
          metadatas: list[dict[str, Any]],
      ) -> int:
          if not ids:
              return 0
          collection = self.get_collection()
          try:
              batch_size = self._config.ingestion_batch_size
              total_added = 0
              for i in range(0, len(ids), batch_size):
                  batch_ids = ids[i : i + batch_size]
                  batch_docs = documents[i : i + batch_size]
                  batch_embs = embeddings[i : i + batch_size]
                  batch_meta = metadatas[i : i + batch_size]
                  collection.add(
                      ids=batch_ids,
                      documents=batch_docs,
                      embeddings=batch_embs,
                      metadatas=batch_meta,
                  )
                  total_added += len(batch_ids)
              logger.info("Added %d chunks to ChromaDB", total_added)
              return total_added
          except Exception as e:
              msg = f"Failed to add documents to ChromaDB: {e}"
              raise ChromaDBError(msg) from e

      def similarity_search(
          self,
          query_embedding: list[float],
          top_k: int | None = None,
      ) -> dict[str, Any]:
          collection = self.get_collection()
          k = top_k or self._config.retrieval_top_k
          try:
              results = collection.query(
                  query_embeddings=[query_embedding],
                  n_results=k,
                  include=["documents", "metadatas", "distances"],
              )
              return results
          except Exception as e:
              msg = f"ChromaDB similarity search failed: {e}"
              raise ChromaDBError(msg) from e

      def mmr_search(
          self,
          query_embedding: list[float],
          top_k: int | None = None,
      ) -> dict[str, Any]:
          collection = self.get_collection()
          k = top_k or self._config.retrieval_top_k
          try:
              results = collection.query(
                  query_embeddings=[query_embedding],
                  n_results=k * 3,
                  include=["documents", "metadatas", "distances"],
              )
              selected_indices = self._mmr_select(
                  query_embedding,
                  results,
                  k,
              )
              filtered: dict[str, Any] = {"ids": [[]], "documents": [[]], "metadatas": [[]], "distances": [[]]}
              for idx in selected_indices:
                  filtered["documents"][0].append(results["documents"][0][idx])
                  filtered["metadatas"][0].append(results["metadatas"][0][idx])
                  filtered["distances"][0].append(results["distances"][0][idx])
                  if "ids" in results:
                      filtered["ids"][0].append(results["ids"][0][idx])
              return filtered
          except Exception as e:
              msg = f"ChromaDB MMR search failed: {e}"
              raise ChromaDBError(msg) from e

      def _mmr_select(
          self,
          query_embedding: list[float],
          results: dict[str, Any],
          k: int,
      ) -> list[int]:
          import numpy as np

          lambda_param = self._config.retrieval_mmr_lambda
          doc_embeddings_raw = results.get("embeddings", [None])
          if doc_embeddings_raw[0] is None:
              return list(range(min(k, len(results["documents"][0]))))

          doc_embs = np.array(doc_embeddings_raw[0])
          query_emb = np.array(query_embedding)

          n = doc_embs.shape[0]
          selected: list[int] = []
          remaining = list(range(n))

          sim_to_query = np.dot(doc_embs, query_emb) / (
              np.linalg.norm(doc_embs, axis=1) * np.linalg.norm(query_emb) + 1e-10
          )

          for _ in range(min(k, n)):
              if not remaining:
                  break
              best_idx = -1
              best_score = float("-inf")
              for idx in remaining:
                  relevance = lambda_param * sim_to_query[idx]
                  diversity = 0.0
                  if selected:
                      sims = np.dot(doc_embs[idx], doc_embs[selected].T) / (
                          np.linalg.norm(doc_embs[idx])
                          * np.linalg.norm(doc_embs[selected], axis=1)
                          + 1e-10
                      )
                      diversity = (1 - lambda_param) * float(np.max(sims))
                  score = relevance - diversity
                  if score > best_score:
                      best_score = score
                      best_idx = idx
              if best_idx >= 0:
                  selected.append(best_idx)
                  remaining.remove(best_idx)
          return selected

      def reset_collection(self) -> None:
          client = self._get_client()
          try:
              client.delete_collection(self._config.chroma_collection_name)
              logger.warning("ChromaDB collection '%s' DELETED", self._config.chroma_collection_name)
          except Exception as e:
              msg = f"Failed to reset collection: {e}"
              raise ChromaDBError(msg) from e
          self._collection = None

      def count(self) -> int:
          collection = self.get_collection()
          return collection.count()

  """RAG Knowledge Engine — Document Ingestion Pipeline.

  Loads, chunks, embeds, and stores documents into ChromaDB.
  Supported formats: PDF, DOCX, TXT, MD
  Chunks use RecursiveCharacterTextSplitter with overlap.
  Metadata includes: source_path, doc_type, page_number, chunk_index.
  """

  from __future__ import annotations

  import hashlib
  import logging
  import time
  from pathlib import Path
  from typing import Any

  from rag_knowledge.config import RAGConfig, get_config
  from rag_knowledge.embeddings import EmbeddingEngine, get_embedding_engine
  from rag_knowledge.exceptions import IngestionError, IngestionFileError
  from rag_knowledge.models import IngestResult
  from rag_knowledge.vectorstore import VectorStore

  logger = logging.getLogger(__name__)

  DOC_TYPE_MAP: dict[str, str] = {
      "rbi_policy": {".pdf"},
      "sebi_circular": {".pdf"},
      "trade_journal": {".md", ".txt"},
      "strategy_notes": {".txt", ".md"},
      "research": {".pdf", ".docx"},
  }


  class DocumentIngester:
      def __init__(
          self,
          config: RAGConfig | None = None,
          embedding_engine: EmbeddingEngine | None = None,
          vector_store: VectorStore | None = None,
      ) -> None:
          self._config = config or get_config()
          self._embedding_engine = embedding_engine or get_embedding_engine(config=self._config)
          self._vector_store = vector_store or VectorStore(config=self._config)

      def ingest_directory(self, directory: Path) -> IngestResult:
          if not directory.exists():
              msg = f"Ingestion directory does not exist: {directory}"
              raise IngestionError(msg)
          start = time.monotonic()
          total_chunks = 0
          total_sources = 0
          errors: list[str] = []

          supported_extensions = {".pdf", ".docx", ".txt", ".md"}
          files = [
              f
              for f in directory.rglob("*")
              if f.is_file() and f.suffix.lower() in supported_extensions
          ]

          if not files:
              logger.warning("No supported documents found in: %s", directory)
              return IngestResult(
                  documents_ingested=0,
                  sources_processed=0,
                  errors=["No supported documents found"],
                  duration_seconds=0.0,
              )

          for file_path in files:
              try:
                  chunks = self._load_and_chunk(file_path)
                  if not chunks:
                      errors.append(f"No text extracted from: {file_path.name}")
                      continue
                  doc_type = self._infer_doc_type(file_path)
                  chunk_ids = [self._generate_chunk_id(file_path, i) for i in range(len(chunks))]
                  metadatas = [
                      {
                          "source_path": str(file_path),
                          "doc_type": doc_type,
                          "chunk_index": str(i),
                          "file_name": file_path.name,
                          "file_hash": self._file_hash(file_path),
                      }
                      for i in range(len(chunks))
                  ]
                  embeddings = self._embedding_engine.embed_documents(chunks)
                  self._vector_store.add_documents(
                      ids=chunk_ids,
                      documents=chunks,
                      embeddings=embeddings,
                      metadatas=metadatas,
                  )
                  total_chunks += len(chunks)
                  total_sources += 1
                  logger.info("Ingested: %s → %d chunks", file_path.name, len(chunks))
              except Exception as e:
                  error_msg = f"Failed to ingest {file_path.name}: {e}"
                  logger.error(error_msg)
                  errors.append(error_msg)

          duration = time.monotonic() - start
          return IngestResult(
              documents_ingested=total_chunks,
              sources_processed=total_sources,
              errors=errors,
              duration_seconds=round(duration, 3),
          )

      def _load_and_chunk(self, file_path: Path) -> list[str]:
          raw_text = self._extract_text(file_path)
          if not raw_text or not raw_text.strip():
              return []
          from langchain_text_splitters import RecursiveCharacterTextSplitter

          splitter = RecursiveCharacterTextSplitter(
              chunk_size=self._config.chunk_size,
              chunk_overlap=self._config.chunk_overlap,
              separators=["\n\n", "\n", ". ", " ", ""],
              length_function=len,
          )
          chunks = splitter.split_text(raw_text)
          return [c.strip() for c in chunks if c.strip()]

      def _extract_text(self, file_path: Path) -> str:
          suffix = file_path.suffix.lower()
          try:
              if suffix == ".pdf":
                  return self._extract_pdf(file_path)
              elif suffix == ".docx":
                  return self._extract_docx(file_path)
              elif suffix in {".txt", ".md"}:
                  return file_path.read_text(encoding="utf-8", errors="replace")
              else:
                  msg = f"Unsupported file format: {suffix}"
                  raise IngestionFileError(msg)
          except IngestionFileError:
              raise
          except Exception as e:
              msg = f"Text extraction failed for {file_path.name}: {e}"
              raise IngestionFileError(msg) from e

      def _extract_pdf(self, file_path: Path) -> str:
          try:
              from pypdf import PdfReader

              reader = PdfReader(str(file_path))
              pages = []
              for page in reader.pages:
                  text = page.extract_text()
                  if text:
                      pages.append(text)
              return "\n\n".join(pages)
          except Exception as e:
              msg = f"PDF extraction failed for {file_path.name}: {e}"
              raise IngestionFileError(msg) from e

      def _extract_docx(self, file_path: Path) -> str:
          try:
              import docx

              doc = docx.Document(str(file_path))
              paragraphs = [p.text for p in doc.paragraphs if p.text.strip()]
              return "\n\n".join(paragraphs)
          except Exception as e:
              msg = f"DOCX extraction failed for {file_path.name}: {e}"
              raise IngestionFileError(msg) from e

      def _infer_doc_type(self, file_path: Path) -> str:
          name_lower = file_path.name.lower()
          if any(kw in name_lower for kw in ["rbi", "monetary_policy", "repo_rate"]):
              return "rbi_policy"
          if any(kw in name_lower for kw in ["sebi", "circular", "regulation"]):
              return "sebi_circular"
          if any(kw in name_lower for kw in ["journal", "trade_log", "diary"]):
              return "trade_journal"
          if any(kw in name_lower for kw in ["strategy", "notes", "playbook"]):
              return "strategy_notes"
          if any(kw in name_lower for kw in ["research", "analysis", "report"]):
              return "research"
          return "unknown"

      @staticmethod
      def _generate_chunk_id(file_path: Path, chunk_index: int) -> str:
          raw = f"{file_path.name}:{chunk_index}"
          return hashlib.sha256(raw.encode()).hexdigest()[:16]

      @staticmethod
      def _file_hash(file_path: Path) -> str:
          h = hashlib.sha256()
          with open(file_path, "rb") as f:
              for chunk in iter(lambda: f.read(8192), b""):
                  h.update(chunk)
          return h.hexdigest()[:16]

  """RAG Knowledge Engine — Document Retriever.

  Handles:
  - Query validation (trade prediction detection per SEBI compliance)
  - Query sanitization (SI-10 input validation)
  - Embedding-based retrieval from ChromaDB
  - Source document construction with metadata
  """

  from __future__ import annotations

  import logging
  import re

  from rag_knowledge.config import RAGConfig, get_config
  from rag_knowledge.embeddings import EmbeddingEngine, get_embedding_engine
  from rag_knowledge.exceptions import QueryValidationError, RetrievalError
  from rag_knowledge.models import QueryValidation, SourceDocument
  from rag_knowledge.vectorstore import VectorStore

  logger = logging.getLogger(__name__)


  class DocumentRetriever:
      def __init__(
          self,
          config: RAGConfig | None = None,
          embedding_engine: EmbeddingEngine | None = None,
          vector_store: VectorStore | None = None,
      ) -> None:
          self._config = config or get_config()
          self._embedding_engine = embedding_engine or get_embedding_engine(config=self._config)
          self._vector_store = vector_store or VectorStore(config=self._config)
          self._disallowed_patterns = [re.compile(p, re.IGNORECASE) for p in self._config.disallowed_patterns]

      def validate_query(self, query: str) -> QueryValidation:
          if not query or not query.strip():
              return QueryValidation(
                  is_valid=False,
                  sanitized_query="",
                  is_trade_prediction=False,
                  rejection_reason="Query is empty or whitespace-only",
              )
          if len(query) > self._config.max_query_length:
              return QueryValidation(
                  is_valid=False,
                  sanitized_query="",
                  is_trade_prediction=False,
                  rejection_reason=f"Query exceeds maximum length of {self._config.max_query_length} characters",
              )
          sanitized = self._sanitize(query)
          for pattern in self._disallowed_patterns:
              if pattern.search(sanitized):
                  return QueryValidation(
                      is_valid=False,
                      sanitized_query=sanitized,
                      is_trade_prediction=True,
                      rejection_reason=(
                          "Query requests trade prediction or investment advice. "
                          "This RAG system answers historical-context questions only, "
                          "per SEBI compliance requirements."
                      ),
                  )
          return QueryValidation(
              is_valid=True,
              sanitized_query=sanitized,
              is_trade_prediction=False,
              rejection_reason=None,
          )

      def retrieve(self, query: str, top_k: int | None = None) -> list[SourceDocument]:
          validation = self.validate_query(query)
          if not validation.is_valid:
              if validation.is_trade_prediction:
                  raise QueryValidationError(validation.rejection_reason or "Trade prediction query")
              raise QueryValidationError(validation.rejection_reason or "Invalid query")

          try:
              query_embedding = self._embedding_engine.embed_query(validation.sanitized_query)
          except Exception as e:
              msg = f"Query embedding failed: {e}"
              raise RetrievalError(msg) from e

          try:
              if self._config.retrieval_search_type == "mmr":
                  results = self._vector_store.mmr_search(query_embedding, top_k)
              else:
                  results = self._vector_store.similarity_search(query_embedding, top_k)
          except Exception as e:
              msg = f"Vector store retrieval failed: {e}"
              raise RetrievalError(msg) from e

          return self._build_source_documents(results)

      def _build_source_documents(self, results: dict[str, Any]) -> list[SourceDocument]:
          if not results.get("documents") or not results["documents"][0]:
              return []
          docs = results["documents"][0]
          metas = results.get("metadatas", [[]])[0]
          distances = results.get("distances", [[]])[0]
          source_docs: list[SourceDocument] = []
          for i, doc_text in enumerate(docs):
              meta = metas[i] if i < len(metas) else {}
              distance = distances[i] if i < len(distances) else 1.0
              relevance = max(0.0, min(1.0, 1.0 - distance))
              source_docs.append(
                  SourceDocument(
                      content=doc_text,
                      source_path=meta.get("source_path", "unknown"),
                      page_number=int(meta.get("page_number", -1)) if meta.get("page_number") else None,
                      relevance_score=relevance,
                      chunk_index=int(meta.get("chunk_index", 0)),
                      doc_type=meta.get("doc_type", "unknown"),
                      metadata={k: str(v) for k, v in meta.items() if k not in {"source_path", "doc_type", "chunk_index"}},
                  )
              )
          return source_docs

      @staticmethod
      def _sanitize(query: str) -> str:
          sanitized = query.strip()
          sanitized = re.sub(r"[\x00-\x08\x0b\x0c\x0e-\x1f\x7f]", "", sanitized)
          return sanitized

  """RAG Knowledge Engine — LLM Generator (Ollama + Qwen 2.5 7B).

  Generates answers from retrieved context using local Ollama inference.
  System prompt enforces:
  - Historical context ONLY (no trade predictions)
  - Source citation requirement
  - Fallback when context is insufficient
  - Per ISO A.8.28: no eval/exec on retrieved content
  """

  from __future__ import annotations

  import logging

  from tenacity import retry, stop_after_attempt, wait_exponential, retry_if_exception_type

  from rag_knowledge.config import RAGConfig, get_config
  from rag_knowledge.exceptions import OllamaConnectionError, OllamaGenerationError
  from rag_knowledge.models import SourceDocument

  logger = logging.getLogger(__name__)

  SYSTEM_PROMPT = """You are a RAG Knowledge Engine for Indian financial markets (NIFTY, BANKNIFTY, F&O, options, RBI policy, SEBI regulations).

STRICT RULES — VIOLATION IS UNACCEPTABLE:
1. Answer ONLY with historical-context information from the provided sources.
2. NEVER predict future prices, recommend trades, or give buy/sell/hold advice.
3. NEVER say "you should buy", "you should sell", or "I recommend trading".
4. If the sources do not contain relevant information, respond: "I don't have sufficient historical data to answer this question."
5. Always cite which source document(s) your answer is based on.
6. Distinguish between what is stated in sources vs. general knowledge.
7. Format your answer clearly with sections when appropriate.
8. Use precise dates, numbers, and figures from the sources when available.

QUERY CONTEXT:
{context}

SOURCE DOCUMENTS:
{sources}

Answer the following question based ONLY on the provided source documents.
If the sources are insufficient, state so explicitly."""


  class LLMGenerator:
      def __init__(self, config: RAGConfig | None = None) -> None:
          self._config = config or get_config()
          self._llm: object | None = None

      def _get_llm(self) -> Any:
          if self._llm is not None:
              return self._llm
          try:
              from langchain_ollama import OllamaLLM

              self._llm = OllamaLLM(
                  model=self._config.ollama_model,
                  base_url=self._config.ollama_base_url,
                  temperature=self._config.ollama_temperature,
                  num_ctx=self._config.ollama_num_ctx,
                  timeout=self._config.ollama_timeout_seconds,
              )
              return self._llm
          except Exception as e:
              msg = f"Failed to initialize Ollama LLM: {e}"
              raise OllamaConnectionError(msg) from e

      @retry(
          stop=stop_after_attempt(3),
          wait=wait_exponential(multiplier=1, min=2, max=30),
          retry=retry_if_exception_type(OllamaConnectionError),
          reraise=True,
      )
      def generate(self, query: str, sources: list[SourceDocument]) -> str:
          if not sources:
              return (
                  "I don't have sufficient historical data to answer this question. "
                  "No relevant source documents were found in the knowledge base."
              )
          context_block = query
          sources_block = self._format_sources(sources)
          prompt = SYSTEM_PROMPT.format(
              context=context_block,
              sources=sources_block,
          )
          full_prompt = f"{prompt}\n\nQUESTION: {query}\n\nANSWER:"
          try:
              llm = self._get_llm()
              response = llm.invoke(full_prompt)
              answer = str(response).strip()
              if not answer:
                  return "The language model returned an empty response. Please retry."
              return answer
          except OllamaConnectionError:
              raise
          except Exception as e:
              msg = f"Ollama generation failed: {e}"
              raise OllamaGenerationError(msg) from e

      def health_check(self) -> bool:
          try:
              import httpx

              response = httpx.get(
                  f"{self._config.ollama_base_url}/api/tags",
                  timeout=10.0,
              )
              return response.status_code == 200
          except Exception:
              return False

      @staticmethod
      def _format_sources(sources: list[SourceDocument]) -> str:
          formatted: list[str] = []
          for i, src in enumerate(sources, 1):
              source_line = f"[Source {i}] {src.source_path} (type={src.doc_type}, relevance={src.relevance_score:.3f})"
              content_line = src.content[:500]
              if len(src.content) > 500:
                  content_line += "..."
              formatted.append(f"{source_line}\n{content_line}")
          return "\n\n".join(formatted)

INSTRUCTION 15.1 — Implement multi-factor confidence scoring:

  File: G:\OC\02June26\src\rag_knowledge\confidence.py
  ─────────────────────────────────────────────
  """RAG Knowledge Engine — Confidence Scoring.

  Multi-factor confidence assessment based on:
  1. Source relevance scores (vector similarity)
  2. Source count (more sources = higher confidence)
  3. Source diversity (multiple doc_types = higher confidence)
  4. Answer length heuristic (very short answers = lower confidence)

  Outputs: ConfidenceLevel enum + numerical score.
  Per "Evaluating RAG Systems" (Es et al., 2024) — faithfulness + relevance.
  """

  from __future__ import annotations

  import logging

  from rag_knowledge.config import RAGConfig, get_config
  from rag_knowledge.exceptions import ConfidenceScoringError
  from rag_knowledge.models import ConfidenceLevel, SourceDocument

  logger = logging.getLogger(__name__)


  class ConfidenceScorer:
      def __init__(self, config: RAGConfig | None = None) -> None:
          self._config = config or get_config()

      def score(self, sources: list[SourceDocument], answer: str) -> tuple[ConfidenceLevel, float]:
          if not sources:
              return ConfidenceLevel.INSUFFICIENT, 0.0
          try:
              relevance_factor = self._relevance_factor(sources)
              count_factor = self._count_factor(sources)
              diversity_factor = self._diversity_factor(sources)
              length_factor = self._length_factor(answer)

              raw_score = (
                  relevance_factor * 0.40
                  + count_factor * 0.25
                  + diversity_factor * 0.20
                  + length_factor * 0.15
              )
              final_score = max(0.0, min(1.0, raw_score))
              level = self._score_to_level(final_score)
              return level, round(final_score, 4)
          except Exception as e:
              msg = f"Confidence scoring failed: {e}"
              raise ConfidenceScoringError(msg) from e

      def _relevance_factor(self, sources: list[SourceDocument]) -> float:
          if not sources:
              return 0.0
          scores = [s.relevance_score for s in sources]
          avg = sum(scores) / len(scores)
          max_score = max(scores)
          return (avg * 0.6 + max_score * 0.4)

      def _count_factor(self, sources: list[SourceDocument]) -> float:
          n = len(sources)
          min_sources = self._config.confidence_min_sources
          if n >= min_sources:
              return min(1.0, n / (min_sources * 2))
          return n / min_sources

      def _diversity_factor(self, sources: list[SourceDocument]) -> float:
          doc_types = {s.doc_type for s in sources if s.doc_type != "unknown"}
          source_paths = {s.source_path for s in sources}
          type_diversity = min(1.0, len(doc_types) / 3.0)
          source_diversity = min(1.0, len(source_paths) / 3.0)
          return (type_diversity * 0.5 + source_diversity * 0.5)

      @staticmethod
      def _length_factor(answer: str) -> float:
          word_count = len(answer.split())
          if word_count < 10:
              return 0.2
          if word_count < 30:
              return 0.5
          if word_count < 100:
              return 0.7
          return 1.0

      def _score_to_level(self, score: float) -> ConfidenceLevel:
          if score >= self._config.confidence_high_threshold:
              return ConfidenceLevel.HIGH
          if score >= self._config.confidence_medium_threshold:
              return ConfidenceLevel.MEDIUM
          if score > 0.0:
              return ConfidenceLevel.LOW
          return ConfidenceLevel.INSUFFICIENT

  import json

      def log_query(self, response: RAGResponse) -> None:

              return

              log_path = self._config.audit_log_path

                  "previous_hash": self._last_hash,
              }
              entry_str = json.dumps(entry, default=str, ensure_ascii=True)
              current_hash = hashlib.sha256(entry_str.encode()).hexdigest()
              entry["entry_hash"] = current_hash
              self._last_hash = current_hash
              final_entry = json.dumps(entry, default=str, ensure_ascii=True)
              with open(log_path, "a", encoding="utf-8") as f:
                  f.write(final_entry + "\n")
          except Exception as e:

  """RAG Knowledge Engine — Pipeline Orchestrator.

  DELIVERABLE: query_rag(question) -> RAGResponse(answer, sources, confidence)

  Orchestrates: validation → retrieval → generation → confidence → audit
  Enforces: historical-context ONLY, no trade predictions (SEBI compliance)
  """

  from __future__ import annotations

  import logging
  import time

  from rag_knowledge.audit_logger import AuditLogger
  from rag_knowledge.config import RAGConfig, get_config
  from rag_knowledge.confidence import ConfidenceScorer
  from rag_knowledge.exceptions import QueryValidationError, RAGError
  from rag_knowledge.generator import LLMGenerator
  from rag_knowledge.models import ConfidenceLevel, RAGResponse, SourceDocument
  from rag_knowledge.retriever import DocumentRetriever

  logger = logging.getLogger(__name__)


  class RAGPipeline:
      def __init__(self, config: RAGConfig | None = None) -> None:
          self._config = config or get_config()
          self._retriever = DocumentRetriever(config=self._config)
          self._generator = LLMGenerator(config=self._config)
          self._scorer = ConfidenceScorer(config=self._config)
          self._audit_logger = AuditLogger(config=self._config)

      def query_rag(self, question: str) -> RAGResponse:
          start = time.monotonic()
          validation = self._retriever.validate_query(question)

          if not validation.is_valid:
              latency = (time.monotonic() - start) * 1000
              response = RAGResponse(
                  answer=(
                      "I cannot answer this question. "
                      + (validation.rejection_reason or "Query validation failed.")
                  ),
                  sources=[],
                  confidence=ConfidenceLevel.INSUFFICIENT,
                  confidence_score=0.0,
                  query=question,
                  is_trade_prediction=validation.is_trade_prediction,
                  rejection_reason=validation.rejection_reason,
                  latency_ms=round(latency, 2),
                  model_used=self._config.ollama_model,
              )
              self._audit_logger.log_query(response)
              return response

          try:
              sources = self._retriever.retrieve(question)
          except QueryValidationError:
              raise
          except RAGError as e:
              latency = (time.monotonic() - start) * 1000
              logger.error("Retrieval failed: %s", e)
              response = RAGResponse(
                  answer=f"Retrieval error occurred: {e}",
                  sources=[],
                  confidence=ConfidenceLevel.INSUFFICIENT,
                  confidence_score=0.0,
                  query=question,
                  latency_ms=round(latency, 2),
                  model_used=self._config.ollama_model,
              )
              self._audit_logger.log_query(response)
              return response

          try:
              answer = self._generator.generate(question, sources)
          except RAGError as e:
              latency = (time.monotonic() - start) * 1000
              logger.error("Generation failed: %s", e)
              response = RAGResponse(
                  answer=f"Generation error occurred: {e}",
                  sources=sources,
                  confidence=ConfidenceLevel.INSUFFICIENT,
                  confidence_score=0.0,
                  query=question,
                  latency_ms=round(latency, 2),
                  model_used=self._config.ollama_model,
              )
              self._audit_logger.log_query(response)
              return response

          confidence_level, confidence_score = self._scorer.score(sources, answer)
          latency = (time.monotonic() - start) * 1000

          response = RAGResponse(
              answer=answer,
              sources=sources,
              confidence=confidence_level,
              confidence_score=confidence_score,
              query=question,
              latency_ms=round(latency, 2),
              model_used=self._config.ollama_model,
          )

          self._audit_logger.log_query(response)
          logger.info(
              "RAG query completed: confidence=%s score=%.3f latency=%.0fms sources=%d",
              confidence_level.value,
              confidence_score,
              latency,
              len(sources),
          )
          return response


  _pipeline: RAGPipeline | None = None


  def query_rag(question: str) -> RAGResponse:
      """Production API: query_rag(question) -> RAGResponse(answer, sources, confidence).

      This is the primary deliverable function.
      Answers historical-context questions ONLY. Does NOT predict trades.
      """
      global _pipeline
      if _pipeline is None:
          _pipeline = RAGPipeline()
      return _pipeline.query_rag(question)


  def init_pipeline(config: RAGConfig | None = None) -> RAGPipeline:
      global _pipeline
      _pipeline = RAGPipeline(config=config)
      return _pipeline

  """RAG Knowledge Engine — Locust Load Test.

  Run (when FastAPI endpoint is available):
    locust -f locustfile_rag.py --host http://localhost:8000 --headless --users 10 --spawn-rate 2 --run-time 3m

  Currently tests the function directly via subprocess call.
  For API-based load testing, wrap query_rag() in a FastAPI endpoint first.
  """

  import json
  import random
  from locust import HttpUser, task, between, tag, events

  RAG_QUERIES = [
      "How did NIFTY options behave after previous RBI repo rate hikes?",
      "What happened to BANKNIFTY IV in the week before quarterly expiry?",
      "Historical OI buildup patterns when India VIX was above 20",
      "RBI monetary policy impact on F&O segment",
      "SEBI circular effects on options trading volumes",
      "What was the NIFTY option chain structure during 2020 crash?",
      "How did FII DII data correlate with BANKNIFTY movement in 2023?",
      "Historical gamma risk patterns during India VIX spikes",
      "RBI rate cut vs rate hike: differential impact on NIFTY IV",
      "Weekly vs monthly expiry: historical OI patterns comparison",
  ]

  PREDICTION_QUERIES = [
      "Predict NIFTY price tomorrow",
      "What should I buy today?",
      "Give me a guaranteed return strategy",
      "What will BANKNIFTY do next week?",
  ]


  class RAGQueryUser(HttpUser):
      wait_time = between(1, 5)
      host = "http://localhost:8000"

      def on_start(self):
          self.headers = {"Content-Type": "application/json"}

      @task(8)
      @tag("historical-query")
      def query_historical(self):
          query = random.choice(RAG_QUERIES)
          with self.client.post(
              "/api/v1/rag/query",
              json={"question": query},
              headers=self.headers,
              catch_response=True,
          ) as response:
              if response.status_code == 200:
                  data = response.json()
                  if data.get("confidence") in ["HIGH", "MEDIUM", "LOW", "INSUFFICIENT"]:
                      response.success()
                  else:
                      response.failure(f"Invalid confidence: {data.get('confidence')}")
              else:
                  response.failure(f"HTTP {response.status_code}")

      @task(2)
      @tag("prediction-rejection")
      def query_prediction_rejection(self):
          query = random.choice(PREDICTION_QUERIES)
          with self.client.post(
              "/api/v1/rag/query",
              json={"question": query},
              headers=self.headers,
              catch_response=True,
          ) as response:
              if response.status_code == 200:
                  data = response.json()
                  if data.get("is_trade_prediction") is True:
                      response.success()
                  else:
                      response.failure("Prediction query should be rejected")
              else:
                  response.failure(f"HTTP {response.status_code}")
