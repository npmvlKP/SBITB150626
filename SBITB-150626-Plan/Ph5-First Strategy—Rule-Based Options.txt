4.1 Position Sizing Architecture
import math
import structlog
from config.settings import StrategySettings

logger = structlog.get_logger(__name__)


def compute_lots(
    account_capital: float,
    stop_distance_per_lot: float,
    settings: StrategySettings,
    trade_history: list[dict] | None = None,
    win_rate: float | None = None,
    avg_win_loss_ratio: float | None = None,
) -> int:
    """Compute number of lots to trade using the configured sizing method.

    Dispatches to the appropriate sizing method based on settings.POSITION_SIZING_METHOD.

    Args:
        account_capital: Total account capital in INR
        stop_distance_per_lot: Maximum risk per lot in INR (stop loss distance × lot_size)
        settings: Strategy settings
        trade_history: Historical trades list for optimal-f (each dict: {"pnl": float, ...})
        win_rate: Win rate for Kelly (if None, computed from trade_history)
        avg_win_loss_ratio: Average win / average loss for Kelly (if None, computed from trade_history)

    Returns:
        Number of lots (integer, >= 1, <= MAX_LOTS_PER_TRADE)

    Reference: Kaufman Ch.9-10 (position sizing methods)
    Correction #18: Three selectable methods, fixed-fractional 2% default.
    """
    if stop_distance_per_lot <= 0:
        logger.warning("stop_distance_per_lot must be > 0, returning 1 lot")
        return 1

    method = settings.POSITION_SIZING_METHOD

    if method == "fixed_fractional":
        lots = _fixed_fractional(account_capital, stop_distance_per_lot, settings)
    elif method == "kelly":
        lots = _kelly_fraction(account_capital, stop_distance_per_lot, settings, win_rate, avg_win_loss_ratio)
    elif method == "optimal_f":
        lots = _optimal_f(account_capital, stop_distance_per_lot, settings, trade_history)
    else:
        logger.warning("Unknown position sizing method, falling back to fixed_fractional", method=method)
        lots = _fixed_fractional(account_capital, stop_distance_per_lot, settings)

    lots = max(1, min(lots, settings.MAX_LOTS_PER_TRADE))
    return lots

4.2 Fixed Fractional Method
def _fixed_fractional(
    account_capital: float,
    stop_distance_per_lot: float,
    settings: StrategySettings,
) -> int:
    """Fixed fractional position sizing.

    Risk per trade = account_capital × FIXED_FRACTIONAL_PCT (default 2%).
    lots = risk_per_trade / stop_distance_per_lot

    Reference: Kaufman Ch.9 (fixed fractional risk)
    """
    risk_per_trade = account_capital * settings.FIXED_FRACTIONAL_PCT
    lots = int(risk_per_trade / stop_distance_per_lot)
    return max(1, lots)

4.3 Kelly Fraction Method
def _kelly_fraction(
    account_capital: float,
    stop_distance_per_lot: float,
    settings: StrategySettings,
    win_rate: float | None = None,
    avg_win_loss_ratio: float | None = None,
) -> int:
    """Kelly fraction position sizing with half-Kelly for options.

    Kelly formula: f* = (p × b - q) / b
    where p = win_rate, q = 1 - p, b = avg_win / avg_loss

    Half-Kelly: f* / 2 (options have fat tails, Kaufman Ch.9)
    lots = (account_capital × kelly_fraction) / stop_distance_per_lot

    If win_rate or avg_win_loss_ratio not provided, defaults to conservative:
    win_rate=0.55, avg_win_loss_ratio=1.5 (realistic for selling strategy)

    Reference: Kaufman Ch.9 (Kelly criterion); Correction #18 (half-Kelly for options)
    """
    p = win_rate if win_rate is not None else 0.55
    b = avg_win_loss_ratio if avg_win_loss_ratio is not None else 1.5
    q = 1.0 - p

    if b <= 0 or p <= 0:
        return 1

    kelly = (p * b - q) / b
    if kelly <= 0:
        logger.warning("Kelly fraction <= 0, no trade recommended", kelly=kelly, win_rate=p, ratio=b)
        return 1

    if settings.KELLY_USE_HALF:
        kelly = kelly / 2.0

    risk_per_trade = account_capital * kelly
    lots = int(risk_per_trade / stop_distance_per_lot)
    return max(1, lots)

4.4 Optimal-f Method
def _optimal_f(
    account_capital: float,
    stop_distance_per_lot: float,
    settings: StrategySettings,
    trade_history: list[dict] | None = None,
) -> int:
    """Optimal-f position sizing (Ralph Vince method).

    Optimal-f is the fraction that maximizes geometric growth.
    Computed by iterating f from 0.01 to 1.0 and finding the value
    that maximizes the geometric mean of (1 + f × (-trade_loss / largest_loss)).

    If trade_history is None or insufficient, falls back to fixed_fractional.

    Reference: Kaufman Ch.9 (optimal-f); Ralph Vince "Portfolio Management Formulas"
    """
    if trade_history is None or len(trade_history) < settings.OPTIMAL_F_LOOKBACK_TRADES:
        logger.warning("Insufficient trade history for optimal-f, falling back to fixed_fractional",
                       history_len=len(trade_history) if trade_history else 0,
                       required=settings.OPTIMAL_F_LOOKBACK_TRADES)
        return _fixed_fractional(account_capital, stop_distance_per_lot, settings)

    losses = [t["pnl"] for t in trade_history if t["pnl"] < 0]
    if not losses:
        largest_loss = -1.0
    else:
        largest_loss = min(losses)

    if largest_loss >= 0:
        return _fixed_fractional(account_capital, stop_distance_per_lot, settings)

    best_f = 0.01
    best_geo_mean = -float("inf")

    for f_pct in [i / 100.0 for i in range(1, 100)]:
        geo_product = 1.0
        for trade in trade_history:
            hpr = 1.0 + f_pct * (-trade["pnl"] / largest_loss)
            if hpr <= 0:
                geo_product = 0
                break
            geo_product *= hpr
        n = len(trade_history)
        geo_mean = geo_product ** (1.0 / n) if geo_product > 0 and n > 0 else 0
        if geo_mean > best_geo_mean:
            best_geo_mean = geo_mean
            best_f = f_pct

    risk_per_trade = account_capital * best_f
    lots = int(risk_per_trade / stop_distance_per_lot)
    return max(1, lots)

def select_selling_strikes(
    spot: float,
    atr_14: float,
    option_chain: OptionChainSnapshot,
    settings: StrategySettings,
    as_of_date: object,  # date type — avoids circular import
) -> tuple[float, float] | None:

    Selling strike selection (2SD from spot using ATR, Correction #14):
    - CE strike ≈ spot + (SELLING_STRIKE_SD_MULTIPLIER × ATR(14))
    - PE strike ≈ spot - (SELLING_STRIKE_SD_MULTIPLIER × ATR(14))
    - Round to nearest 50 (NIFTY strike interval)
    - Validate OI > SELLING_STRIKE_MIN_OI AND volume > SELLING_STRIKE_MIN_VOLUME (Correction #21)
    - If primary strike fails validation, search ±2 strikes away

    Returns:
        (ce_strike, pe_strike) or None if no valid strikes found

    Reference: Kaufman Ch.7 (2SD ≈ 2×ATR for normal distribution); Correction #14, #21
    """

    ce_target = spot + sd_distance
    pe_target = spot - sd_distance

    ce_strike = _round_to_strike_interval(ce_target, "NIFTY")
    pe_strike = _round_to_strike_interval(pe_target, "NIFTY")

    pe_valid = _validate_strike(pe_strike, "PE", option_chain, settings)

        ce_strike = _find_nearest_valid_strike(ce_target, "CE", option_chain, settings, max_search=2)
    if not pe_valid:

        logger.warning("No valid selling strikes found",
                       ce_target=ce_target, pe_target=pe_target,
                       as_of_date=str(as_of_date))
        return None

    return (ce_strike, pe_strike)

                       direction=direction, option_type=option_type,

5.4 Helper Functions
def _round_to_strike_interval(price: float, symbol: str) -> float:
    """Round price to nearest valid strike interval.

    NIFTY: multiples of 50.
    BANKNIFTY: multiples of 100 (future extension).
    """
    interval = 50 if symbol == "NIFTY" else 100
    return round(price / interval) * interval


def _validate_strike(
    strike: float,
    option_type: str,
    option_chain: OptionChainSnapshot,
    settings: StrategySettings,
) -> bool:
    """Validate strike has sufficient OI and volume (Correction #21)."""
    for s in option_chain.strikes:
        if s.strike == strike and s.option_type == option_type:
            min_oi = settings.SELLING_STRIKE_MIN_OI
            min_vol = settings.SELLING_STRIKE_MIN_VOLUME
            return s.oi >= min_oi and s.volume >= min_vol
    return False


def _find_nearest_valid_strike(
    target_price: float,
    option_type: str,
    option_chain: OptionChainSnapshot,
    settings: StrategySettings,
    max_search: int = 2,
) -> float | None:
    """Find nearest valid strike within ±max_search intervals from target."""
    interval = 50
    min_oi = settings.SELLING_STRIKE_MIN_OI
    min_vol = settings.SELLING_STRIKE_MIN_VOLUME

    for offset in range(1, max_search + 1):
        for candidate in [target_price + offset * interval, target_price - offset * interval]:
            candidate_rounded = _round_to_strike_interval(candidate, "NIFTY")
            for s in option_chain.strikes:
                if s.strike == candidate_rounded and s.option_type == option_type:
                    if s.oi >= min_oi and s.volume >= min_vol:
                        return candidate_rounded
    return None


def _find_delta_targeted_strike(
    option_type: str,
    delta_min: float,
    delta_max: float,
    delta_target: float,
    option_chain: OptionChainSnapshot,
    settings: StrategySettings,
) -> float | None:
    """Find strike with delta closest to target within [delta_min, delta_max] range.

    Validates OI and volume on the selected strike (Correction #21).
    """
    candidates = []
    for s in option_chain.strikes:
        if s.option_type != option_type or s.delta is None:
            continue
        abs_delta = abs(s.delta)
        if delta_min <= abs_delta <= delta_max:
            if s.oi >= settings.BUYING_STRIKE_MIN_OI and s.volume >= settings.BUYING_STRIKE_MIN_VOLUME:
                delta_distance = abs(abs_delta - delta_target)
                candidates.append((delta_distance, s.strike))

    if not candidates:
        return None

    candidates.sort(key=lambda x: x[0])
    return candidates[0][1]

def check_selling_entry(
    iv_rank: float,
    adx: float,
    market_strength_score: float,
    india_vix: float,
    settings: StrategySettings,
) -> bool:
    """Check all entry conditions for options selling strategy.

    Entry conditions (Plan.txt line 390-406, Chan Ch.5):
    1. IV rank > SELLING_IV_RANK_MIN (default 40) — high IV = overpriced = sell premium
    2. ADX < SELLING_ADX_MAX (default 25) — range-bound market
    3. Market strength in [SELLING_MKT_STRENGTH_MIN, SELLING_MKT_STRENGTH_MAX] — NEUTRAL regime
    4. India VIX > SELLING_VIX_MIN (default 15) — sufficient premium to collect

    ALL conditions must be True simultaneously.

    Returns:
        True if all entry conditions met

    Reference: Chan Ch.5 (mean-reversion when IV is high); Kaufman Ch.10 (premium collection)
    """
    conditions = [
        ("iv_rank", iv_rank > settings.SELLING_IV_RANK_MIN, iv_rank, settings.SELLING_IV_RANK_MIN),
        ("adx", adx < settings.SELLING_ADX_MAX, adx, settings.SELLING_ADX_MAX),
        ("mkt_strength_range", settings.SELLING_MKT_STRENGTH_MIN <= market_strength_score <= settings.SELLING_MKT_STRENGTH_MAX, market_strength_score, f"[{settings.SELLING_MKT_STRENGTH_MIN}, {settings.SELLING_MKT_STRENGTH_MAX}]"),
        ("vix", india_vix > settings.SELLING_VIX_MIN, india_vix, settings.SELLING_VIX_MIN),
    ]

    all_met = all(c[1] for c in conditions)
    if not all_met:
        for name, met, actual, threshold in conditions:
            if not met:
                logger.debug("Selling entry condition not met", condition=name, actual=actual, threshold=threshold)

6.3 Exit Condition Checks
def check_selling_exit(
    current_premium_ce: float,
    current_premium_pe: float,
    entry_premium_ce: float,
    entry_premium_pe: float,
    market_regime: MarketRegime,
    days_to_expiry: int,
    settings: StrategySettings,
) -> Signal | None:
    """Check all exit conditions for options selling position.

    Exit conditions (Plan.txt line 404-407, Kaufman Ch.10):
    1. Premium decay ≥ SELLING_PREMIUM_DECAY_EXIT_PCT (default 50%) — time value profit captured
    2. Stop loss: combined premium ≥ SELLING_SL_MULTIPLIER × combined entry premium — risk limit
    3. Time stop: days_to_expiry ≤ SELLING_DAYS_BEFORE_EXPIRY_EXIT (default 2) — gamma risk
    4. Regime change: market_regime NOT NEUTRAL (Correction #10 — generates EXIT signal)

    Returns:
        Signal with action=EXIT if any exit condition met, None otherwise

    Reference: Kaufman Ch.10 (strangle exits); Correction #10, #15, #16
    """
    combined_entry = entry_premium_ce + entry_premium_pe
    combined_current = current_premium_ce + current_premium_pe

    if combined_entry <= 0:
        logger.warning("Combined entry premium <= 0, cannot evaluate exits")
        return None

    # Exit 1: Premium decay
    decay_pct = (combined_entry - combined_current) / combined_entry
    if settings.SELLING_PREMIUM_DECAY_SCOPE == "either_leg":
        decay_ce = (entry_premium_ce - current_premium_ce) / entry_premium_ce if entry_premium_ce > 0 else 0
        decay_pe = (entry_premium_pe - current_premium_pe) / entry_premium_pe if entry_premium_pe > 0 else 0
        decay_met = decay_ce >= settings.SELLING_PREMIUM_DECAY_EXIT_PCT or decay_pe >= settings.SELLING_PREMIUM_DECAY_EXIT_PCT
    else:
        decay_met = decay_pct >= settings.SELLING_PREMIUM_DECAY_EXIT_PCT

    if decay_met:
        logger.info("Selling exit: premium decay target reached", decay_pct=round(decay_pct, 4))
        return Signal(
            action=SignalAction.EXIT,
            strategy="options_selling",
            direction=SignalDirection.NEUTRAL,
            ce_strike=None, pe_strike=None, ce_pe="BOTH",
            lots=0, entry_premium=combined_entry, stop_loss=0.0, target=0.0,
            as_of_date=date.today(),
            metadata={"exit_reason": "premium_decay", "decay_pct": round(decay_pct, 4)},
        )

    # Exit 2: Stop loss (combined premium ≥ 2× entry, Correction #16)
    if combined_current >= combined_entry * settings.SELLING_SL_MULTIPLIER:
        logger.warning("Selling exit: stop loss hit", combined_current=combined_current, combined_entry=combined_entry)
        return Signal(
            action=SignalAction.EXIT,
            strategy="options_selling",
            direction=SignalDirection.NEUTRAL,
            ce_strike=None, pe_strike=None, ce_pe="BOTH",
            lots=0, entry_premium=combined_entry, stop_loss=0.0, target=0.0,
            as_of_date=date.today(),
            metadata={"exit_reason": "stop_loss", "combined_current": combined_current, "sl_threshold": combined_entry * settings.SELLING_SL_MULTIPLIER},
        )

    # Exit 3: Time stop (gamma risk)
    if days_to_expiry <= settings.SELLING_DAYS_BEFORE_EXPIRY_EXIT:
        logger.info("Selling exit: time stop (near expiry)", days_to_expiry=days_to_expiry)
        return Signal(
            action=SignalAction.EXIT,
            strategy="options_selling",
            direction=SignalDirection.NEUTRAL,
            ce_strike=None, pe_strike=None, ce_pe="BOTH",
            lots=0, entry_premium=combined_entry, stop_loss=0.0, target=0.0,
            as_of_date=date.today(),
            metadata={"exit_reason": "time_stop", "days_to_expiry": days_to_expiry},
        )

    # Exit 4: Regime change (Correction #10)
    if settings.SELLING_REGIME_CHANGE_EXIT and market_regime != MarketRegime.NEUTRAL:
        logger.info("Selling exit: regime change", regime=market_regime.value)
        return Signal(
            action=SignalAction.EXIT,
            strategy="options_selling",
            direction=SignalDirection.NEUTRAL,
            ce_strike=None, pe_strike=None, ce_pe="BOTH",
            lots=0, entry_premium=combined_entry, stop_loss=0.0, target=0.0,
            as_of_date=date.today(),
            metadata={"exit_reason": "regime_change", "regime": market_regime.value},
        )

    return None

def check_buying_entry(
    iv_rank: float,
    adx: float,
    market_strength_score: float,
    supertrend_direction: str,
    settings: StrategySettings,
    sentiment_score: float | None = None,
) -> SignalDirection | None:
    """Check all entry conditions for options buying strategy.

    Entry conditions (Plan.txt line 410-418, Chan Ch.6):
    1. IV rank < BUYING_IV_RANK_MAX (default 30) — low IV = cheap options
    2. ADX > BUYING_ADX_MIN (default 25) — trending market
    3. Market strength > BULLISH (60) OR < BEARISH (30) — directional regime
    4. Supertrend confirms direction (UP for bullish, DOWN for bearish)
    5. Sentiment confirms direction (optional, weight 0.0 by default, Correction #17)

    Returns:
        SignalDirection.BULLISH or SignalDirection.BEARISH if entry conditions met, None otherwise

    Reference: Chan Ch.6 (momentum when IV is low); Correction #17
    """
    if iv_rank >= settings.BUYING_IV_RANK_MAX:
        return None

    if adx <= settings.BUYING_ADX_MIN:
        return None

    direction = None
    if market_strength_score > settings.BUYING_MKT_STRENGTH_BULLISH and supertrend_direction == "UP":
        direction = SignalDirection.BULLISH
    elif market_strength_score < settings.BUYING_MKT_STRENGTH_BEARISH and supertrend_direction == "DOWN":
        direction = SignalDirection.BEARISH

    if direction is None:
        return None

    if settings.BUYING_SENTIMENT_WEIGHT > 0 and sentiment_score is not None:
        if direction == SignalDirection.BULLISH and sentiment_score < 0:
            logger.debug("Buying entry blocked: sentiment contradicts bullish direction", sentiment=sentiment_score)
            return None
        if direction == SignalDirection.BEARISH and sentiment_score > 0:
            logger.debug("Buying entry blocked: sentiment contradicts bearish direction", sentiment=sentiment_score)
            return None

    return direction

7.3 Exit Condition Checks
def check_buying_exit(
    current_premium: float,
    entry_premium: float,
    trailing_stop: float,
    target: float,
    days_to_expiry: int,
    settings: StrategySettings,
) -> Signal | None:
    """Check all exit conditions for options buying position.

    Exit conditions (Plan.txt line 421-424, Kaufman Ch.10):
    1. Trailing stop hit: current_premium ≤ trailing_stop
    2. Target reached: current_premium ≥ target (2:1 R:R minimum)
    3. Time stop: days_to_expiry ≤ BUYING_DAYS_BEFORE_EXPIRY_EXIT (default 3)

    Returns:
        Signal with action=EXIT if any exit condition met, None otherwise

    Reference: Kaufman Ch.10 (trailing stops); Chan Ch.6 (R:R); Correction #23
    """
    if entry_premium <= 0:
        return None

    if current_premium <= trailing_stop:
        logger.info("Buying exit: trailing stop hit", current=current_premium, trail=trailing_stop)
        return Signal(
            action=SignalAction.EXIT, strategy="options_buying",
            direction=SignalDirection.BULLISH,
            ce_strike=None, pe_strike=None, ce_pe="",
            lots=0, entry_premium=entry_premium, stop_loss=trailing_stop, target=target,
            as_of_date=date.today(),
            metadata={"exit_reason": "trailing_stop", "current_premium": current_premium, "trailing_stop": trailing_stop},
        )

    if current_premium >= target:
        logger.info("Buying exit: target reached", current=current_premium, target=target)
        return Signal(
            action=SignalAction.EXIT, strategy="options_buying",
            direction=SignalDirection.BULLISH,
            ce_strike=None, pe_strike=None, ce_pe="",
            lots=0, entry_premium=entry_premium, stop_loss=trailing_stop, target=target,
            as_of_date=date.today(),
            metadata={"exit_reason": "target_reached", "current_premium": current_premium, "target": target},
        )

    if days_to_expiry <= settings.BUYING_DAYS_BEFORE_EXPIRY_EXIT:
        logger.info("Buying exit: time stop", days_to_expiry=days_to_expiry)
        return Signal(
            action=SignalAction.EXIT, strategy="options_buying",
            direction=SignalDirection.BULLISH,
            ce_strike=None, pe_strike=None, ce_pe="",
            lots=0, entry_premium=entry_premium, stop_loss=trailing_stop, target=target,
            as_of_date=date.today(),
            metadata={"exit_reason": "time_stop", "days_to_expiry": days_to_expiry},
        )

    return None

7.4 Adaptive Trailing Stop (Correction #19)
def compute_adaptive_trailing_stop(
    base_atr: float,
    atr_mult: float,
    india_vix: float,
    market_strength_score: float,
    volume_ratio: float,
    settings: StrategySettings,
) -> float:
    """Compute adaptive trailing stop distance.

    Base = atr_mult × ATR(14) (default 1.5 × ATR)
    Adjusted by:
    - VIX factor: higher VIX → wider stop (trail × (1 + VIX/100 × VIX_SCALE))
    - Market strength factor: stronger regime → wider stop (allows trend to breathe)
    - Volume spike factor: volume > threshold → tighter stop (reversal imminent)

    Formula:
        trail = base_atr × atr_mult
        trail = trail × (1 + (india_vix / 100) × TRAIL_VIX_SCALE_FACTOR)
        trail = trail × (1 + (market_strength_score / 100) × TRAIL_STRENGTH_SCALE_FACTOR)
        if volume_ratio > TRAIL_VOLUME_SPIKE_THRESHOLD:
            trail = trail × (1 - TRAIL_VOLUME_SPIKE_TIGHTEN)

    Returns:
        Adjusted trailing stop distance in rupees (positive value)

    Reference: Kaufman Ch.10 (adaptive stops); Correction #19
    """
    trail = base_atr * atr_mult

    vix_adjustment = 1.0 + (india_vix / 100.0) * settings.TRAIL_VIX_SCALE_FACTOR
    trail = trail * vix_adjustment

    strength_adjustment = 1.0 + (market_strength_score / 100.0) * settings.TRAIL_STRENGTH_SCALE_FACTOR
    trail = trail * strength_adjustment

    if volume_ratio > settings.TRAIL_VOLUME_SPIKE_THRESHOLD:
        trail = trail * (1.0 - settings.TRAIL_VOLUME_SPIKE_TIGHTEN)

    return max(trail, base_atr * 0.5)

class OptionsBuyingStrategy:
    """Options buying (momentum) strategy.

    Entry: IV rank < 30, ADX > 25, market strength directional, Supertrend confirms
    Action: Buy ATM/1-ITM option (delta 0.50-0.60) in trend direction
    Exit: Adaptive trailing SL / 2:1 R:R target / 3-days-before-expiry

    Reference: Chan Ch.6 (momentum); Kaufman Ch.10 (trailing stops); Plan.txt line 408-424
    """

    def __init__(self, settings: StrategySettings | None = None):
        self.settings = settings or StrategySettings()

    def generate_signal(
        self,
        market_data: dict,
        technical_report: object,
        market_strength: MarketStrength,
        iv_rank: float,
        option_chain: OptionChainSnapshot,
        account_capital: float,
        as_of_date: date,
        sentiment_score: float | None = None,
        current_position: dict | None = None,
    ) -> Signal | None:
        """Generate a trading signal for options buying.

        Returns:
            Signal with BUY_LONG action or EXIT action or None (HOLD)
        """
        adx = getattr(getattr(technical_report, "indicators", None), "adx", None)
        if adx is None:
            adx = market_data.get("adx", 0.0)
        atr_14 = market_data.get("atr_14", 0.0)
        spot = market_data.get("spot", 0.0)
        india_vix = market_data.get("india_vix", 0.0)
        volume_ratio = market_data.get("volume_ratio", 1.0)
        supertrend_direction = market_data.get("supertrend_direction", "NEUTRAL")

        if current_position is not None:
            return self._check_exits(
                current_position=current_position, market_data=market_data,
                market_strength=market_strength, option_chain=option_chain,
                as_of_date=as_of_date, atr_14=atr_14,
                india_vix=india_vix, volume_ratio=volume_ratio,
            )

        direction = check_buying_entry(
            iv_rank=iv_rank, adx=adx,
            market_strength_score=market_strength.score,
            supertrend_direction=supertrend_direction,
            settings=self.settings, sentiment_score=sentiment_score,
        )
        if direction is None:
            return None

        strike = select_buying_strike(
            spot=spot, direction=direction.value,
            option_chain=option_chain, settings=self.settings,
            as_of_date=as_of_date,
        )
        if strike is None:
            return None

        option_type = "CE" if direction == SignalDirection.BULLISH else "PE"
        premium = self._get_premium(option_chain, strike, option_type)
        if premium is None or premium <= 0:
            logger.warning("Cannot get premium for selected strike",
                           strike=strike, option_type=option_type)
            return None

        trailing_stop_distance = compute_adaptive_trailing_stop(
            base_atr=atr_14, atr_mult=self.settings.BUYING_TRAILING_SL_ATR_MULT,
            india_vix=india_vix, market_strength_score=market_strength.score,
            volume_ratio=volume_ratio, settings=self.settings,
        )
        trailing_stop = max(premium - trailing_stop_distance, 0.0)

        risk_amount = premium - trailing_stop
        if risk_amount <= 0:
            return None
        target = premium + risk_amount * self.settings.BUYING_TARGET_RR_MIN

        actual_rr = (target - premium) / risk_amount if risk_amount > 0 else 0
        if actual_rr < self.settings.BUYING_TARGET_RR_MIN:
            logger.debug("Buying entry rejected: R:R below minimum", actual_rr=actual_rr)
            return None

        stop_distance_per_lot = risk_amount * self.settings.NIFTY_LOT_SIZE
        lots = compute_lots(
            account_capital=account_capital,
            stop_distance_per_lot=stop_distance_per_lot,
            settings=self.settings,
        )

        signal = Signal(
            action=SignalAction.BUY_LONG, strategy="options_buying",
            direction=direction,
            ce_strike=strike if option_type == "CE" else None,
            pe_strike=strike if option_type == "PE" else None,
            ce_pe=option_type, lots=lots,
            entry_premium=premium, stop_loss=trailing_stop, target=target,
            as_of_date=as_of_date,
            metadata={
                "iv_rank": iv_rank, "adx": adx,
                "market_strength_score": market_strength.score,
                "market_regime": market_strength.regime.value,
                "atr_14": atr_14,
                "supertrend_direction": supertrend_direction,
                "sentiment_score": sentiment_score,
                "delta": self._get_delta(option_chain, strike, option_type),
                "oi": self._get_oi(option_chain, strike, option_type),
                "volume": self._get_volume(option_chain, strike, option_type),
                "risk_amount": risk_amount * lots * self.settings.NIFTY_LOT_SIZE,
                "position_sizing_method": self.settings.POSITION_SIZING_METHOD,
                "estimated_slippage_pct": 0.01,
                "trailing_stop_base": atr_14 * self.settings.BUYING_TRAILING_SL_ATR_MULT,
                "trailing_stop_adjusted": trailing_stop_distance,
            },
        )

        logger.info("Buying signal generated",
                     direction=direction.value, strike=strike,
                     lots=lots, premium=premium,
                     trailing_stop=trailing_stop, target=target)
        return signal
