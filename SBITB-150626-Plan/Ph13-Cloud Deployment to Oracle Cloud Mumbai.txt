PHASE 13: CLOUD DEPLOYMENT TO ORACLE CLOUD MUMBAI (WEEKS 32-34)
PRODUCTION-GRADE IMPLEMENTATION PLAN
================================================================================

Books: Kleppmann Ch.6-9 (Scalability, Fault Tolerance)
Goal: Production deployment with <5ms latency to NSE

Kleppmann Ch.6-9 Extractions Applied to This Plan:
  Ch.6 (Partitioning): Co-locate trading engine + broker API client in same
    Mumbai AD as NSE gateways; partition TimescaleDB by time (hypertable) so
    hot data (last 30 days) stays in memory; partition Redis by key namespace
    (order_cache, position_cache, rate_limit) to avoid key collisions.
  Ch.7 (Transactions): Use exactly-once order submission via idempotency keys
    stored in Redis; implement compensating transactions for failed order
    chains (cancel-if-place-fails); use serializable isolation for daily
    reconciliation reads; implement saga pattern for multi-step order flows.
  Ch.8 (Distributed Systems - Fault Tolerance): Design for process crashes
    using supervised process restart (systemd/watchdog); implement circuit
    breaker for broker API calls (Zerodha API rate limits, timeout patterns);
    use timeout + retry with exponential backoff for all network calls;
    implement fencing tokens to prevent stale leader orders after restart.
  Ch.9 (Consistency + Consensus): Use single-leader architecture (only one
    bot instance places orders at a time); implement leader election via
    Redis SETNX with TTL for multi-instance failover; use NTP-synchronized
    timestamps (mandatory for audit per MiFID II Art.25); enforce linearizable
    kill switch state via Redis atomic operations; implement read-repair for
    position state on restart by reconciling with broker.

# Multi-stage Dockerfile for ARM64 (Oracle Ampere A1) + amd64 (dev)
# Kleppmann Ch.8: Build reproducible images with pinned base versions

FROM --platform=$TARGETPLATFORM python:3.12-slim AS builder

WORKDIR /app

# Install system dependencies for QuantLib, TimescaleDB client
RUN apt-get update && apt-get install -y --no-install-recommends \
    build-essential \
    libpq-dev \
    && rm -rf /var/lib/apt/lists/*

COPY pyproject.toml ./
COPY src/ src/
COPY config/ config/
COPY scripts/ scripts/

RUN pip install --no-cache-dir --upgrade pip setuptools wheel && \
    pip install --no-cache-dir -e ".[dev]" && \
    pip check

# Production stage
FROM --platform=$TARGETPLATFORM python:3.12-slim AS production

WORKDIR /app

RUN apt-get update && apt-get install -y --no-install-recommends \
    libpq5 \
    curl \
    && rm -rf /var/lib/apt/lists/* \
    && groupadd -r trading && useradd -r -g trading -d /app -s /sbin/nologin trading

COPY --from=builder /usr/local/lib/python3.12/site-packages /usr/local/lib/python3.12/site-packages
COPY --from=builder /usr/local/bin /usr/local/bin
COPY --from=builder /app /app

# NTP configuration for IST (Kleppmann Ch.9: clock synchronization)
RUN ln -sf /usr/share/zoneinfo/Asia/Kolkata /etc/localtime && \
    echo "Asia/Kolkata" > /etc/timezone

# Health check endpoint
HEALTHCHECK --interval=30s --timeout=10s --start-period=60s --retries=3 \
    CMD curl -f http://localhost:8000/health || exit 1

USER trading

EXPOSE 8000

CMD ["python", "-m", "src.main"]

-- TimescaleDB initialization for Indian Trading Bot
-- Kleppmann Ch.6: Time-partitioned hypertable for efficient range queries

CREATE EXTENSION IF NOT EXISTS timescaledb;

-- Audit trail table (append-only, per MiFID II Art.25, NIST AU-9)
CREATE TABLE IF NOT EXISTS audit_events (
    event_id UUID PRIMARY KEY,
    timestamp TIMESTAMPTZ NOT NULL,
    event_type VARCHAR(50) NOT NULL,
    source VARCHAR(100) NOT NULL,
    details JSONB NOT NULL DEFAULT '{}',
    checksum VARCHAR(64) NOT NULL,
    ntp_offset_ms FLOAT,
    created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- Convert to hypertable (partition by time, 1-day chunks)
SELECT create_hypertable('audit_events', 'timestamp', chunk_time_interval => INTERVAL '1 day', if_not_exists => TRUE);

-- Index for common query patterns
CREATE INDEX IF NOT EXISTS idx_audit_events_type ON audit_events (event_type, timestamp DESC);
CREATE INDEX IF NOT EXISTS idx_audit_events_source ON audit_events (source, timestamp DESC);
CREATE INDEX IF NOT EXISTS idx_audit_events_created ON audit_events (created_at DESC);

-- Orders table (for reconciliation per ISO A.8.15)
CREATE TABLE IF NOT EXISTS orders (
    order_id VARCHAR(50) PRIMARY KEY,
    timestamp TIMESTAMPTZ NOT NULL,
    symbol VARCHAR(20) NOT NULL,
    segment VARCHAR(10) NOT NULL,
    side VARCHAR(10) NOT NULL,
    order_type VARCHAR(20) NOT NULL,
    quantity INTEGER NOT NULL,
    price DECIMAL(20,4),
    order_value DECIMAL(20,4),
    status VARCHAR(20) NOT NULL DEFAULT 'PENDING',
    strategy_id VARCHAR(50),
    algo_tag VARCHAR(20),
    broker_order_id VARCHAR(50),
    fill_price DECIMAL(20,4),
    fill_quantity INTEGER,
    rejection_reason TEXT,
    created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

SELECT create_hypertable('orders', 'timestamp', chunk_time_interval => INTERVAL '1 day', if_not_exists => TRUE);

CREATE INDEX IF NOT EXISTS idx_orders_symbol ON orders (symbol, timestamp DESC);
CREATE INDEX IF NOT EXISTS idx_orders_status ON orders (status, timestamp DESC);
CREATE INDEX IF NOT EXISTS idx_orders_strategy ON orders (strategy_id, timestamp DESC);

-- Kill switch events table (per MiFID II Art.17, NIST RS.RP-1)
CREATE TABLE IF NOT EXISTS kill_switch_events (
    id SERIAL PRIMARY KEY,
    timestamp TIMESTAMPTZ NOT NULL,
    level VARCHAR(20) NOT NULL,
    trigger_source VARCHAR(50) NOT NULL,
    reason TEXT NOT NULL,
    previous_level VARCHAR(20) NOT NULL,
    created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

SELECT create_hypertable('kill_switch_events', 'timestamp', chunk_time_interval => INTERVAL '1 day', if_not_exists => TRUE);

-- Daily reconciliation table (per ISO A.8.15)
CREATE TABLE IF NOT EXISTS daily_reconciliation (
    reconcile_date DATE PRIMARY KEY,
    total_orders INTEGER NOT NULL DEFAULT 0,
    total_fills INTEGER NOT NULL DEFAULT 0,
    total_rejections INTEGER NOT NULL DEFAULT 0,
    realized_pnl DECIMAL(20,4),
    unrealized_pnl DECIMAL(20,4),
    margin_utilization DECIMAL(5,4),
    position_count INTEGER DEFAULT 0,
    broker_order_count INTEGER,
    mismatch_flag BOOLEAN DEFAULT FALSE,
    notes TEXT,
    created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- Retention policy (7 years per SEBI requirement)
SELECT add_retention_policy('audit_events', INTERVAL '7 years', if_not_exists => TRUE);
SELECT add_retention_policy('orders', INTERVAL '7 years', if_not_exists => TRUE);
SELECT add_retention_policy('kill_switch_events', INTERVAL '7 years', if_not_exists => TRUE);

-- Continuous aggregates for monitoring (Kleppmann Ch.6: materialized views)
CREATE MATERIALIZED VIEW IF NOT EXISTS orders_per_minute
WITH (timescaledb.continuous) AS
SELECT
    time_bucket('1 minute', timestamp) AS bucket,
    count(*) AS order_count,
    count(*) FILTER (WHERE status = 'FILLED') AS fill_count,
    count(*) FILTER (WHERE status = 'REJECTED') AS rejection_count,
    sum(order_value) AS total_value
FROM orders
GROUP BY bucket
WITH NO DATA;

SELECT add_continuous_aggregate_policy('orders_per_minute',
    start_offset => INTERVAL '3 hours',
    end_offset => INTERVAL '1 minute',
    schedule_interval => INTERVAL '1 minute',
    if_not_exists => TRUE
);

# Prometheus alerting rules for Indian Trading Bot
# Per NIST DE.AE, ISO A.8.16, MiFID II Art.17

groups:
  - name: trading_bot_critical
    rules:
      - alert: KillSwitchActivated
        expr: trading_bot_kill_switch_level > 0
        for: 0s
        labels:
          severity: critical
          component: risk
          reference: MiFID II Art.17
        annotations:
          summary: "Kill switch activated (level={{ $value }})"
          description: "Kill switch is NOT INACTIVE. All trading may be halted."
          runbook_url: "https://wiki.internal/runbooks/kill-switch"

      - alert: MarginUtilizationCritical
        expr: trading_bot_margin_utilization >= 0.95
        for: 0s
        labels:
          severity: critical
          component: risk
          reference: CIR/MRD/DP/09/2012
        annotations:
          summary: "Margin utilization >= 95% (KILL threshold)"
          description: "Current utilization: {{ $value | humanizePercentage }}. Kill switch should auto-activate."

      - alert: DailyLossLimitExceeded
        expr: trading_bot_daily_pnl <= -50000
        for: 0s
        labels:
          severity: critical
          component: risk
          reference: CIR/MRD/DP/09/2012
        annotations:
          summary: "Daily loss limit exceeded (Rs {{ $value }})"

      - alert: OrderRejectionRateHigh
        expr: rate(trading_bot_orders_rejected_total[1m]) >= 10
        for: 0s
        labels:
          severity: critical
          component: risk
          reference: CIR/MRD/DP/09/2012
        annotations:
          summary: "Order rejection rate >= 10/min (KILL threshold)"

      - alert: BotProcessDown
        expr: up{job="trading_bot"} == 0
        for: 30s
        labels:
          severity: critical
          component: infrastructure
        annotations:
          summary: "Trading bot process is down"
          description: "No metrics received from trading bot for 30s."

  - name: trading_bot_warning
    rules:
      - alert: MarginUtilizationWarning

        for: 1m
        labels:
          severity: warning
          component: risk

          severity: warning

          severity: warning

@asynccontextmanager
async def measure_api_latency(endpoint: str, method: str = "GET"):
    """Context manager to measure broker API latency.

    Kleppmann Ch.8: Measure tail latencies, not just averages.
    """
    start = time.monotonic()
    try:
        yield
    finally:
        elapsed = time.monotonic() - start
        API_LATENCY.labels(endpoint=endpoint, method=method).observe(elapsed)


def set_kill_switch_level(level_name: str) -> None:
    """Update kill switch gauge from level name."""
    level_map = {"INACTIVE": 0, "THROTTLE": 1, "PAUSE": 2, "KILL": 3}
    KILL_SWITCH_LEVEL.set(level_map.get(level_name, 0))


async def metrics_handler(_request: object) -> object:
    """ASGI handler for /metrics endpoint."""
    from prometheus_client import CONTENT_TYPE_LATEST

    class MetricsResponse:
        def __init__(self, body: bytes):
            self.body = body
            self.status_code = 200
            self.headers = {"Content-Type": CONTENT_TYPE_LATEST}

    body = generate_latest(registry=REGISTRY)
    return MetricsResponse(body)


async def health_handler(_request: object) -> object:
    """ASGI handler for /health endpoint."""
    from src.risk.audit import NTPClock

    ntp = NTPClock()
    try:
        offset = await ntp.check_offset()
        ntp_ok = abs(offset) < 500
    except Exception:
        offset = 0.0
        ntp_ok = False

    class HealthResponse:
        def __init__(self, body: bytes, status: int):
            self.body = body
            self.status_code = status

    import json

    health = {
        "status": "healthy" if ntp_ok else "degraded",
        "ntp_offset_ms": offset,
        "ntp_synced": ntp_ok,
        "timestamp": time.time(),
    }

    status = 200 if ntp_ok else 503
    return HealthResponse(json.dumps(health).encode(), status)


def get_metrics_app():
    """Return an ASGI app serving /metrics and /health.

"""Alert dispatcher for Telegram + Email notifications.

Sends alerts on: kill switch, margin breach, API error, bot crash.
Per NIST RS.CO-2: Incident info shared.
Per ISO A.8.16: Monitoring activities.
Per SOC 2 CC7.2: Alerting.

Kleppmann Ch.8: Alerts must be actionable, not noisy.
  Only alert on conditions requiring human intervention.
"""

from __future__ import annotations

import enum
import json
import smtplib
from email.mime.multipart import MIMEMultipart
from email.mime.text import MIMEText
from typing import TYPE_CHECKING

import structlog

if TYPE_CHECKING:
    pass

logger = structlog.get_logger(__name__)


class AlertLevel(enum.Enum):
    INFO = "INFO"
    WARNING = "WARNING"
    CRITICAL = "CRITICAL"


class AlertDispatcher:
    """Dispatch alerts via Telegram and Email.

    Args:
        telegram_bot_token: Bot token from @BotFather.
        telegram_chat_id: Chat ID for alerts.
        smtp_host: SMTP server hostname.
        smtp_port: SMTP server port.
        smtp_user: SMTP authentication user.
        smtp_password: SMTP authentication password.
        email_from: Sender email address.
        email_to: Recipient email address.
    """

    def __init__(
        self,
        telegram_bot_token: str | None = None,
        telegram_chat_id: str | None = None,
        smtp_host: str | None = None,
        smtp_port: int = 587,
        smtp_user: str | None = None,
        smtp_password: str | None = None,
        email_from: str | None = None,
        email_to: str | None = None,
    ) -> None:
        self._telegram_bot_token = telegram_bot_token
        self._telegram_chat_id = telegram_chat_id
        self._smtp_host = smtp_host
        self._smtp_port = smtp_port
        self._smtp_user = smtp_user
        self._smtp_password = smtp_password
        self._email_from = email_from
        self._email_to = email_to

    async def send_alert(
        self,
        level: AlertLevel,
        title: str,
        message: str,
        details: dict | None = None,
    ) -> None:
        """Send alert via all configured channels.

        Args:
            level: Alert severity.
            title: Short alert title.
            message: Detailed alert message.
            details: Optional structured details.
        """
        emoji = {"INFO": "\u2139\ufe0f", "WARNING": "\u26a0\ufe0f", "CRITICAL": "\U0001f6d1"}

        formatted = (
            f"{emoji.get(level.value, '')} [{level.value}] {title}\n\n"
            f"{message}\n\n"
            f"Details: {json.dumps(details or {}, default=str, indent=2)}"
        )

        if self._telegram_bot_token and self._telegram_chat_id:
            await self._send_telegram(formatted)

        if self._smtp_host:
            self._send_email(
                subject=f"[{level.value}] {title}",
                body=formatted,
            )

        logger.info(
            "alert_dispatched",
            level=level.value,
            title=title,
            channels=["telegram" if self._telegram_bot_token else None, "email" if self._smtp_host else None],
        )

    async def _send_telegram(self, message: str) -> None:
        """Send message via Telegram Bot API."""
        try:
            import httpx

            url = f"https://api.telegram.org/bot{self._telegram_bot_token}/sendMessage"
            payload = {
                "chat_id": self._telegram_chat_id,
                "text": message,
                "parse_mode": "HTML",
            }
            async with httpx.AsyncClient(timeout=10.0) as client:
                response = await client.post(url, json=payload)
                response.raise_for_status()
        except Exception as e:
            logger.error("telegram_alert_failed", error=str(e))

    def _send_email(self, subject: str, body: str) -> None:
        """Send email via SMTP."""
        try:
            msg = MIMEMultipart()
            msg["From"] = self._email_from
            msg["To"] = self._email_to
            msg["Subject"] = subject
            msg.attach(MIMEText(body, "plain"))

            with smtplib.SMTP(self._smtp_host, self._smtp_port) as server:
                server.starttls()
                if self._smtp_user and self._smtp_password:
                    server.login(self._smtp_user, self._smtp_password)
                server.sendmail(self._email_from, self._email_to, msg.as_string())
        except Exception as e:
            logger.error("email_alert_failed", error=str(e))

"""Circuit breaker for broker API fault tolerance.

Kleppmann Ch.8: "A system that has to tolerate faults must detect them
  and then take action." Circuit breaker prevents cascading failures
  by failing fast when the broker API is degraded.

Implements three states: CLOSED (normal), OPEN (failing fast),
  HALF_OPEN (probing recovery).

Per NIST DE.CM-6: External service monitoring.
Per ISO A.8.20: Networks security.
"""

from __future__ import annotations

import time
from enum import Enum

import structlog

logger = structlog.get_logger(__name__)


class CircuitState(Enum):
    CLOSED = "CLOSED"
    OPEN = "OPEN"
    HALF_OPEN = "HALF_OPEN"


class CircuitBreaker:
    """Circuit breaker for broker API calls.

    Kleppmann Ch.8: Use timeouts + circuit breakers to handle
      partial failures in distributed systems.

    Args:
        failure_threshold: Consecutive failures before opening.
        recovery_timeout_seconds: Seconds before transitioning to HALF_OPEN.
        half_open_max_calls: Probes allowed in HALF_OPEN state.
    """

    def __init__(
        self,
        failure_threshold: int = 5,
        recovery_timeout_seconds: float = 30.0,
        half_open_max_calls: int = 1,
    ) -> None:
        self._failure_threshold = failure_threshold
        self._recovery_timeout = recovery_timeout_seconds
        self._half_open_max_calls = half_open_max_calls

        self._state = CircuitState.CLOSED
        self._failure_count = 0
        self._success_count = 0
        self._last_failure_time: float = 0.0
        self._half_open_calls = 0

    @property
    def state(self) -> CircuitState:
        """Current circuit breaker state, with automatic HALF_OPEN transition."""
        if self._state == CircuitState.OPEN:
            if time.monotonic() - self._last_failure_time >= self._recovery_timeout:
                self._state = CircuitState.HALF_OPEN
                self._half_open_calls = 0
                logger.info("circuit_breaker_half_open", message="Probing for recovery")
        return self._state

    def allow_request(self) -> bool:
        """Check if a request is allowed through the circuit breaker.

        Returns:
            True if request should proceed, False if should fail fast.
        """
        current_state = self.state
        if current_state == CircuitState.CLOSED:
            return True
        if current_state == CircuitState.HALF_OPEN:
            if self._half_open_calls < self._half_open_max_calls:
                self._half_open_calls += 1
                return True
            return False
        return False

    def record_success(self) -> None:
        """Record a successful request."""
        if self._state == CircuitState.HALF_OPEN:
            self._success_count += 1
            if self._success_count >= self._half_open_max_calls:
                self._state = CircuitState.CLOSED
                self._failure_count = 0
                self._success_count = 0
                logger.info("circuit_breaker_closed", message="API recovered, circuit closed")
        else:
            self._failure_count = 0

    def record_failure(self) -> None:
        """Record a failed request."""
        self._failure_count += 1
        self._last_failure_time = time.monotonic()
        self._success_count = 0

        if self._state == CircuitState.HALF_OPEN:
            self._state = CircuitState.OPEN
            logger.warning("circuit_breaker_open", reason="HALF_OPEN probe failed")
        elif self._failure_count >= self._failure_threshold:
            self._state = CircuitState.OPEN
            logger.critical(
                "circuit_breaker_open",
                failure_count=self._failure_count,
                threshold=self._failure_threshold,
                message="API degraded, circuit OPEN - failing fast",
            )

"""Retry with exponential backoff for broker API calls.

Kleppmann Ch.8: "Retrying is one of the simplest ways to tolerate
  transient faults." Use exponential backoff to avoid overwhelming
  a struggling service.

Per NIST SI-4: System monitoring (detect retried calls).
"""

from __future__ import annotations

import asyncio
import functools
import random
import time
from typing import Any, Callable, TypeVar

import structlog

logger = structlog.get_logger(__name__)

T = TypeVar("T")


def retry_with_backoff(
    max_retries: int = 3,
    base_delay_seconds: float = 1.0,
    max_delay_seconds: float = 30.0,
    jitter: bool = True,
    retryable_exceptions: tuple[type[Exception], ...] = (ConnectionError, TimeoutError, OSError),
) -> Callable:
    """Decorator for retry with exponential backoff.

    Kleppmann Ch.8: Add jitter to avoid thundering herd problem
      where all clients retry simultaneously.

    Args:
        max_retries: Maximum number of retry attempts.
        base_delay_seconds: Initial delay between retries.
        max_delay_seconds: Maximum delay between retries.
        jitter: Add random jitter to prevent thundering herd.
        retryable_exceptions: Exception types that should trigger retry.
    """

    def decorator(func: Callable) -> Callable:
        @functools.wraps(func)
        async def async_wrapper(*args: Any, **kwargs: Any) -> Any:
            last_exception: Exception | None = None
            for attempt in range(max_retries + 1):
                try:
                    return await func(*args, **kwargs)
                except retryable_exceptions as e:
                    last_exception = e
                    if attempt < max_retries:
                        delay = min(
                            base_delay_seconds * (2 ** attempt),
                            max_delay_seconds,
                        )
                        if jitter:
                            delay *= random.uniform(0.5, 1.5)
                        logger.warning(
                            "retry_attempt",
                            function=func.__name__,
                            attempt=attempt + 1,
                            max_retries=max_retries,
                            delay_seconds=round(delay, 3),
                            error=str(e),
                        )
                        await asyncio.sleep(delay)
                    else:
                        logger.error(
                            "retry_exhausted",
                            function=func.__name__,
                            attempts=max_retries + 1,
                            error=str(e),
                        )
            raise last_exception  # type: ignore[misc]

        @functools.wraps(func)
        def sync_wrapper(*args: Any, **kwargs: Any) -> Any:
            last_exception: Exception | None = None
            for attempt in range(max_retries + 1):
                try:
                    return func(*args, **kwargs)
                except retryable_exceptions as e:
                    last_exception = e
                    if attempt < max_retries:
                        delay = min(
                            base_delay_seconds * (2 ** attempt),
                            max_delay_seconds,
                        )
                        if jitter:
                            delay *= random.uniform(0.5, 1.5)
                        logger.warning(
                            "retry_attempt",
                            function=func.__name__,
                            attempt=attempt + 1,
                            max_retries=max_retries,
                            delay_seconds=round(delay, 3),
                            error=str(e),
                        )
                        time.sleep(delay)
                    else:
                        logger.error(
                            "retry_exhausted",
                            function=func.__name__,
                            attempts=max_retries + 1,
                            error=str(e),
                        )
            raise last_exception  # type: ignore[misc]

        if asyncio.iscoroutinefunction(func):
            return async_wrapper
        return sync_wrapper

    return decorator

                body = response.text
                if "trading_bot_" in body:
                    response.success()
                else:
                    response.failure("Missing trading_bot metrics")
            else:
                response.failure(f"Status {response.status_code}")
